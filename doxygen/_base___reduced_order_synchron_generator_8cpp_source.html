50 template <typename VarType>
51 void Base::ReducedOrderSynchronGenerator<VarType>::setBaseParameters(
52  Real nomPower, Real nomVolt, Real nomFreq) {
53 
55 
56  // set base nominal values
57  mNomPower = nomPower;
58  mNomVolt = nomVolt;
59  mNomFreq = nomFreq;
60  mNomOmega = nomFreq * 2 * PI;
61 
62  // Set base stator values
63  mBase_V_RMS = mNomVolt;
64  mBase_V = mNomVolt / sqrt(3) * sqrt(2);
65  mBase_I_RMS = mNomPower / mBase_V_RMS;
66  mBase_I = mNomPower / ((3./2.)* mBase_V);
67  mBase_Z = mBase_V_RMS / mBase_I_RMS;
68  mBase_OmElec = mNomOmega;
69  mBase_OmMech = mBase_OmElec;
70  mBase_L = mBase_Z / mBase_OmElec;
71 }

129 template <typename VarType>
130 void Base::ReducedOrderSynchronGenerator<VarType>::setInitialValues(
131  Complex initComplexElectricalPower, Real initMechanicalPower, Complex initTerminalVoltage) {
132 
133  mInitElecPower = initComplexElectricalPower;
134  mInitMechPower = initMechanicalPower;
135 
136  mInitVoltage = initTerminalVoltage;
137  mInitVoltageAngle = Math::phase(mInitVoltage);
138 
139  mInitCurrent = std::conj(mInitElecPower / mInitVoltage);
140  mInitCurrentAngle = Math::phase(mInitCurrent);
141 
142  mInitVoltage = mInitVoltage / mBase_V_RMS;
143  mInitCurrent = mInitCurrent / mBase_I_RMS;
144 
145  mInitialValuesSet = true;
146 }

148 template <>
149 void Base::ReducedOrderSynchronGenerator<Real>::initializeFromNodesAndTerminals(Real frequency) {
150 
151  this->updateMatrixNodeIndices();
152 
153  if(!mInitialValuesSet)
154  this->setInitialValues(-this->terminal(0)->singlePower(), mInitMechPower, this->initialSingleVoltage(0));
155 
156  // Initialize mechanical torque
157  mMechTorque = mInitMechPower / mNomPower;
158 
159  // calculate steady state machine emf (i.e. voltage behind synchronous reactance)
160  Complex Eq0 = mInitVoltage + Complex(0, mLq) * mInitCurrent;
161 
162  // Load angle
163  **mDelta = Math::phase(Eq0);
164 
165  // convert currrents to dq reference frame
166  (**mIdq0)(0,0) = Math::abs(mInitCurrent) * sin(**mDelta - mInitCurrentAngle);
167  (**mIdq0)(1,0) = Math::abs(mInitCurrent) * cos(**mDelta - mInitCurrentAngle);
168 
169  // convert voltages to dq reference frame
170  (**mVdq0)(0,0) = Math::abs(mInitVoltage) * sin(**mDelta - mInitVoltageAngle);
171  (**mVdq0)(1,0) = Math::abs(mInitVoltage) * cos(**mDelta - mInitVoltageAngle);
172 
173  // calculate Ef
174  mEf = Math::abs(Eq0) + (mLd - mLq) * (**mIdq0)(0,0);
175 
176  // initial electrical torque
177  **mElecTorque = (**mVdq0)(0,0) * (**mIdq0)(0,0) + (**mVdq0)(1,0) * (**mIdq0)(1,0);
178 
179  // Initialize omega mech with nominal system frequency
180  **mOmMech = mNomOmega / mBase_OmMech;
181 
182  // initialize theta and calculate transform matrix
183  **mThetaMech = **mDelta - PI / 2.;
184 
185  mSLog->info(
186  "\n--- Initialization from power flow ---"
187  "\nInitial Vd (per unit): {:f}"
188  "\nInitial Vq (per unit): {:f}"
189  "\nInitial Id (per unit): {:f}"
190  "\nInitial Iq (per unit): {:f}"
191  "\nInitial Ef (per unit): {:f}"
192  "\nInitial mechanical torque (per unit): {:f}"
193  "\nInitial electrical torque (per unit): {:f}"
194  "\nInitial initial mechanical theta (per unit): {:f}"
195  "\nInitial delta (per unit): {:f}"
196  "\n--- Initialization from power flow finished ---",
197 
198  (**mVdq0)(0,0),
199  (**mVdq0)(1,0),
200  (**mIdq0)(0,0),
201  (**mIdq0)(1,0),
202  mEf,
203  mMechTorque,
204  **mElecTorque,
205  **mThetaMech,
206  **mDelta
207  );
208  mSLog->flush();
209 }
