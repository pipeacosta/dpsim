49 PyObject* Python::Component::getattro(Python::Component* self, PyObject* name)
50 {
51  PyObject *attr;
52 
53  // Check is there is already an attibute with this name
54  attr = PyObject_GenericGetAttr((PyObject *) self, name);
55  if (!attr) {
56  if (!PyErr_ExceptionMatches(PyExc_AttributeError))
57  return NULL;
58 
59  PyErr_Clear();
60  }
61  else
62  return attr;
63 
64  if (!self->comp) {
65  PyErr_SetString(PyExc_ValueError, "getattr on unitialized Component");
66  return nullptr;
67  }
68 
69  try {
70  auto attr = self->comp->attribute(PyUnicode_AsUTF8(name));
71 
72  return attr->toPyObject();
73  }
74  catch (const CPS::InvalidAttributeException &) {
75  PyErr_Format(PyExc_AttributeError, "Component has no attribute '%s'", PyUnicode_AsUTF8(name));
76  return nullptr;
77  }
78  catch (...) {
79  PyErr_Format(PyExc_RuntimeError, "Unkown Error Occured", name);
80  return nullptr;
81  }

85 {

90 
91  try {
92  auto attr = self->comp->attribute(PyUnicode_AsUTF8(name));
93  attr->fromPyObject(v);

96  PyErr_Format(PyExc_AttributeError, "Component has no attribute '%s'", PyUnicode_AsUTF8(name));

99  catch (const CPS::TypeException &) {

101  return -1;
102  }
103  catch (const CPS::AccessException &) {
104  PyErr_Format(PyExc_AttributeError, "Attribute '%s' is not modifiable", PyUnicode_AsUTF8(name));
105  return -1;
106  }
107  catch (...) {
108  PyErr_Format(PyExc_RuntimeError, "Unkown Error Occured", PyUnicode_AsUTF8(name));
109  return -1;
110  }
111 
112  return 0;

115 CPS::IdentifiedObject::List Python::compsFromPython(PyObject* list)
116 {
117  CPS::IdentifiedObject::List comps;
118 
119  if (!PyList_Check(list))
120  throw std::invalid_argument("argument must be a list");
121 
122  for (int i = 0; i < PyList_Size(list); i++) {
123  PyObject* obj = PyList_GetItem(list, i);
124  if (!PyObject_TypeCheck(obj, &Python::Component::type))
125  throw std::invalid_argument( "list element is not a dpsim.Component" );
126 
127  Component* pyComp = (Component*) obj;
128  comps.push_back(pyComp->comp);
129  }
130 
131  return comps;
132 }

135 PyObject* Python::Component::connect(Component* self, PyObject* args)
136 {
137  PyObject *pyNodes;
138 
139  if (!PyArg_ParseTuple(args, "O", &pyNodes))
140  return nullptr;
141 
142  try {
143  using EMTComponent = CPS::SimPowerComp<CPS::Real>;
144  using DPComponent = CPS::SimPowerComp<CPS::Complex>;
145 
146  if (auto emtComp = std::dynamic_pointer_cast<EMTComponent>(self->comp)) {
147  auto nodes = Python::Node<CPS::Real>::fromPython(pyNodes);
148 
149  emtComp->connect(nodes);
150 
151  }
152  else if (auto dpComp = std::dynamic_pointer_cast<DPComponent>(self->comp)) {
153  auto nodes = Python::Node<CPS::Complex>::fromPython(pyNodes);
154 
155  dpComp->connect(nodes);
156  }
157  else {
158  PyErr_SetString(PyExc_TypeError, "Failed to connect nodes");
159  return nullptr;
160  }
161 
162  Py_RETURN_NONE;
163  } catch (...) {
164  PyErr_SetString(PyExc_TypeError, "Failed to connect nodes");
165  return nullptr;
166  }
