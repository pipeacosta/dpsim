152 void DP::Ph1::AvVoltageSourceInverterDQ::initializeFromNodesAndTerminals(Real frequency) {
153 
154  // set initial interface quantities
155  mIntfVoltage(0, 0) = initialSingleVoltage(0);
156  mIntfCurrent(0, 0) = - std::conj(Complex(mPref, mQref) / mIntfVoltage(0,0));
157 
158  Complex filterInterfaceInitialVoltage;
159  Complex filterInterfaceInitialCurrent;
160 
161  if (mWithConnectionTransformer) {
162  // calculate quantities of low voltage side of transformer (being the interface quantities of the filter)
163  filterInterfaceInitialVoltage = (mIntfVoltage(0, 0) - Complex(mTransformerResistance, mTransformerInductance*mOmegaN)*mIntfCurrent(0, 0)) / Complex(mTransformerRatioAbs, mTransformerRatioPhase);
164  filterInterfaceInitialCurrent = mIntfCurrent(0, 0) * Complex(mTransformerRatioAbs, mTransformerRatioPhase);
165 
166  // connect transformer
167  mVirtualNodes[3]->setInitialVoltage(filterInterfaceInitialVoltage);
168  mConnectionTransformer->connect({ mTerminals[0]->node(), mVirtualNodes[3] });
169  } else {
170  // if no transformer used, filter interface equal to inverter interface
171  filterInterfaceInitialVoltage = mIntfVoltage(0, 0);
172  filterInterfaceInitialCurrent = mIntfCurrent(0, 0);
173  }
174 
175  // derive initialization quantities of filter
176  Complex vcInit = filterInterfaceInitialVoltage - filterInterfaceInitialCurrent * mRc;
177  Complex icfInit = vcInit * Complex(0., 2. * PI * frequency * mCf);
178  Complex vfInit = vcInit - (filterInterfaceInitialCurrent - icfInit) * Complex(0., 2. * PI * frequency * mLf);
179  Complex vsInit = vfInit - (filterInterfaceInitialCurrent - icfInit) * Complex(mRf, 0);
180  mVirtualNodes[0]->setInitialVoltage(vsInit);
181  mVirtualNodes[1]->setInitialVoltage(vfInit);
182  mVirtualNodes[2]->setInitialVoltage(vcInit);
183 
184  // Set parameters electrical subcomponents
185  mVsref(0,0) = mVirtualNodes[0]->initialSingleVoltage();
186  mSubCtrledVoltageSource->setParameters(mVsref(0,0));
187 
188  // Connect electrical subcomponents
189  mSubCtrledVoltageSource->connect({ SimNode::GND, mVirtualNodes[0] });
190  mSubResistorF->connect({ mVirtualNodes[0], mVirtualNodes[1] });
191  mSubInductorF->connect({ mVirtualNodes[1], mVirtualNodes[2] });
192  mSubCapacitorF->connect({ mVirtualNodes[2], SimNode::GND });
193  if (mWithConnectionTransformer)
194  mSubResistorC->connect({ mVirtualNodes[2], mVirtualNodes[3]});
195  else
196  mSubResistorC->connect({ mVirtualNodes[2], mTerminals[0]->node()});
197 
198  // Initialize electrical subcomponents
199  for (auto subcomp: mSubComponents) {
200  subcomp->initialize(mFrequencies);
201  subcomp->initializeFromNodesAndTerminals(frequency);
202  }
203 
204  // Initialize control subcomponents
205  // current and voltage inputs to PLL and power controller
206  Complex vcdq, ircdq;
207  vcdq = Math::rotatingFrame2to1(mVirtualNodes[3]->initialSingleVoltage(), std::arg(mVirtualNodes[3]->initialSingleVoltage()), 0);
208  ircdq = Math::rotatingFrame2to1(-1. * mSubResistorC->attribute<MatrixComp>("i_intf")->get()(0, 0), std::arg(mVirtualNodes[3]->initialSingleVoltage()), 0);
209  mVcd = vcdq.real();
210  mVcq = vcdq.imag();
211  mIrcd = ircdq.real();
212  mIrcq = ircdq.imag();
213  // angle input
214  Matrix matrixStateInit = Matrix::Zero(2,1);
215  matrixStateInit(0,0) = std::arg(mVirtualNodes[3]->initialSingleVoltage());
216  mPLL->setInitialValues(mVcq, matrixStateInit, Matrix::Zero(2,1));
217 
218  mSLog->info(
219  "\n--- Initialization from powerflow ---"
220  "\nInterface voltage across: {:s}"
221  "\nInterface current: {:s}"
222  "\nTerminal 0 initial voltage: {:s}"
223  "\nTerminal 0 connected to {:s} = sim node {:d}"
224  "\nVirtual node 0 initial voltage: {:s}"
225  "\nVirtual node 1 initial voltage: {:s}"
226  "\nVirtual node 2 initial voltage: {:s}",
227  Logger::phasorToString(mIntfVoltage(0, 0)),
228  Logger::phasorToString(mIntfCurrent(0, 0)),
229  Logger::phasorToString(initialSingleVoltage(0)),
230  mTerminals[0]->node()->name(), mTerminals[0]->node()->matrixNodeIndex(),
231  Logger::phasorToString(mVirtualNodes[0]->initialSingleVoltage()),
232  Logger::phasorToString(mVirtualNodes[1]->initialSingleVoltage()),
233  Logger::phasorToString(mVirtualNodes[2]->initialSingleVoltage()));
234  if (mWithConnectionTransformer)
235  mSLog->info("\nVirtual node 3 initial voltage: {:s}", Logger::phasorToString(mVirtualNodes[3]->initialSingleVoltage()));
236  mSLog->info("\n--- Initialization from powerflow finished ---");
237 }

239 void DP::Ph1::AvVoltageSourceInverterDQ::mnaInitialize(Real omega, Real timeStep, Attribute<Matrix>::Ptr leftVector) {
240  MNAInterface::mnaInitialize(omega, timeStep);
241  updateMatrixNodeIndices();
242  mTimeStep = timeStep;
243 
244  // initialize electrical subcomponents with MNA interface
245  for (auto subcomp: mSubComponents)
246  if (auto mnasubcomp = std::dynamic_pointer_cast<MNAInterface>(subcomp))
247  mnasubcomp->mnaInitialize(omega, timeStep, leftVector);
248 
249  // initialize state space controller
250  mPowerControllerVSI->initializeStateSpaceModel(omega, timeStep, leftVector);
251  mPLL->setSimulationParameters(timeStep);
252 
253  // collect right side vectors of subcomponents
254  mRightVectorStamps.push_back(&mSubCapacitorF->attribute<Matrix>("right_vector")->get());
255  mRightVectorStamps.push_back(&mSubInductorF->attribute<Matrix>("right_vector")->get());
256  mRightVectorStamps.push_back(&mSubCtrledVoltageSource->attribute<Matrix>("right_vector")->get());
257  if (mWithConnectionTransformer)
258  mRightVectorStamps.push_back(&mConnectionTransformer->attribute<Matrix>("right_vector")->get());
259 
260  // collect tasks
261  mMnaTasks.push_back(std::make_shared<MnaPreStep>(*this));
262  mMnaTasks.push_back(std::make_shared<MnaPostStep>(*this, leftVector));
263 
264  // TODO: these are actually no MNA tasks
265  mMnaTasks.push_back(std::make_shared<ControlPreStep>(*this));
266  mMnaTasks.push_back(std::make_shared<ControlStep>(*this));
267 
268  mRightVector = Matrix::Zero(leftVector->get().rows(), 1);
269 }

306 void DP::Ph1::AvVoltageSourceInverterDQ::controlStep(Real time, Int timeStepCount) {
307  // Transformation interface forward
308  Complex vcdq, ircdq;
309  vcdq = Math::rotatingFrame2to1(mVirtualNodes[3]->singleVoltage(), mPLL->attribute<Matrix>("output_prev")->get()(0, 0), mThetaN);
310  ircdq = Math::rotatingFrame2to1(-1. * mSubResistorC->attribute<MatrixComp>("i_intf")->get()(0, 0), mPLL->attribute<Matrix>("output_prev")->get()(0, 0), mThetaN);
311  mVcd = vcdq.real();
312  mVcq = vcdq.imag();
313  mIrcd = ircdq.real();
314  mIrcq = ircdq.imag();
315 
316  // add step of subcomponents
317  mPLL->signalStep(time, timeStepCount);
318  mPowerControllerVSI->signalStep(time, timeStepCount);
319 
320  // Transformation interface backward
321  mVsref(0,0) = Math::rotatingFrame2to1(Complex(mPowerControllerVSI->attribute<Matrix>("output_curr")->get()(0, 0), mPowerControllerVSI->attribute<Matrix>("output_curr")->get()(1, 0)), mThetaN, mPLL->attribute<Matrix>("output_prev")->get()(0, 0));
322 
323  // Update nominal system angle
324  mThetaN = mThetaN + mTimeStep * mOmegaN;
325 }
