38 void DP::Ph1::Capacitor::initializeFromNodesAndTerminals(Real frequency) {
39 
40  Real omega = 2 * PI * frequency;
41  Complex impedance = { 0, - 1. / (omega * **mCapacitance) };
42  (**mIntfVoltage)(0,0) = initialSingleVoltage(1) - initialSingleVoltage(0);
43  (**mIntfCurrent)(0,0) = (**mIntfVoltage)(0,0) / impedance;

45  mSLog->info("\nCapacitance [F]: {:s}"

51  "\nVoltage across: {:s}"
52  "\nCurrent: {:s}"
53  "\nTerminal 0 voltage: {:s}"
54  "\nTerminal 1 voltage: {:s}"
55  "\n--- Initialization from powerflow finished ---",

57  Logger::phasorToString((**mIntfCurrent)(0,0)),

62 void DP::Ph1::Capacitor::mnaInitialize(Real omega, Real timeStep, Attribute<Matrix>::Ptr leftVector) {
63  MNAInterface::mnaInitialize(omega, timeStep);
64  updateMatrixNodeIndices();
65 
66  Real equivCondReal = 2.0 * **mCapacitance / timeStep;
67  Real prevVoltCoeffReal = 2.0 * **mCapacitance / timeStep;
68 
69  for (UInt freq = 0; freq < mNumFreqs; freq++) {
70  Real equivCondImag = 2.*PI * mFrequencies(freq,0) * **mCapacitance;
71  mEquivCond(freq,0) = { equivCondReal, equivCondImag };
72  Real prevVoltCoeffImag = - 2.*PI * mFrequencies(freq,0) * **mCapacitance;
73  mPrevVoltCoeff(freq,0) = { prevVoltCoeffReal, prevVoltCoeffImag };
74 
75  mEquivCurrent(freq,0) = -(**mIntfCurrent)(0,freq) + -mPrevVoltCoeff(freq,0) * (**mIntfVoltage)(0,freq);
76  (**mIntfCurrent)(0, freq) = mEquivCond(freq,0) * (**mIntfVoltage)(0,freq) + mEquivCurrent(freq,0);
77  }
78 
79  **mRightVector = Matrix::Zero(leftVector->get().rows(), 1);
80  mMnaTasks.push_back(std::make_shared<MnaPreStep>(*this));
81  mMnaTasks.push_back(std::make_shared<MnaPostStep>(*this, leftVector));
82 
83  mSLog->info(
84  "\n--- MNA initialization ---"
85  "\nInitial voltage {:s}"
86  "\nInitial current {:s}"
87  "\nEquiv. current {:s}"
88  "\n--- MNA initialization finished ---",
89  Logger::phasorToString((**mIntfVoltage)(0,0)),
90  Logger::phasorToString((**mIntfCurrent)(0,0)),
91  Logger::complexToString(mEquivCurrent(0,0)));
92 }

116 void DP::Ph1::Capacitor::mnaApplySystemMatrixStamp(Matrix& systemMatrix) {
117  for (UInt freq = 0; freq < mNumFreqs; freq++) {
118  if (terminalNotGrounded(0))
119  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0), matrixNodeIndex(0), mEquivCond(freq,0), mNumFreqs, freq);
120  if (terminalNotGrounded(1))
121  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1), matrixNodeIndex(1), mEquivCond(freq,0), mNumFreqs, freq);
122  if (terminalNotGrounded(0) && terminalNotGrounded(1)) {
123  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0), matrixNodeIndex(1), -mEquivCond(freq,0), mNumFreqs, freq);
124  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1), matrixNodeIndex(0), -mEquivCond(freq,0), mNumFreqs, freq);
125  }
126 
127  mSLog->info("-- Stamp frequency {:d} ---", freq);
128  if (terminalNotGrounded(0))
129  mSLog->info("Add {:e}+j{:e} to system at ({:d},{:d})",
130  mEquivCond(freq,0).real(), mEquivCond(freq,0).imag(), matrixNodeIndex(0), matrixNodeIndex(0));
131  if (terminalNotGrounded(1))
132  mSLog->info("Add {:e}+j{:e} to system at ({:d},{:d})",
133  mEquivCond(freq,0).real(), mEquivCond(freq,0).imag(), matrixNodeIndex(1), matrixNodeIndex(1));
134  if (terminalNotGrounded(0) && terminalNotGrounded(1)) {
135  mSLog->info("Add {:e}+j{:e} to system at ({:d},{:d})",
136  -mEquivCond(freq,0).real(), -mEquivCond(freq,0).imag(), matrixNodeIndex(0), matrixNodeIndex(1));
137  mSLog->info("Add {:e}+j{:e} to system at ({:d},{:d})",
138  -mEquivCond(freq,0).real(), -mEquivCond(freq,0).imag(), matrixNodeIndex(1), matrixNodeIndex(0));
139  }
140  }

168 void DP::Ph1::Capacitor::mnaApplyRightSideVectorStamp(Matrix& rightVector) {
169  for (UInt freq = 0; freq < mNumFreqs; freq++) {
170  //mCureqr = mCurrr + mGcr * mDeltavr + mGci * mDeltavi;
171  //mCureqi = mCurri + mGcr * mDeltavi - mGci * mDeltavr;
172  mEquivCurrent(freq,0) = -(**mIntfCurrent)(0,freq)
173  + -mPrevVoltCoeff(freq,0) * (**mIntfVoltage)(0,freq);
174 
175  if (terminalNotGrounded(0))
176  Math::setVectorElement(rightVector, matrixNodeIndex(0), mEquivCurrent(freq,0), mNumFreqs, freq);
177  if (terminalNotGrounded(1))
178  Math::setVectorElement(rightVector, matrixNodeIndex(1), -mEquivCurrent(freq,0), mNumFreqs, freq);
179 
180  SPDLOG_LOGGER_DEBUG(mSLog, "MNA EquivCurrent {:f}+j{:f}",
181  mEquivCurrent(freq,0).real(), mEquivCurrent(freq,0).imag());
182  if (terminalNotGrounded(0))
183  SPDLOG_LOGGER_DEBUG(mSLog, "Add {:f}+j{:f} to source vector at {:d}",
184  mEquivCurrent(freq,0).real(), mEquivCurrent(freq,0).imag(), matrixNodeIndex(0));
185  if (terminalNotGrounded(1))
186  SPDLOG_LOGGER_DEBUG(mSLog, "Add {:f}+j{:f} to source vector at {:d}",
187  -mEquivCurrent(freq,0).real(), -mEquivCurrent(freq,0).imag(), matrixNodeIndex(1));
188  }

249 void DP::Ph1::Capacitor::mnaUpdateVoltage(const Matrix& leftVector) {
250  // v1 - v0
251  for (UInt freq = 0; freq < mNumFreqs; freq++) {
252  (**mIntfVoltage)(0,freq) = 0;
253  if (terminalNotGrounded(1))
254  (**mIntfVoltage)(0,freq) = Math::complexFromVectorElement(leftVector, matrixNodeIndex(1), mNumFreqs, freq);
255  if (terminalNotGrounded(0))

257 
258  SPDLOG_LOGGER_DEBUG(mSLog, "Voltage {:e}<{:e}", std::abs((**mIntfVoltage)(0,freq)), std::arg((**mIntfVoltage)(0,freq)));
259  }
