83  Real besselFirstKind_n(Int n, Int k_max, Real x) {
84  Real Jn = 0;
85  for (Int k = 0; k <= k_max; ++k) {
86  Real Jn_k = pow(-1,k) / factorial(k) * multInvFactorial(k+n) * pow(x/2., 2.*k+n);
87  Jn = Jn + Jn_k;
88  //mSLog->info("Jn_n = {:f}", Jn_n);
89  }
90  return Jn;
91  }

94  Real besselFirstKind_n_opt(Int n, Int k_max, Real x) {
95  Real Jn = 0;
96  for (Int k = 0; k <= k_max; ++k) {
97  Real Jn_k = pow(-1,k) / mFactorials[k] * mMultInvFactorials[k+n] * pow(x/2., 2.*k+n);
98  Jn = Jn + Jn_k;
99  }
100  return Jn;
101  }
