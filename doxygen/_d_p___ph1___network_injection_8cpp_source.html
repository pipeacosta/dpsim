99 void DP::Ph1::NetworkInjection::mnaInitialize(Real omega, Real timeStep, Attribute<Matrix>::Ptr leftVector) {
100  MNAInterface::mnaInitialize(omega, timeStep);
101  updateMatrixNodeIndices();
102 
103  // initialize electrical subcomponents
104  for (auto subcomp: mSubComponents)
105  if (auto mnasubcomp = std::dynamic_pointer_cast<MNAInterface>(subcomp))
106  mnasubcomp->mnaInitialize(omega, timeStep, leftVector);
107 
108  // collect right side vectors of subcomponents
110  mRightVectorStamps.push_back(&**(mSubVoltageSource->mRightVector));
111 
112  // collect tasks
113  mMnaTasks.push_back(std::make_shared<MnaPreStep>(*this));
114  mMnaTasks.push_back(std::make_shared<MnaPostStep>(*this, leftVector));
115 
116  **mRightVector = Matrix::Zero(leftVector->get().rows(), 1);
117 }

183 void DP::Ph1::NetworkInjection::daeResidual(double ttime, const double state[], const double dstate_dt[], double resid[], std::vector<int>& off){
184  /* new state vector definintion:
185  state[0]=node0_voltage
186  state[1]=node1_voltage
187  ....
188  state[n]=noden_voltage
189  state[n+1]=component0_voltage
190  state[n+2]=component0_inductance (not yet implemented)
191  ...
192  state[m-1]=componentm_voltage
193  state[m]=componentm_inductance
194  */
195 
196  int Pos1 = matrixNodeIndex(0);
197  int Pos2 = matrixNodeIndex(1);
198  int c_offset = off[0]+off[1]; //current offset for component
199  int n_offset_1 = c_offset + Pos1 +1;// current offset for first nodal equation
200  int n_offset_2 = c_offset + Pos2 +1;// current offset for second nodal equation
201  resid[c_offset] = (state[Pos2]-state[Pos1]) - state[c_offset]; // Voltage equation for Resistor
202  //resid[++c_offset] = ; //TODO : add inductance equation
203  resid[n_offset_1] += (**mIntfCurrent)(0, 0).real();
204  resid[n_offset_2] += (**mIntfCurrent)(0, 0).real();
205  off[1] += 1;
206 }
