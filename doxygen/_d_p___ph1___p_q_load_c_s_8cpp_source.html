56 void DP::Ph1::PQLoadCS::initializeFromNodesAndTerminals(Real frequency) {
57  // Get power from Terminals if it was not set previously.
58  if (**mActivePower == 0 && **mReactivePower == 0 && !mParametersSet) {
59  **mActivePower = mTerminals[0]->singleActivePower();
60  **mReactivePower = mTerminals[0]->singleReactivePower();
61  **mNomVoltage = std::abs(mTerminals[0]->initialSingleVoltage());
62  }
63  Complex power = Complex(**mActivePower, **mReactivePower);
64 
65  Complex current;
67  if (**mNomVoltage != 0)
68  current = std::conj(power / **mNomVoltage);
69  else
70  current = 0;
71 
72  mSubCurrentSource = std::make_shared<DP::Ph1::CurrentSource>(**mName + "_cs", mLogLevel);
73  mSubCurrentSource->setParameters(current);
74  // A positive power should result in a positive current to ground.
75  mSubCurrentSource->connect({ mTerminals[0]->node(), SimNode::GND });
76  mSubCurrentSource->initializeFromNodesAndTerminals(frequency);
77  updateIntfValues();
78 
79  mSLog->info(
80  "\n--- Initialization from powerflow ---"
81  "\nVoltage across: {:s}"
82  "\nCurrent: {:s}"
83  "\nTerminal 0 voltage: {:s}"
84  "\nCurrent set point: {:s}"
85  "\n--- Initialization from powerflow finished ---",
86  Logger::phasorToString((**mIntfVoltage)(0,0)),
87  Logger::phasorToString((**mIntfCurrent)(0,0)),
88  Logger::phasorToString(initialSingleVoltage(0)),
89  Logger::phasorToString(current));
90 }

112 void DP::Ph1::PQLoadCS::updateSetPoint() {
113  // Calculate new current set point.
114  Complex power = { **mActivePower, **mReactivePower};
115  Complex current = power / **mNomVoltage;
116  //Complex current = power / (**mIntfVoltage)(0,0);
117 
118  **mSubCurrentSource->mCurrentRef = std::conj(current);
119  mSLog->debug(
120  "\n--- update set points ---"
121  "\npower: {:s}"
122  "\nCurrent: {:s}",
123  Logger::phasorToString(power),
124  Logger::phasorToString(std::conj(current)));
125 }
