32 void DP::Ph1::RXLoadSwitch::initializeFromNodesAndTerminals(Real frequency) {

50  mIntfVoltage = mSubRXLoad->attributeMatrixComp("v_intf")->get();
51  mIntfCurrent = mSubRXLoad->attributeMatrixComp("i_intf")->get();

145 void DP::Ph1::RXLoadSwitch::updateSwitchState(Real time) {
146 
147  if (time > mSwitchTimeOffset && mSubSwitch->isClosed()) {
148  Real VRef = Math::abs(mSubRXLoad->attributeComplex("V_nom")->get());
149  Real V = Math::abs(mIntfVoltage(0, 0));
150 
151  Real deltaV = Math::abs((V - VRef) / VRef);
152 
153  if (deltaV > 0.1) {
154  mSubSwitch->open();
155  mSLog->info("Opened Switch at {}", (float)time);
156  }
157  }
158 }
