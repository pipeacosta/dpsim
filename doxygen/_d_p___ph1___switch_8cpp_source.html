56 void DP::Ph1::Switch::mnaApplySystemMatrixStamp(Matrix& systemMatrix) {
57  Complex conductance = (**mIsClosed) ?
58  Complex( 1. / **mClosedResistance, 0 ) : Complex( 1. / **mOpenResistance, 0 );
59 
60  // Set diagonal entries
61  if (terminalNotGrounded(0))
62  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0), matrixNodeIndex(0), conductance);
63  if (terminalNotGrounded(1))
64  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1), matrixNodeIndex(1), conductance);
65  // Set off diagonal entries
66  if (terminalNotGrounded(0) && terminalNotGrounded(1)) {
67  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0), matrixNodeIndex(1), -conductance);
68  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1), matrixNodeIndex(0), -conductance);
69  }
70 
71  mSLog->info("-- Stamp ---");
72  if (terminalNotGrounded(0))
73  mSLog->info("Add {:s} to system at ({:d},{:d})", Logger::complexToString(conductance), matrixNodeIndex(0), matrixNodeIndex(0));
74  if (terminalNotGrounded(1))
75  mSLog->info("Add {:s} to system at ({:d},{:d})", Logger::complexToString(conductance), matrixNodeIndex(1), matrixNodeIndex(1));
76  if (terminalNotGrounded(0) && terminalNotGrounded(1)) {
77  mSLog->info("Add {:s} to system at ({:d},{:d})", Logger::complexToString(-conductance), matrixNodeIndex(0), matrixNodeIndex(1));
78  mSLog->info("Add {:s} to system at ({:d},{:d})", Logger::complexToString(-conductance), matrixNodeIndex(1), matrixNodeIndex(0));
79  }
80 }

112 void DP::Ph1::Switch::mnaUpdateVoltage(const Matrix& leftVector) {
113  // Voltage across component is defined as V1 - V0
114  (**mIntfVoltage)(0, 0) = 0;
115  if (terminalNotGrounded(1)) (**mIntfVoltage)(0,0) = Math::complexFromVectorElement(leftVector, matrixNodeIndex(1,0));
116  if (terminalNotGrounded(0)) (**mIntfVoltage)(0,0) = (**mIntfVoltage)(0,0) - Math::complexFromVectorElement(leftVector, matrixNodeIndex(0));
117 }
