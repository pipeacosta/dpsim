45 void DP::Ph1::SynchronGeneratorVBR::calculateConductanceMatrix() {
46  Matrix resistanceMatrix = Matrix::Zero(2,2);
47  resistanceMatrix(0,0) = mR_const_1ph.real() + mKa_1ph.real() + mKb_1ph.real();
48  resistanceMatrix(0,1) = -mR_const_1ph.imag() - mKa_1ph.imag() + mKb_1ph.imag();
49  resistanceMatrix(1,0) = mR_const_1ph.imag() + mKa_1ph.imag() + mKb_1ph.imag();
50  resistanceMatrix(1,1) = mR_const_1ph.real() + mKa_1ph.real() - mKb_1ph.real();
51  resistanceMatrix = resistanceMatrix * mBase_Z;
52  mConductanceMatrix = resistanceMatrix.inverse();
53 }

55 void DP::Ph1::SynchronGeneratorVBR::calculateAuxiliarVariables() {
56  mKa = Matrix::Zero(1,3);
57  mKa = mKc * Complex(cos(2. * mThetaMech), sin(2. * mThetaMech));
58  mKa_1ph = (mKa * mShiftVector)(0,0);
59 
60  mKb = Matrix::Zero(1,3);
61  Real arg = 2. * mThetaMech - 2. * mBase_OmMech * mSimTime ;
62  mKb = mKc * Complex(cos(arg), sin(arg));
63  mKb_1ph = (mKb * mShiftVectorConj)(0,0);
64 
65  mKvbr = Matrix::Zero(1,2);
66  mKvbr(0,0) = Complex(cos(mThetaMech - mBase_OmMech * mSimTime), sin(mThetaMech - mBase_OmMech * mSimTime));
67  mKvbr(0,1) = -Complex(cos(mThetaMech - mBase_OmMech * mSimTime - PI/2.), sin(mThetaMech - mBase_OmMech * mSimTime - PI/2.));
68 }

92 void DP::Ph1::SynchronGeneratorVBR::mnaPostStep(const Matrix& leftVector) {
93  // update armature voltage and current
94  mIntfVoltage(0, 0) = Math::complexFromVectorElement(leftVector, matrixNodeIndex(0, 0));
95  mIntfCurrent(0, 0) = Math::complexFromVectorElement(leftVector, mVirtualNodes[1]->matrixNodeIndex());
96 
97  // convert armature voltage into dq reference frame
98  Matrix parkTransform = get_parkTransformMatrix();
99  MatrixComp Vabc_ = mIntfVoltage(0, 0) * mShiftVector * Complex(cos(mNomOmega * mSimTime), sin(mNomOmega * mSimTime));
100  Matrix Vabc = Matrix(3,1);
101  Vabc << Vabc_(0,0).real(), Vabc_(1,0).real(), Vabc_(2,0).real();
102  mVdq = parkTransform * Vabc / mBase_V_RMS;
103 
104  // convert armature current into dq reference frame
105  MatrixComp Iabc_ = mIntfCurrent(0, 0) * mShiftVector * Complex(cos(mNomOmega * mSimTime), sin(mNomOmega * mSimTime));
106  Matrix Iabc = Matrix(3,1);
107  Iabc << Iabc_(0,0).real(), Iabc_(1,0).real(), Iabc_(2,0).real();
108  mIdq = parkTransform * Iabc / mBase_I_RMS;
109 }

111 Matrix DP::Ph1::SynchronGeneratorVBR::get_parkTransformMatrix() {
