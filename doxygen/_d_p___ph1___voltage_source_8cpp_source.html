66 void DP::Ph1::VoltageSource::initializeFromNodesAndTerminals(Real frequency) {
67  Complex voltageRef = attribute<Complex>("V_ref")->get();
68 
69  if (voltageRef == Complex(0, 0))
70  voltageRef = initialSingleVoltage(1) - initialSingleVoltage(0);
71 
72  if (mSrcSig == nullptr) {
73  Signal::SineWaveGenerator srcSigSine(mName);
74  srcSigSine.setParameters(voltageRef);
75  mSrcSig = std::make_shared<Signal::SineWaveGenerator>(srcSigSine);
76 
77  setAttributeRef("V_ref", mSrcSig->attribute<Complex>("sigOut"));
78  setAttributeRef("f_src", mSrcSig->attribute<Real>("freq"));
79  } else {
80  attribute<Complex>("V_ref")->set(voltageRef);
81  }
82 
83  mSLog->info(
84  "\n--- Initialization from node voltages ---"
85  "\nVoltage across: {:s}"
86  "\nTerminal 0 voltage: {:s}"
87  "\nTerminal 1 voltage: {:s}"
88  "\n--- Initialization from node voltages ---",
89  Logger::phasorToString(mSrcSig->getSignal()),
90  Logger::phasorToString(initialSingleVoltage(0)),
91  Logger::phasorToString(initialSingleVoltage(1)));
92 }

136 void DP::Ph1::VoltageSource::mnaApplySystemMatrixStamp(Matrix& systemMatrix) {
137  for (UInt freq = 0; freq < mNumFreqs; freq++) {
138  if (terminalNotGrounded(0)) {
139  Math::setMatrixElement(systemMatrix, mVirtualNodes[0]->matrixNodeIndex(), matrixNodeIndex(0), Complex(-1, 0), mNumFreqs, freq);
140  Math::setMatrixElement(systemMatrix, matrixNodeIndex(0), mVirtualNodes[0]->matrixNodeIndex(), Complex(-1, 0), mNumFreqs, freq);
141  }
142  if (terminalNotGrounded(1)) {
143  Math::setMatrixElement(systemMatrix, mVirtualNodes[0]->matrixNodeIndex(), matrixNodeIndex(1), Complex(1, 0), mNumFreqs, freq);
144  Math::setMatrixElement(systemMatrix, matrixNodeIndex(1), mVirtualNodes[0]->matrixNodeIndex(), Complex(1, 0), mNumFreqs, freq);
145  }
146 
147  mSLog->info("-- Stamp frequency {:d} ---", freq);
148  if (terminalNotGrounded(0)) {
149  mSLog->info("Add {:f} to system at ({:d},{:d})", -1., matrixNodeIndex(0), mVirtualNodes[0]->matrixNodeIndex());
150  mSLog->info("Add {:f} to system at ({:d},{:d})", -1., mVirtualNodes[0]->matrixNodeIndex(), matrixNodeIndex(0));
151  }
152  if (terminalNotGrounded(1)) {
153  mSLog->info("Add {:f} to system at ({:d},{:d})", 1., mVirtualNodes[0]->matrixNodeIndex(), matrixNodeIndex(1));

231 void DP::Ph1::VoltageSource::daeResidual(double ttime, const double state[], const double dstate_dt[], double resid[], std::vector<int>& off){
232  /* new state vector definintion:
233  state[0]=node0_voltage
234  state[1]=node1_voltage
235  ....
236  state[n]=noden_voltage
237  state[n+1]=component0_voltage
238  state[n+2]=component0_inductance (not yet implemented)
239  ...
240  state[m-1]=componentm_voltage
241  state[m]=componentm_inductance
242  */
243 
244  int Pos1 = matrixNodeIndex(0);
245  int Pos2 = matrixNodeIndex(1);
246  int c_offset = off[0]+off[1]; //current offset for component
247  int n_offset_1 = c_offset + Pos1 +1;// current offset for first nodal equation
248  int n_offset_2 = c_offset + Pos2 +1;// current offset for second nodal equation
249  resid[c_offset] = (state[Pos2]-state[Pos1]) - state[c_offset]; // Voltage equation for Resistor
250  //resid[++c_offset] = ; //TODO : add inductance equation
251  resid[n_offset_1] += mIntfCurrent(0, 0).real();
252  resid[n_offset_2] += mIntfCurrent(0, 0).real();
253  off[1] += 1;
254 }
