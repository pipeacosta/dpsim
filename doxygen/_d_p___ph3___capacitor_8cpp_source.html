34 void DP::Ph3::Capacitor::initializeFromNodesAndTerminals(Real frequency) {
35 
36  Real omega = 2 * PI * frequency;
37  MatrixComp susceptance = Matrix::Zero(3, 3);
38 
39  susceptance <<
40  Complex(0, omega * (**mCapacitance)(0, 0)), Complex(0, omega * (**mCapacitance)(0, 1)), Complex(0, omega * (**mCapacitance)(0, 2)),
41  Complex(0, omega * (**mCapacitance)(1, 0)), Complex(0, omega * (**mCapacitance)(1, 1)), Complex(0, omega * (**mCapacitance)(1, 2)),
42  Complex(0, omega * (**mCapacitance)(2, 0)), Complex(0, omega * (**mCapacitance)(2, 1)), Complex(0, omega * (**mCapacitance)(2, 2));
43 
44 
45  // IntfVoltage initialization for each phase
46  (**mIntfVoltage)(0, 0) = initialSingleVoltage(1) - initialSingleVoltage(0);
47  Real voltMag = Math::abs((**mIntfVoltage)(0, 0));
48  Real voltPhase = Math::phase((**mIntfVoltage)(0, 0));
49  (**mIntfVoltage)(1, 0) = Complex(
50  voltMag*cos(voltPhase - 2. / 3.*M_PI),
51  voltMag*sin(voltPhase - 2. / 3.*M_PI));
52  (**mIntfVoltage)(2, 0) = Complex(
53  voltMag*cos(voltPhase + 2. / 3.*M_PI),
54  voltMag*sin(voltPhase + 2. / 3.*M_PI));
55 
56  **mIntfCurrent = susceptance * **mIntfVoltage;
57 
58  mSLog->info( "\n--- Initialize from power flow ---" );
59  // << "Impedance: " << impedance << std::endl
60  // << "Voltage across: " << std::abs((**mIntfVoltage)(0,0))
61  // << "<" << Math::phaseDeg((**mIntfVoltage)(0,0)) << std::endl
62  // << "Current: " << std::abs((**mIntfCurrent)(0,0))
63  // << "<" << Math::phaseDeg((**mIntfCurrent)(0,0)) << std::endl
64  // << "Terminal 0 voltage: " << std::abs(initialSingleVoltage(0))
65  // << "<" << Math::phaseDeg(initialSingleVoltage(0)) << std::endl
66  // << "Terminal 1 voltage: " << std::abs(initialSingleVoltage(1))
67  // << "<" << Math::phaseDeg(initialSingleVoltage(1)) << std::endl
68  // << "--- Power flow initialization finished ---" << std::endl;
69 }

72 void DP::Ph3::Capacitor::initVars(Real omega, Real timeStep) {
73  Matrix a = timeStep / 2 * (**mCapacitance).inverse();
74  Real b = timeStep * omega / 2.;
75 
76  Matrix equivCondReal = a.inverse();
77  Matrix equivCondImag = b * equivCondReal;
78  mEquivCond = Matrix::Zero(3, 3);
79  mEquivCond <<
80  Complex(equivCondReal(0, 0), equivCondImag(0, 0)), Complex(equivCondReal(0, 1), equivCondImag(0, 1)), Complex(equivCondReal(0, 2), equivCondImag(0, 2)),
81  Complex(equivCondReal(1, 0), equivCondImag(1, 0)), Complex(equivCondReal(1, 1), equivCondImag(1, 1)), Complex(equivCondReal(1, 2), equivCondImag(1, 2)),
82  Complex(equivCondReal(2, 0), equivCondImag(2, 0)), Complex(equivCondReal(2, 1), equivCondImag(2, 1)), Complex(equivCondReal(2, 2), equivCondImag(2, 2));
83 
84  // since equivCondReal == a.inverse()
85  // and
86  /*Matrix mPrevVoltCoeffReal = a.inverse();
87  Matrix mPrevVoltCoeffImag = -b * a.inverse();*/
88  Matrix mPrevVoltCoeffReal = equivCondReal;
89  Matrix mPrevVoltCoeffImag = -b * equivCondReal;
90 
91  mPrevVoltCoeff = Matrix::Zero(3, 3);
92  mPrevVoltCoeff <<
93  Complex(mPrevVoltCoeffReal(0, 0), mPrevVoltCoeffImag(0, 0)), Complex(mPrevVoltCoeffReal(0, 1), mPrevVoltCoeffImag(0, 1)), Complex(mPrevVoltCoeffReal(0, 2), mPrevVoltCoeffImag(0, 2)),
94  Complex(mPrevVoltCoeffReal(1, 0), mPrevVoltCoeffImag(1, 0)), Complex(mPrevVoltCoeffReal(1, 1), mPrevVoltCoeffImag(1, 1)), Complex(mPrevVoltCoeffReal(1, 2), mPrevVoltCoeffImag(1, 2)),
95  Complex(mPrevVoltCoeffReal(2, 0), mPrevVoltCoeffImag(2, 0)), Complex(mPrevVoltCoeffReal(2, 1), mPrevVoltCoeffImag(2, 1)), Complex(mPrevVoltCoeffReal(2, 2), mPrevVoltCoeffImag(2, 2));
96 
97 
98  mEquivCurrent = -mPrevVoltCoeff * **mIntfVoltage - **mIntfCurrent;
99 }

229 void DP::Ph3::Capacitor::mnaUpdateVoltage(const Matrix& leftVector) {
