31 void DP::Ph3::Inductor::initializeFromNodesAndTerminals(Real frequency) {
32 
33  Real omega = 2 * PI * frequency;
34 
35  MatrixComp reactance = MatrixComp::Zero(3, 3);
36  reactance <<
37  Complex(0, omega * mInductance(0, 0)), Complex(0, omega * mInductance(0, 1)), Complex(0, omega * mInductance(0, 2)),
38  Complex(0, omega * mInductance(1, 0)), Complex(0, omega * mInductance(1, 1)), Complex(0, omega * mInductance(1, 2)),
39  Complex(0, omega * mInductance(2, 0)), Complex(0, omega * mInductance(2, 1)), Complex(0, omega * mInductance(2, 2));
40  MatrixComp susceptance = reactance.inverse();
41  // IntfVoltage initialization for each phase
42  mIntfVoltage(0, 0) = initialSingleVoltage(1) - initialSingleVoltage(0);
43  Real voltMag = Math::abs(mIntfVoltage(0, 0));
44  Real voltPhase = Math::phase(mIntfVoltage(0, 0));
45  mIntfVoltage(1, 0) = Complex(
46  voltMag*cos(voltPhase - 2. / 3.*M_PI),
47  voltMag*sin(voltPhase - 2. / 3.*M_PI));
48  mIntfVoltage(2, 0) = Complex(
49  voltMag*cos(voltPhase + 2. / 3.*M_PI),
50  voltMag*sin(voltPhase + 2. / 3.*M_PI));
51 
52  mIntfCurrent = susceptance * mIntfVoltage;
53 
54  //TODO
55  mSLog->info( "--- Initialize according to power flow ---" );
56  // << "in phase A: " << std::endl
57  // << "Voltage across: " << std::abs(mIntfVoltage(0,0))
58  // << "<" << Math::phaseDeg(mIntfVoltage(0,0)) << std::endl
59  // << "Current: " << std::abs(mIntfCurrent(0,0))
60  // << "<" << Math::phaseDeg(mIntfCurrent(0,0)) << std::endl
61  // << "Terminal 0 voltage: " << std::abs(initialSingleVoltage(0))
62  // << "<" << Math::phaseDeg(initialSingleVoltage(0)) << std::endl
63  // << "Terminal 1 voltage: " << std::abs(initialSingleVoltage(1))
64  // << "<" << Math::phaseDeg(initialSingleVoltage(1)) << std::endl
65  // << "--- Power flow initialization finished ---" << std::endl;
66 }

68 void DP::Ph3::Inductor::initVars(Real omega, Real timeStep) {
69  Matrix a = timeStep / 2. * mInductance.inverse();
70  Real b = timeStep * omega / 2.;
71 
72  Matrix equivCondReal = a / (1. + b * b);
73  Matrix equivCondImag = -a * b / (Real(1.) + b * b);
74  mEquivCond = MatrixComp::Zero(3, 3);
75  mEquivCond <<
76  Complex(equivCondReal(0, 0), equivCondImag(0, 0)), Complex(equivCondReal(0, 1), equivCondImag(0, 1)), Complex(equivCondReal(0, 2), equivCondImag(0, 2)),
77  Complex(equivCondReal(1, 0), equivCondImag(1, 0)), Complex(equivCondReal(1, 1), equivCondImag(1, 1)), Complex(equivCondReal(1, 2), equivCondImag(1, 2)),
78  Complex(equivCondReal(2, 0), equivCondImag(2, 0)), Complex(equivCondReal(2, 1), equivCondImag(2, 1)), Complex(equivCondReal(2, 2), equivCondImag(2, 2));
79 
80  Real preCurrFracReal = (1. - b * b) / (1. + b * b);
81  Real preCurrFracImag = (-2. * b) / (1. + b * b);
82  mPrevCurrFac = Complex(preCurrFracReal, preCurrFracImag);
83 
84 
85  // TODO: check if this is correct or if it should be only computed before the step
86  mEquivCurrent = mEquivCond * mIntfVoltage + mPrevCurrFac * mIntfCurrent;
87  // no need to update this now
88  //mIntfCurrent = mEquivCond.cwiseProduct(mIntfVoltage) + mEquivCurrent;
89 }

162

187 void DP::Ph3::Inductor::mnaUpdateVoltage(const Matrix& leftVector) {
