29 void DP::Ph3::SeriesResistor::initializeFromNodesAndTerminals(Real frequency) {
30 
31  mTerminals[0]->setPhaseType(PhaseType::ABC);
32  mTerminals[1]->setPhaseType(PhaseType::ABC);
33 
34  Matrix impedance = Matrix::Zero(3, 1);
35  impedance <<
36  **mResistance,
37  **mResistance,
38  **mResistance;
39  (**mIntfVoltage)(0, 0) = initialSingleVoltage(1) - initialSingleVoltage(0);
40 
41  Real voltMag = Math::abs((**mIntfVoltage)(0, 0));
42  Real voltPhase = Math::phase((**mIntfVoltage)(0, 0));
43  (**mIntfVoltage)(1, 0) = Complex(
44  voltMag * cos(voltPhase - 2. / 3. * PI),
45  voltMag * sin(voltPhase - 2. / 3. * PI));
46  (**mIntfVoltage)(2, 0) = Complex(
47  voltMag * cos(voltPhase + 2. / 3. * PI),
48  voltMag * sin(voltPhase + 2. / 3. * PI));
49 
50  **mIntfCurrent = impedance.cwiseInverse().cwiseProduct(**mIntfVoltage);
51 
52  mSLog->info("\n--- Initialization from powerflow ---"
53  "\nVoltage across amplitude and phase: \n{:s}"
54  "\nCurrent amplitude and phase: \n{:s}"
55  "\nTerminal 0 voltage amplitude and phase: \n{:s}"
56  "\nTerminal 1 voltage amplitude and phase: \n{:s}"
57  "\n--- Initialization from powerflow finished ---",
58  Logger::phasorMatrixToString(**mIntfVoltage),
59  Logger::phasorMatrixToString(**mIntfCurrent),
60  Logger::phasorMatrixToString(initialVoltage(0)),
61  Logger::phasorMatrixToString(initialVoltage(1)));
62 }

114 void DP::Ph3::SeriesResistor::mnaUpdateVoltage(const Matrix& leftVector) {
115  // Voltage across component is defined as V1 - V0
116  **mIntfVoltage = MatrixComp::Zero(3,1);
117  if (terminalNotGrounded(1)) {
118  (**mIntfVoltage)(0,0) = Math::complexFromVectorElement(leftVector, matrixNodeIndex(1,0));
119  (**mIntfVoltage)(1,0) = Math::complexFromVectorElement(leftVector, matrixNodeIndex(1,1));
120  (**mIntfVoltage)(2,0) = Math::complexFromVectorElement(leftVector, matrixNodeIndex(1,2));
121  }
122  if (terminalNotGrounded(0)) {
123  (**mIntfVoltage)(0,0) = (**mIntfVoltage)(0,0) - Math::complexFromVectorElement(leftVector, matrixNodeIndex(0,0));
124  (**mIntfVoltage)(1,0) = (**mIntfVoltage)(1,0) - Math::complexFromVectorElement(leftVector, matrixNodeIndex(0,1));
125  (**mIntfVoltage)(2,0) = (**mIntfVoltage)(2,0) - Math::complexFromVectorElement(leftVector, matrixNodeIndex(0,2));
126  }
127 
128  SPDLOG_LOGGER_DEBUG(mSLog, "Voltage A: {} < {}", std::abs((**mIntfVoltage)(0,0)), std::arg((**mIntfVoltage)(0,0)));
129 }
