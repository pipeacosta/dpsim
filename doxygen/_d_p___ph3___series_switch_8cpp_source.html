48 void DP::Ph3::SeriesSwitch::mnaApplySystemMatrixStamp(Matrix& systemMatrix) {
49  Complex conductance = (**mIsClosed)
50  ? Complex( 1. / **mClosedResistance, 0 )
51  : Complex( 1. / **mOpenResistance, 0 );
52 
53  // Set diagonal entries
54  if (terminalNotGrounded(0))
55  Math::addToMatrixElement(systemMatrix, matrixNodeIndices(0), matrixNodeIndices(0), conductance);
56  if (terminalNotGrounded(1))
57  Math::addToMatrixElement(systemMatrix, matrixNodeIndices(1), matrixNodeIndices(1), conductance);
58  // Set off diagonal entries
59  if (terminalNotGrounded(0) && terminalNotGrounded(1)) {
60  Math::addToMatrixElement(systemMatrix, matrixNodeIndices(0), matrixNodeIndices(1), -conductance);
61  Math::addToMatrixElement(systemMatrix, matrixNodeIndices(1), matrixNodeIndices(0), -conductance);
62  }
63 
64  if (terminalNotGrounded(0))
65  mSLog->info("Add {} to {}, {}", conductance, matrixNodeIndices(0)[0], matrixNodeIndices(0)[0]);
66  if (terminalNotGrounded(1))
67  mSLog->info("Add {} to {}, {}", conductance, matrixNodeIndices(1)[0], matrixNodeIndices(1)[0]);
68  if (terminalNotGrounded(0) && terminalNotGrounded(1)) {
69  mSLog->info("Add {} to {}, {}", -conductance, matrixNodeIndices(0)[0], matrixNodeIndices(1)[0]);
70  mSLog->info("Add {} to {}, {}", -conductance, matrixNodeIndices(1)[0], matrixNodeIndices(0)[0]);
71  }
72 }

105 void DP::Ph3::SeriesSwitch::mnaUpdateVoltage(const Matrix& leftVector) {
106  // Voltage across component is defined as V1 - V0
107  **mIntfVoltage = MatrixComp::Zero(3,1);
108  if (terminalNotGrounded(1)) {
109  (**mIntfVoltage)(0,0) = Math::complexFromVectorElement(leftVector, matrixNodeIndex(1,0));
110  (**mIntfVoltage)(1,0) = Math::complexFromVectorElement(leftVector, matrixNodeIndex(1,1));
111  (**mIntfVoltage)(2,0) = Math::complexFromVectorElement(leftVector, matrixNodeIndex(1,2));
112  }
113  if (terminalNotGrounded(0)) {
114  (**mIntfVoltage)(0,0) = (**mIntfVoltage)(0,0) - Math::complexFromVectorElement(leftVector, matrixNodeIndex(0,0));
115  (**mIntfVoltage)(1,0) = (**mIntfVoltage)(1,0) - Math::complexFromVectorElement(leftVector, matrixNodeIndex(0,1));
116  (**mIntfVoltage)(2,0) = (**mIntfVoltage)(2,0) - Math::complexFromVectorElement(leftVector, matrixNodeIndex(0,2));
117  }
118 
119  SPDLOG_LOGGER_DEBUG(mSLog, "Voltage A: {} < {}", std::abs((**mIntfVoltage)(0,0)), std::arg((**mIntfVoltage)(0,0)));
120 }
