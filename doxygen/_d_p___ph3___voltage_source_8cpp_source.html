47 void DP::Ph3::VoltageSource::mnaInitialize(Real omega, Real timeStep, Attribute<Matrix>::Ptr leftVector) {
48  updateMatrixNodeIndices();
49  (**mIntfVoltage)(0, 0) = **mVoltageRef;
50  (**mIntfVoltage)(1, 0) = Complex(Math::abs(**mVoltageRef) * cos(Math::phase(**mVoltageRef) - 2. / 3. * M_PI),
51  Math::abs(**mVoltageRef) * sin(Math::phase(**mVoltageRef) - 2. / 3. * M_PI));
52  (**mIntfVoltage)(2, 0) = Complex(Math::abs(**mVoltageRef) * cos(Math::phase(**mVoltageRef) + 2. / 3. * M_PI),
53  Math::abs(**mVoltageRef) * sin(Math::phase(**mVoltageRef) + 2. / 3. * M_PI));
54 
55  mMnaTasks.push_back(std::make_shared<MnaPreStep>(*this));
56  mMnaTasks.push_back(std::make_shared<MnaPostStep>(*this, leftVector));
57  **mRightVector = Matrix::Zero(leftVector->get().rows(), 1);
58 }
