29 void EMT::Ph1::Capacitor::initializeFromNodesAndTerminals(Real frequency) {
30 
31  Real omega = 2 * PI * frequency;
32  Complex impedance = { 0, - 1. / (omega * mCapacitance) };
33  mIntfVoltage(0,0) = (initialSingleVoltage(1) - initialSingleVoltage(0)).real();
34  mIntfCurrent(0,0) = ((initialSingleVoltage(1) - initialSingleVoltage(0)) / impedance).real();
35 
36  mSLog->info(
37  "\n--- Initialization from powerflow ---"
38  "\nVoltage across: {:f}"
39  "\nCurrent: {:f}"
40  "\nTerminal 0 voltage: {:f}"
41  "\nTerminal 1 voltage: {:f}"
42  "\n--- Initialization from powerflow finished ---",
43  mIntfVoltage(0,0),
44  mIntfCurrent(0,0),
45  initialSingleVoltage(0).real(),
46  initialSingleVoltage(1).real());
47 }

49 void EMT::Ph1::Capacitor::mnaInitialize(Real omega, Real timeStep, Attribute<Matrix>::Ptr leftVector) {
50  MNAInterface::mnaInitialize(omega, timeStep);
51  updateMatrixNodeIndices();
52 
53  mEquivCond = (2.0 * mCapacitance) / timeStep;
54  // Update internal state
55  mEquivCurrent = -mIntfCurrent(0,0) + -mEquivCond * mIntfVoltage(0,0);
56 
57  mRightVector = Matrix::Zero(leftVector->get().rows(), 1);
58  mMnaTasks.push_back(std::make_shared<MnaPreStep>(*this));
59  mMnaTasks.push_back(std::make_shared<MnaPostStep>(*this, leftVector));
60 }

73 void EMT::Ph1::Capacitor::mnaApplyRightSideVectorStamp(Matrix& rightVector) {
