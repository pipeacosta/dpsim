52 void EMT::Ph1::CurrentSource::updateState(Real time) {
53  Complex currentRef = mCurrentRef->get();
54  Real srcFreq = mSrcFreq->get();
55  if (srcFreq > 0)
56  (**mIntfCurrent)(0,0) = Math::abs(currentRef) * cos(time * 2.*PI*srcFreq + Math::phase(currentRef));
57  else
58  (**mIntfCurrent)(0,0) = currentRef.real();
59 }
