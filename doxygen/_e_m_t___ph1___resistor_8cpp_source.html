54 void EMT::Ph1::Resistor::mnaApplySystemMatrixStamp(Matrix& systemMatrix) {
55  Real conductance = 1. / **mResistance;
56  // Set diagonal entries
57  if (terminalNotGrounded(0))
58  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0), matrixNodeIndex(0), conductance);
59  if (terminalNotGrounded(1))
60  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1), matrixNodeIndex(1), conductance);
61  // Set off diagonal entries
62  if (terminalNotGrounded(0) && terminalNotGrounded(1)) {
63  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0), matrixNodeIndex(1), -conductance);
64  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1), matrixNodeIndex(0), -conductance);
65  }
66 
67  if (terminalNotGrounded(0))
68  mSLog->info("Add {:f} to system at ({:d},{:d})", conductance, matrixNodeIndex(0), matrixNodeIndex(0));
69  if (terminalNotGrounded(1))
70  mSLog->info("Add {:f} to system at ({:d},{:d})", conductance, matrixNodeIndex(1), matrixNodeIndex(1));
71  if ( terminalNotGrounded(0) && terminalNotGrounded(1) ) {
72  mSLog->info("Add {:f} to system at ({:d},{:d})", -conductance, matrixNodeIndex(0), matrixNodeIndex(1));
73  mSLog->info("Add {:f} to system at ({:d},{:d})", -conductance, matrixNodeIndex(1), matrixNodeIndex(0));
74  }
75 }
