48 void EMT::Ph1::VoltageSourceNorton::mnaApplySystemMatrixStamp(Matrix& systemMatrix) {
49  // Apply matrix stamp for equivalent resistance
50  if (terminalNotGrounded(0))
51  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0), matrixNodeIndex(0), mConductance);
52  if (terminalNotGrounded(1))
53  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1), matrixNodeIndex(1), mConductance);
54  if (terminalNotGrounded(0) && terminalNotGrounded(1)) {
55  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0), matrixNodeIndex(1), -mConductance);

68 void EMT::Ph1::VoltageSourceNorton::updateState(Real time) {
69  // Check if set source was called
70  if (Math::abs(mVoltageRef) > 0)
71  mIntfVoltage(0,0) = Math::abs(mVoltageRef) * cos(2.*PI*mSrcFreq*time + Math::phase(mVoltageRef));
