49 void EMT::Ph1::VoltageSourceNorton::mnaApplySystemMatrixStamp(Matrix& systemMatrix) {
50  // Apply matrix stamp for equivalent resistance
51  if (terminalNotGrounded(0))
52  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0), matrixNodeIndex(0), mConductance);
53  if (terminalNotGrounded(1))
54  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1), matrixNodeIndex(1), mConductance);
55  if (terminalNotGrounded(0) && terminalNotGrounded(1)) {

69 void EMT::Ph1::VoltageSourceNorton::updateState(Real time) {
70  // Check if set source was called
71  if (Math::abs(**mVoltageRef) > 0)
72  (**mIntfVoltage)(0,0) = Math::abs(**mVoltageRef) * cos(2.*PI* **mSrcFreq*time + Math::phase(**mVoltageRef));
