79 void EMT::Ph1::VoltageSourceRamp::updateState(Real time) {
80  Real voltageAbs = Math::abs(**mVoltageRef);
81  Real voltagePhase = Math::phase(**mVoltageRef);
82  (**mIntfVoltage)(0,0) = voltageAbs * cos(voltagePhase + **mSrcFreq * time);
83 
84  if (time >= mSwitchTime && time < mSwitchTime + mRampTime) {
85  voltageAbs = Math::abs(**mVoltageRef + (time - mSwitchTime) / mRampTime * mAddVoltage);
86  voltagePhase = Math::phase(**mVoltageRef + (time - mSwitchTime) / mRampTime * mAddVoltage);
87  Real fadeInOut = 0.5 + 0.5 * sin((time - mSwitchTime) / mRampTime * PI + -PI / 2);
88  (**mIntfVoltage)(0,0) = voltageAbs * cos(voltagePhase + (**mSrcFreq + fadeInOut * mAddSrcFreq) * time);
89  }
90  else if (time >= mSwitchTime + mRampTime) {
91  voltageAbs = Math::abs(**mVoltageRef + mAddVoltage);
92  voltagePhase = Math::phase(**mVoltageRef + mAddVoltage);
93  (**mIntfVoltage)(0,0) = voltageAbs * cos(voltagePhase + (**mSrcFreq + mAddSrcFreq) * time);
94  }
95 }
