28 void EMT::Ph3::CurrentSource::initializeFromNodesAndTerminals(Real frequency) {
29  mSLog->info("\n--- Initialization from node voltages and terminal ---");
30  if (!mParametersSet) {
31  auto srcSigSine = Signal::SineWaveGenerator::make(mName + "_sw", Logger::Level::off);
32  // Complex(1,0) is used as initialPhasor for signal generator as only phase is used
33  srcSigSine->setParameters(Complex(1,0), frequency);
34  mSrcSig = srcSigSine;
35 
36  Complex v_ref = initialSingleVoltage(1) - initialSingleVoltage(0);
37  Complex s_ref = terminal(1)->singlePower() - terminal(0)->singlePower();
38 
39  // Current flowing from T1 to T0 (rms value)
40  Complex i_ref = std::conj(s_ref/v_ref/sqrt(3.));
41 
42  attribute<MatrixComp>("I_ref")->set(CPS::Math::singlePhaseVariableToThreePhase(i_ref));
43  setAttributeRef("f_src", mSrcSig->attribute<Real>("freq"));
44 
45  mSLog->info("\nReference current: {:s}"
46  "\nReference voltage: {:s}"
47  "\nReference power: {:s}"
48  "\nTerminal 0 voltage: {:s}"
49  "\nTerminal 1 voltage: {:s}"
50  "\nTerminal 0 power: {:s}"
51  "\nTerminal 1 power: {:s}",
52  Logger::phasorToString(i_ref),
53  Logger::phasorToString(v_ref),
54  Logger::complexToString(s_ref),
55  Logger::phasorToString(initialSingleVoltage(0)),
56  Logger::phasorToString(initialSingleVoltage(1)),
57  Logger::complexToString(terminal(0)->singlePower()),
58  Logger::complexToString(terminal(1)->singlePower()));
59  } else {
60  mSLog->info("\nInitialization from node voltages and terminal omitted (parameter already set)."
61  "\nReference voltage: {:s}",
62  Logger::matrixCompToString(attribute<MatrixComp>("I_ref")->get()));
63  }
64  mSLog->info("\n--- Initialization from node voltages and terminal ---");
65  mSLog->flush();
66 }

88 void EMT::Ph3::CurrentSource::mnaApplyRightSideVectorStamp(Matrix& rightVector) {
89  if (terminalNotGrounded(1)) {
90  Math::setVectorElement(rightVector, matrixNodeIndex(1, 0), - mIntfCurrent(0, 0));
91  Math::setVectorElement(rightVector, matrixNodeIndex(1, 1), - mIntfCurrent(1, 0));
92  Math::setVectorElement(rightVector, matrixNodeIndex(1, 2), - mIntfCurrent(2, 0));

95  Math::setVectorElement(rightVector, matrixNodeIndex(0, 0), mIntfCurrent(0, 0));
96  Math::setVectorElement(rightVector, matrixNodeIndex(0, 1), mIntfCurrent(1, 0));
97  Math::setVectorElement(rightVector, matrixNodeIndex(0, 2), mIntfCurrent(2, 0));

101 void EMT::Ph3::CurrentSource::updateCurrent(Real time) {
102  if(mSrcSig != nullptr) {
103  mSrcSig->step(time);
104  for(int i = 0; i < 3; i++) {
105  mIntfCurrent(i, 0) = RMS_TO_PEAK * Math::abs(attribute<MatrixComp>("I_ref")->get()(i, 0))
106  * cos(Math::phase(mSrcSig->getSignal()) + Math::phase(attribute<MatrixComp>("I_ref")->get()(i, 0)));
107  }
108  } else {
109  mIntfCurrent = RMS_TO_PEAK * attribute<MatrixComp>("I_ref")->get().real();
110  }
111  mSLog->debug(
112  "\nUpdate current: {:s}",
113  Logger::matrixToString(mIntfCurrent)
114  );
115 }

137 void EMT::Ph3::CurrentSource::mnaUpdateVoltage(const Matrix& leftVector) {
138  // v1 - v0
139  mIntfVoltage = Matrix::Zero(3,1);
140  if (terminalNotGrounded(1)) {
141  mIntfVoltage(0, 0) = Math::realFromVectorElement(leftVector, matrixNodeIndex(1, 0));
142  mIntfVoltage(1, 0) = Math::realFromVectorElement(leftVector, matrixNodeIndex(1, 1));
143  mIntfVoltage(2, 0) = Math::realFromVectorElement(leftVector, matrixNodeIndex(1, 2));
144  }
145  if (terminalNotGrounded(0)) {
146  mIntfVoltage(0, 0) = mIntfVoltage(0, 0) - Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 0));
147  mIntfVoltage(1, 0) = mIntfVoltage(1, 0) - Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 1));
148  mIntfVoltage(2, 0) = mIntfVoltage(2, 0) - Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 2));
149  }
