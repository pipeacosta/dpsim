107 void EMT::Ph3::NetworkInjection::mnaInitialize(Real omega, Real timeStep, Attribute<Matrix>::Ptr leftVector) {
108  MNAInterface::mnaInitialize(omega, timeStep);
109  updateMatrixNodeIndices();
110 
111  // initialize electrical subcomponents
112  for (auto subcomp: mSubComponents)
113  if (auto mnasubcomp = std::dynamic_pointer_cast<MNAInterface>(subcomp))
114  mnasubcomp->mnaInitialize(omega, timeStep, leftVector);
115 
116  // collect right side vectors of subcomponents
117  mRightVectorStamps.push_back(&**mSubVoltageSource->mRightVector);
118 
119  // collect tasks
120  mMnaTasks.push_back(std::make_shared<MnaPreStep>(*this));
121  mMnaTasks.push_back(std::make_shared<MnaPostStep>(*this, leftVector));
122 
123  **mRightVector = Matrix::Zero(leftVector->get().rows(), 1);
124 }
