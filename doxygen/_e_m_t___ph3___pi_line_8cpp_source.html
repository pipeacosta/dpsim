36 void EMT::Ph3::PiLine::initializeFromNodesAndTerminals(Real frequency) {
37 
38  // By default there is always a small conductance to ground to
39  // avoid problems with floating nodes.
40  Matrix defaultParallelCond = Matrix::Zero(3, 3);
41  defaultParallelCond <<
42  1e-6, 0, 0,
43  0, 1e-6, 0,
44  0, 0, 1e-6;
45  mParallelCond = (mParallelCond(0, 0) > 0) ? mParallelCond : defaultParallelCond;
46 
47  // Static calculation
48  Real omega = 2. * PI * frequency;
49  MatrixComp impedance = MatrixComp::Zero(3, 3);
50  impedance <<
51  Complex(mSeriesRes(0, 0), omega * mSeriesInd(0, 0)), Complex(mSeriesRes(0, 1), omega * mSeriesInd(0, 1)), Complex(mSeriesRes(0, 2), omega * mSeriesInd(0, 2)),
52  Complex(mSeriesRes(1, 0), omega * mSeriesInd(1, 0)), Complex(mSeriesRes(1, 1), omega * mSeriesInd(1, 1)), Complex(mSeriesRes(1, 2), omega * mSeriesInd(1, 2)),
53  Complex(mSeriesRes(2, 0), omega * mSeriesInd(2, 0)), Complex(mSeriesRes(2, 1), omega * mSeriesInd(2, 1)), Complex(mSeriesRes(2, 2), omega * mSeriesInd(2, 2));
54 
55  MatrixComp vInitABC = MatrixComp::Zero(3, 1);
56  vInitABC(0, 0) = RMS3PH_TO_PEAK1PH * initialSingleVoltage(1) - RMS3PH_TO_PEAK1PH * initialSingleVoltage(0);
57  vInitABC(1, 0) = vInitABC(0, 0) * SHIFT_TO_PHASE_B;
58  vInitABC(2, 0) = vInitABC(0, 0) * SHIFT_TO_PHASE_C;
59  MatrixComp iInit = impedance.inverse() * vInitABC;
60  mIntfCurrent = iInit.real();
61  mIntfVoltage = vInitABC.real();
62 
63  // Initialization of virtual node
64  // Initial voltage of phase B,C is set after A
65  MatrixComp vInitTerm0 = MatrixComp::Zero(3, 1);
66  vInitTerm0(0, 0) = RMS3PH_TO_PEAK1PH * initialSingleVoltage(0);
67  vInitTerm0(1, 0) = vInitTerm0(0, 0) * SHIFT_TO_PHASE_B;
68  vInitTerm0(2, 0) = vInitTerm0(0, 0) * SHIFT_TO_PHASE_C;
69 
70  mVirtualNodes[0]->setInitialVoltage(PEAK1PH_TO_RMS3PH*(vInitTerm0 + mSeriesRes * iInit));
71 
72  // Create series sub components
73  mSubSeriesResistor = std::make_shared<EMT::Ph3::Resistor>(mName + "_res", mLogLevel);
74  mSubSeriesResistor->setParameters(mSeriesRes);
75  mSubSeriesResistor->connect({ mTerminals[0]->node(), mVirtualNodes[0] });
76  mSubSeriesResistor->initialize(mFrequencies);
77  mSubSeriesResistor->initializeFromNodesAndTerminals(frequency);
78 
79  mSubSeriesInductor = std::make_shared<EMT::Ph3::Inductor>(mName + "_ind", mLogLevel);
80  mSubSeriesInductor->setParameters(mSeriesInd);
81  mSubSeriesInductor->connect({ mVirtualNodes[0], mTerminals[1]->node() });
82  mSubSeriesInductor->initialize(mFrequencies);
83  mSubSeriesInductor->initializeFromNodesAndTerminals(frequency);
84 
85  // Create parallel sub components
86  mSubParallelResistor0 = std::make_shared<EMT::Ph3::Resistor>(mName + "_con0", mLogLevel);
87  mSubParallelResistor0->setParameters(2. * mParallelCond.inverse());
88  mSubParallelResistor0->connect(SimNode::List{ SimNode::GND, mTerminals[0]->node() });
89  mSubParallelResistor0->initialize(mFrequencies);
90  mSubParallelResistor0->initializeFromNodesAndTerminals(frequency);
91 
92  mSubParallelResistor1 = std::make_shared<EMT::Ph3::Resistor>(mName + "_con1", mLogLevel);
93  mSubParallelResistor1->setParameters(2. * mParallelCond.inverse());
94  mSubParallelResistor1->connect(SimNode::List{ SimNode::GND, mTerminals[1]->node() });
95  mSubParallelResistor1->initialize(mFrequencies);
96  mSubParallelResistor1->initializeFromNodesAndTerminals(frequency);
97 
98  if (mParallelCap(0,0) > 0) {
99  mSubParallelCapacitor0 = std::make_shared<EMT::Ph3::Capacitor>(mName + "_cap0", mLogLevel);
100  mSubParallelCapacitor0->setParameters(mParallelCap / 2.);
101  mSubParallelCapacitor0->connect(SimNode::List{ SimNode::GND, mTerminals[0]->node() });
102  mSubParallelCapacitor0->initialize(mFrequencies);
103  mSubParallelCapacitor0->initializeFromNodesAndTerminals(frequency);
104 
105  mSubParallelCapacitor1 = std::make_shared<EMT::Ph3::Capacitor>(mName + "_cap1", mLogLevel);
106  mSubParallelCapacitor1->setParameters(mParallelCap / 2.);
107  mSubParallelCapacitor1->connect(SimNode::List{ SimNode::GND, mTerminals[1]->node() });
108  mSubParallelCapacitor1->initialize(mFrequencies);
109  mSubParallelCapacitor1->initializeFromNodesAndTerminals(frequency);
110  }
111 
112  mSLog->debug(
113  "\n--debug--"
114  "\n seriesRes: {:s}"
115  "\n seriesInd: {:s}"
116  "\n Impedance: {:s}"
117  "\n vInit: {:s}"
118  "\n iInit: {:s}",
119  Logger::matrixToString(mSeriesRes),
120  Logger::matrixToString(mSeriesInd),
121  Logger::matrixCompToString(impedance),
122  Logger::matrixCompToString(vInitABC),
123  Logger::matrixCompToString(iInit));
124 
125  mSLog->info(
126  "\n--- Initialization from powerflow ---"
127  "\nVoltage across: {:s}"
128  "\nCurrent: {:s}"
129  "\nTerminal 0 voltage: {:s}"
130  "\nTerminal 1 voltage: {:s}"
131  "\nVirtual Node 1 voltage: {:s}"
132  "\n--- Initialization from powerflow finished ---",
133  Logger::matrixToString(mIntfVoltage),
134  Logger::matrixToString(mIntfCurrent),
135  Logger::phasorToString(RMS3PH_TO_PEAK1PH * initialSingleVoltage(0)),
136  Logger::phasorToString(RMS3PH_TO_PEAK1PH * initialSingleVoltage(1)),
137  Logger::phasorToString(mVirtualNodes[0]->initialSingleVoltage()));
138  mSLog->flush();
139 }

141 void EMT::Ph3::PiLine::mnaInitialize(Real omega, Real timeStep, Attribute<Matrix>::Ptr leftVector) {
142  MNAInterface::mnaInitialize(omega, timeStep);
143  updateMatrixNodeIndices();
144  MNAInterface::List subComps({ mSubSeriesResistor, mSubSeriesInductor });
145 
146  mSubSeriesResistor->mnaInitialize(omega, timeStep, leftVector);
147  mSubSeriesInductor->mnaInitialize(omega, timeStep, leftVector);
148  mRightVectorStamps.push_back(&mSubSeriesInductor->attribute<Matrix>("right_vector")->get());
149 
150  mSubParallelResistor0->mnaInitialize(omega, timeStep, leftVector);
151  mSubParallelResistor1->mnaInitialize(omega, timeStep, leftVector);
152  subComps.push_back(mSubParallelResistor0);
153  subComps.push_back(mSubParallelResistor1);
154 
155  if (mParallelCap(0,0) > 0) {
156  mSubParallelCapacitor0->mnaInitialize(omega, timeStep, leftVector);
157  mSubParallelCapacitor1->mnaInitialize(omega, timeStep, leftVector);
158  mRightVectorStamps.push_back(&mSubParallelCapacitor0->attribute<Matrix>("right_vector")->get());
159  mRightVectorStamps.push_back(&mSubParallelCapacitor1->attribute<Matrix>("right_vector")->get());
160  subComps.push_back(mSubParallelCapacitor0);
161  subComps.push_back(mSubParallelCapacitor1);
162  }
163  mMnaTasks.push_back(std::make_shared<MnaPreStep>(*this));
164  mMnaTasks.push_back(std::make_shared<MnaPostStep>(*this, leftVector));
165  mRightVector = Matrix::Zero(leftVector->get().rows(), 1);
166 }

246 void EMT::Ph3::PiLine::mnaUpdateVoltage(const Matrix& leftVector) {
247  // v1 - v0
248  mIntfVoltage = Matrix::Zero(3, 1);
249  if (terminalNotGrounded(1)) {
250  mIntfVoltage(0, 0) = Math::realFromVectorElement(leftVector, matrixNodeIndex(1, 0));
251  mIntfVoltage(1, 0) = Math::realFromVectorElement(leftVector, matrixNodeIndex(1, 1));
252  mIntfVoltage(2, 0) = Math::realFromVectorElement(leftVector, matrixNodeIndex(1, 2));
253  }
254  if (terminalNotGrounded(0)) {
255  mIntfVoltage(0, 0) = mIntfVoltage(0, 0) - Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 0));
256  mIntfVoltage(1, 0) = mIntfVoltage(1, 0) - Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 1));
257  mIntfVoltage(2, 0) = mIntfVoltage(2, 0) - Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 2));
258  }
259 }
