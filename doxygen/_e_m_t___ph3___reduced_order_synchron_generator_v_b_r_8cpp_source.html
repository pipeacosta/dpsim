104 void EMT::Ph3::ReducedOrderSynchronGeneratorVBR::mnaPostStep(const Matrix& leftVector) {
105  // update armature voltage
106  (**mIntfVoltage)(0, 0) = Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 0));
107  (**mIntfVoltage)(1, 0) = Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 1));

109  **mVdq0 = mAbcToDq0 * **mIntfVoltage / mBase_V;

111  // update armature current

114  (**mIntfCurrent)(2, 0) = Math::realFromVectorElement(leftVector, mVirtualNodes[1]->matrixNodeIndex(PhaseType::C));

118 Matrix EMT::Ph3::ReducedOrderSynchronGeneratorVBR::get_parkTransformMatrix() {
119  Matrix abcToDq0(3, 3);
120 
121  abcToDq0 <<
122  2./3.*cos(**mThetaMech), 2./3.*cos(**mThetaMech - 2.*PI/3.), 2./3.*cos(**mThetaMech + 2.*PI/3.),
123  -2./3.*sin(**mThetaMech), -2./3.*sin(**mThetaMech - 2.*PI/3.), -2./3.*sin(**mThetaMech + 2.*PI/3.),
124  1./3., 1./3., 1./3.;
125 
126  return abcToDq0;
