38 void EMT::Ph3::RxLine::initializeFromNodesAndTerminals(Real frequency) {
39 
40  // Static calculation
41  Real omega = 2. * PI * frequency;
42  MatrixComp impedance = MatrixComp::Zero(3, 3);
43  impedance <<
44  Complex((**mSeriesRes)(0, 0), omega * (**mSeriesInd)(0, 0)), Complex((**mSeriesRes)(0, 1), omega * (**mSeriesInd)(0, 1)), Complex((**mSeriesRes)(0, 2), omega * (**mSeriesInd)(0, 2)),
45  Complex((**mSeriesRes)(1, 0), omega * (**mSeriesInd)(1, 0)), Complex((**mSeriesRes)(1, 1), omega * (**mSeriesInd)(1, 1)), Complex((**mSeriesRes)(1, 2), omega * (**mSeriesInd)(1, 2)),
46  Complex((**mSeriesRes)(2, 0), omega * (**mSeriesInd)(2, 0)), Complex((**mSeriesRes)(2, 1), omega * (**mSeriesInd)(2, 1)), Complex((**mSeriesRes)(2, 2), omega * (**mSeriesInd)(2, 2));
47 
48  MatrixComp vInitABC = MatrixComp::Zero(3, 1);
49  vInitABC(0, 0) = mVirtualNodes[0]->initialSingleVoltage() - initialSingleVoltage(0);
50  vInitABC(1, 0) = vInitABC(0, 0) * SHIFT_TO_PHASE_B;
51  vInitABC(2, 0) = vInitABC(0, 0) * SHIFT_TO_PHASE_C;
52 
53  **mIntfCurrent = (impedance.inverse() * vInitABC).real();
54  **mIntfVoltage = vInitABC.real();
55  // Initialization of virtual node
56  // Initial voltage of phase B,C is set after A
57  MatrixComp vInitTerm0 = MatrixComp::Zero(3, 1);
58  vInitTerm0(0, 0) = initialSingleVoltage(0);
59  vInitTerm0(1, 0) = vInitTerm0(0, 0) * SHIFT_TO_PHASE_B;
60  vInitTerm0(2, 0) = vInitTerm0(0, 0) * SHIFT_TO_PHASE_C;
61 
62  mVirtualNodes[0]->setInitialVoltage(vInitTerm0 + **mSeriesRes * **mIntfCurrent);
63 
64  // Default model with virtual node in between
65  mSubResistor = std::make_shared<EMT::Ph3::Resistor>(**mName + "_res", mLogLevel);
66  mSubResistor->setParameters(**mSeriesRes);
67  mSubResistor->connect({ mTerminals[0]->node(), mVirtualNodes[0] });
68  mSubResistor->initializeFromNodesAndTerminals(frequency);
69 
70  mSubInductor = std::make_shared<EMT::Ph3::Inductor>(**mName + "_ind", mLogLevel);
71  mSubInductor->setParameters(**mSeriesInd);
72  mSubInductor->connect({ mVirtualNodes[0], mTerminals[1]->node() });
73  mSubInductor->initializeFromNodesAndTerminals(frequency);
74 
75  mInitialResistor = std::make_shared<EMT::Ph3::Resistor>(**mName + "_snubber_res", mLogLevel);
76  Matrix defaultSnubRes = Matrix::Zero(3, 1);
77  defaultSnubRes <<
78  1e6, 0, 0,
79  0, 1e6, 0,
80  0, 0, 1e6;
81  mInitialResistor->setParameters(defaultSnubRes);
82  mInitialResistor->connect({ SimNode::GND, mTerminals[1]->node() });
83  mInitialResistor->initializeFromNodesAndTerminals(frequency);
84 
85  mSLog->info(
86  "\n--- Initialization from powerflow ---"
87  "\nVoltage across: {:s}"
88  "\nCurrent: {:s}"
89  "\nTerminal 0 voltage: {:s}"
90  "\nTerminal 1 voltage: {:s}"
91  "\n--- Initialization from powerflow finished ---",
92  Logger::matrixToString(**mIntfVoltage),
93  Logger::matrixToString(**mIntfCurrent),
94  Logger::phasorToString(initialSingleVoltage(0)),
95  Logger::phasorToString(initialSingleVoltage(1)));
96 }

139 void EMT::Ph3::RxLine::mnaUpdateVoltage(const Matrix& leftVector) {
140  // v1 - v0
141  **mIntfVoltage = Matrix::Zero(3, 1);
142  if (terminalNotGrounded(1)) {
143  (**mIntfVoltage)(0, 0) = Math::realFromVectorElement(leftVector, matrixNodeIndex(1, 0));
144  (**mIntfVoltage)(1, 0) = Math::realFromVectorElement(leftVector, matrixNodeIndex(1, 1));
145  (**mIntfVoltage)(2, 0) = Math::realFromVectorElement(leftVector, matrixNodeIndex(1, 2));
146  }
147  if (terminalNotGrounded(0)) {
148  (**mIntfVoltage)(0, 0) = (**mIntfVoltage)(0, 0) - Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 0));
149  (**mIntfVoltage)(1, 0) = (**mIntfVoltage)(1, 0) - Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 1));
150  (**mIntfVoltage)(2, 0) = (**mIntfVoltage)(2, 0) - Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 2));
151  }
152 }
