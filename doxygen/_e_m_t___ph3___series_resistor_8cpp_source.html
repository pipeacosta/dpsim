32 void EMT::Ph3::SeriesResistor::initializeFromNodesAndTerminals(Real frequency) {
33 
34  Complex phasorA = initialSingleVoltage(1) - initialSingleVoltage(0);
35  mIntfVoltage(0,0) = phasorA.real();
36  Complex alpha(cos(2./3.*PI), sin(2./3.*PI));
37  mIntfVoltage(1, 0) = Complex(phasorA * pow(alpha,2)).real();
38  mIntfVoltage(2, 0) = Complex(phasorA * alpha).real();
39 
40  mIntfCurrent = mIntfVoltage / mResistance;
41 
42  mSLog->info("\n--- Initialization from powerflow ---"
43  "\nVoltage across amplitude and phase: \n{}"
44  "\nCurrent amplitude and phase: \n{}"
45  "\nTerminal 0 voltage amplitude and phase: \n{}"
46  "\nTerminal 1 voltage amplitude and phase: \n{}"
47  "\n--- Initialization from powerflow finished ---",
48  Logger::phasorMatrixToString(mIntfVoltage),
49  Logger::phasorMatrixToString(mIntfCurrent),
50  Logger::phasorMatrixToString(initialVoltage(0)),
51  Logger::phasorMatrixToString(initialVoltage(1)));
52 }

61 void EMT::Ph3::SeriesResistor::mnaApplySystemMatrixStamp(Matrix& systemMatrix) {
62  Real conductance = 1./mResistance;
63 
64  // Set diagonal entries
65  if (terminalNotGrounded(0))
66  Math::addToMatrixElement(systemMatrix, matrixNodeIndices(0), matrixNodeIndices(0), conductance);
67  if (terminalNotGrounded(1))
68  Math::addToMatrixElement(systemMatrix, matrixNodeIndices(1), matrixNodeIndices(1), conductance);
69  // Set off diagonal entries
70  if (terminalNotGrounded(0) && terminalNotGrounded(1)) {
71  Math::addToMatrixElement(systemMatrix, matrixNodeIndices(0), matrixNodeIndices(1), -conductance);
72  Math::addToMatrixElement(systemMatrix, matrixNodeIndices(1), matrixNodeIndices(0), -conductance);
73  }
74 
75  if (terminalNotGrounded(0))
76  mSLog->info("Add {} to {}, {}", conductance, matrixNodeIndex(0,0), matrixNodeIndex(0,0));
77  if (terminalNotGrounded(1))
78  mSLog->info("Add {} to {}, {}", conductance, matrixNodeIndex(1,0), matrixNodeIndex(1,0));
79  if (terminalNotGrounded(0) && terminalNotGrounded(1)) {
80  mSLog->info("Add {} to {}, {}", -conductance, matrixNodeIndex(0,0), matrixNodeIndex(1,0));
81  mSLog->info("Add {} to {}, {}", -conductance, matrixNodeIndex(1,0), matrixNodeIndex(0,0));
82  }
83 }

90 void EMT::Ph3::SeriesResistor::mnaUpdateVoltage(const Matrix& leftVector) {
91  // Voltage across component is defined as V1 - V0
92  mIntfVoltage = Matrix::Zero(3,1);
93  if (terminalNotGrounded(1)) {
94  mIntfVoltage(0,0) = Math::realFromVectorElement(leftVector, matrixNodeIndex(1,0));
95  mIntfVoltage(1,0) = Math::realFromVectorElement(leftVector, matrixNodeIndex(1,1));
96  mIntfVoltage(2,0) = Math::realFromVectorElement(leftVector, matrixNodeIndex(1,2));
97  }
98  if (terminalNotGrounded(0)) {
99  mIntfVoltage(0,0) = mIntfVoltage(0,0) - Math::realFromVectorElement(leftVector, matrixNodeIndex(0,0));
100  mIntfVoltage(1,0) = mIntfVoltage(1,0) - Math::realFromVectorElement(leftVector, matrixNodeIndex(0,1));
101  mIntfVoltage(2,0) = mIntfVoltage(2,0) - Math::realFromVectorElement(leftVector, matrixNodeIndex(0,2));
102  }
103 
104  SPDLOG_LOGGER_DEBUG(mSLog, "Voltage A: {}", mIntfVoltage(0,0));
105 }
