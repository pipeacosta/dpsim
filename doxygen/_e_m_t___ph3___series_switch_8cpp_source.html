32 void EMT::Ph3::SeriesSwitch::initializeFromNodesAndTerminals(Real frequency) {
33 
34  Real impedance = (mIsClosed) ? mClosedResistance : mOpenResistance;
35 
36  Complex phasorA = initialSingleVoltage(1) - initialSingleVoltage(0);
37  mIntfVoltage(0,0) = phasorA.real();
38  Complex alpha(cos(2./3.*PI), sin(2./3.*PI));
39  mIntfVoltage(1, 0) = Complex(phasorA * pow(alpha,2)).real();
40  mIntfVoltage(2, 0) = Complex(phasorA * alpha).real();
41 
42  mIntfCurrent = mIntfVoltage / impedance;
43 
44  mSLog->info("\n--- Initialization from powerflow ---"
45  "\nVoltage across amplitude and phase: \n{}"
46  "\nCurrent amplitude and phase: \n{}"
47  "\nTerminal 0 voltage amplitude and phase: \n{}"
48  "\nTerminal 1 voltage amplitude and phase: \n{}"
49  "\n--- Initialization from powerflow finished ---",
50  Logger::phasorMatrixToString(mIntfVoltage),
51  Logger::phasorMatrixToString(mIntfCurrent),
52  Logger::phasorMatrixToString(initialVoltage(0)),
53  Logger::phasorMatrixToString(initialVoltage(1)));
54 }

63 void EMT::Ph3::SeriesSwitch::mnaApplySystemMatrixStamp(Matrix& systemMatrix) {
64  Real conductance = (mIsClosed)
65  ? 1./mClosedResistance
66  : 1./mOpenResistance;
67 
68  // Set diagonal entries
69  if (terminalNotGrounded(0))
70  Math::addToMatrixElement(systemMatrix, matrixNodeIndices(0), matrixNodeIndices(0), conductance);
71  if (terminalNotGrounded(1))
72  Math::addToMatrixElement(systemMatrix, matrixNodeIndices(1), matrixNodeIndices(1), conductance);
73  // Set off diagonal entries
74  if (terminalNotGrounded(0) && terminalNotGrounded(1)) {
75  Math::addToMatrixElement(systemMatrix, matrixNodeIndices(0), matrixNodeIndices(1), -conductance);
76  Math::addToMatrixElement(systemMatrix, matrixNodeIndices(1), matrixNodeIndices(0), -conductance);
77  }
78 
79  if (terminalNotGrounded(0))
80  mSLog->info("Add {} to {}, {}", conductance, matrixNodeIndices(0)[0], matrixNodeIndices(0)[0]);
81  if (terminalNotGrounded(1))
82  mSLog->info("Add {} to {}, {}", conductance, matrixNodeIndices(1)[0], matrixNodeIndices(1)[0]);
83  if (terminalNotGrounded(0) && terminalNotGrounded(1)) {
84  mSLog->info("Add {} to {}, {}", -conductance, matrixNodeIndices(0)[0], matrixNodeIndices(1)[0]);
85  mSLog->info("Add {} to {}, {}", -conductance, matrixNodeIndices(1)[0], matrixNodeIndices(0)[0]);
86  }
87 }

120 void EMT::Ph3::SeriesSwitch::mnaUpdateVoltage(const Matrix& leftVector) {
121  // Voltage across component is defined as V1 - V0
122  mIntfVoltage = Matrix::Zero(3,1);
123  if (terminalNotGrounded(1)) {
124  mIntfVoltage(0,0) = Math::realFromVectorElement(leftVector, matrixNodeIndex(1,0));
125  mIntfVoltage(1,0) = Math::realFromVectorElement(leftVector, matrixNodeIndex(1,1));
126  mIntfVoltage(2,0) = Math::realFromVectorElement(leftVector, matrixNodeIndex(1,2));
127  }
128  if (terminalNotGrounded(0)) {
129  mIntfVoltage(0,0) = mIntfVoltage(0,0) - Math::realFromVectorElement(leftVector, matrixNodeIndex(0,0));
130  mIntfVoltage(1,0) = mIntfVoltage(1,0) - Math::realFromVectorElement(leftVector, matrixNodeIndex(0,1));
131  mIntfVoltage(2,0) = mIntfVoltage(2,0) - Math::realFromVectorElement(leftVector, matrixNodeIndex(0,2));
132  }
133 
134  SPDLOG_LOGGER_DEBUG(mSLog, "Voltage A: {}", mIntfVoltage(0,0));
135 }
