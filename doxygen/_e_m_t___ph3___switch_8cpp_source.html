33 void EMT::Ph3::Switch::initializeFromNodesAndTerminals(Real frequency) {
34 
35  Matrix impedance = (**mSwitchClosed) ? **mClosedResistance : **mOpenResistance;
36  MatrixComp vInitABC = MatrixComp::Zero(3, 1);
37  vInitABC(0,0) = initialSingleVoltage(1) - initialSingleVoltage(0);
38  vInitABC(1, 0) = vInitABC(0, 0) * SHIFT_TO_PHASE_B;
39  vInitABC(2, 0) = vInitABC(0, 0) * SHIFT_TO_PHASE_C;
40  **mIntfVoltage = vInitABC.real();
41  **mIntfCurrent = (impedance.inverse() * vInitABC).real();
42 
43  mSLog->info(
44  "\n--- Initialization from powerflow ---"
45  "\nVoltage across: {:s}"
46  "\nCurrent: {:s}"
47  "\nTerminal 0 voltage: {:s}"
48  "\nTerminal 1 voltage: {:s}"
49  "\n--- Initialization from powerflow finished ---",
50  Logger::matrixToString(**mIntfVoltage),
51  Logger::matrixToString(**mIntfCurrent),
52  Logger::phasorToString(initialSingleVoltage(0)),
53  Logger::phasorToString(initialSingleVoltage(1)));
54 }

185 void EMT::Ph3::Switch::mnaUpdateVoltage(const Matrix& leftVector) {
186  // Voltage across component is defined as V1 - V0
187  **mIntfVoltage = Matrix::Zero(3, 1);
188  if (terminalNotGrounded(1)) {
189  (**mIntfVoltage)(0, 0) = Math::realFromVectorElement(leftVector, matrixNodeIndex(1, 0));
190  (**mIntfVoltage)(1, 0) = Math::realFromVectorElement(leftVector, matrixNodeIndex(1, 1));
191  (**mIntfVoltage)(2, 0) = Math::realFromVectorElement(leftVector, matrixNodeIndex(1, 2));
192  }
193  if (terminalNotGrounded(0)) {
194  (**mIntfVoltage)(0, 0) = (**mIntfVoltage)(0, 0) - Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 0));
195  (**mIntfVoltage)(1, 0) = (**mIntfVoltage)(1, 0) - Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 1));
196  (**mIntfVoltage)(2, 0) = (**mIntfVoltage)(2, 0) - Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 2));
197  }
198 }
