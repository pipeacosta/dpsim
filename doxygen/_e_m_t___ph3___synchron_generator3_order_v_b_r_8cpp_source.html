36 void EMT::Ph3::SynchronGenerator3OrderVBR::specificInitialization() {
37  // initial voltage behind the transient reactance in the dq0 reference frame
38  mEdq0_t(1,0) = mVdq0(1,0) + mIdq0(0,0) * mLd_t;
39 
40  // calculate auxiliar VBR constants
41  calculateAuxiliarConstants();
42 
43  // dq0 resistance matrix
44  mResistanceMatrixDq0 = Matrix::Zero(3,3);
45  mResistanceMatrixDq0 << 0.0, -mLq, 0.0,
46  mLd_t - mAq, 0.0, 0.0,
47  0.0, 0.0, mL0;
48 
49  // initialize conductance matrix
50  mConductanceMatrix = Matrix::Zero(3,3);
51 
52  mSLog->info(
53  "\n--- Model specific initialization ---"
54  "\nInitial Eq_t (per unit): {:f}"

56  mEdq0_t(1,0)

58  mSLog->flush();
59 }

61 void EMT::Ph3::SynchronGenerator3OrderVBR::calculateAuxiliarConstants() {
62  mAq = - mTimeStep * (mLd - mLd_t) / (2 * mTd0_t + mTimeStep);
63  mBq = (2 * mTd0_t - mTimeStep) / (2 * mTd0_t + mTimeStep);
64  mCq = 2 * mTimeStep * mEf / (2 * mTd0_t + mTimeStep);
65 }

67 void EMT::Ph3::SynchronGenerator3OrderVBR::stepInPerUnit() {
68 
69  if (mSimTime>0.0) {
70  // calculate Eq_t at t=k
71  mEdq0_t(1,0) = mIdq0(0,0) * mLd_t + mVdq0(1,0);
72 
73  // calculate mechanical variables at t=k+1 with forward euler
74  mElecTorque = (mVdq0(0,0) * mIdq0(0,0) + mVdq0(1,0) * mIdq0(1,0));
75  mOmMech = mOmMech + mTimeStep * (1. / (2. * mH) * (mMechTorque - mElecTorque));
76  mThetaMech = mThetaMech + mTimeStep * (mOmMech * mBase_OmMech);
77  mDelta = mDelta + mTimeStep * (mOmMech - 1.) * mBase_OmMech;
78  }
79 
80  // get transformation matrix
81  mAbcToDq0 = get_parkTransformMatrix();
82  mDq0ToAbc = get_inverseParkTransformMatrix();
83 
84  // calculate resistance matrix at t=k+1
85  calculateResistanceMatrix();
86 
87  // VBR history voltage
88  mEhs_vbr(0,0) = 0.0;
89  mEhs_vbr(1,0) = mAq * mIdq0(0,0) + mBq * mEdq0_t(1,0) + mCq;
90  mEhs_vbr(2,0) = 0.0;
91 
92  // convert Edq_t into the abc reference frame
93  mEvbr = mDq0ToAbc * mEhs_vbr * mBase_V;
94 }
