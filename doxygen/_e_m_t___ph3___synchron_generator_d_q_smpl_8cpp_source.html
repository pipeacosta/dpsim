131 void EMT::Ph3::SynchronGeneratorDQSmpl::stepInPerUnit(Real om, Real dt, Real time, NumericalMethod numMethod)
132 {
133 
134  // Calculation of rotational speed with euler
135  mElecTorque = (mPsid*mIq - mPsiq*mId);
136  mOmMech = mOmMech + dt * (1 / (2 * mH) * (mMechTorque - mElecTorque));
137 
138  // Calculation of rotor angular position
139  mThetaMech = mThetaMech + dt * (mOmMech * mBase_OmMech);
140 
141 
142  mPsifd = mPsifd + dt*mBase_OmMech*(mVfd - mRfd*mIfd);
143 
144  mR_eq <<
145  -mRs, (mLl + mLmq),
146  -(mLl + mLmd) + mLmd*mLmd / (mLlfd + mLmd), -mRs;
147  mE_eq <<
148  0,
149  (mLmd)*mPsifd / (mLlfd + mLmd);
150 
151  mKrs_teta <<
152  2. / 3. * cos(mThetaMech), 2. / 3. * cos(mThetaMech - 2. * M_PI / 3.), 2. / 3. * cos(mThetaMech + 2. * M_PI / 3.),
153  -2. / 3. * sin(mThetaMech), -2. / 3. * sin(mThetaMech - 2. * M_PI / 3.), -2. / 3. * sin(mThetaMech + 2. * M_PI / 3.);
154 
155  mKrs_teta_inv <<
156  cos(mThetaMech), -sin(mThetaMech),
157  cos(mThetaMech - 2. * M_PI / 3.), -sin(mThetaMech - 2. * M_PI / 3.),
158  cos(mThetaMech + 2. * M_PI / 3.), -sin(mThetaMech + 2. * M_PI / 3.);
159 
160  mR_eq_abc = mKrs_teta_inv*mR_eq*mKrs_teta;
161  mG_eq_abc = (mR_eq_abc - mR_load).inverse();
162  mEq_abc = mKrs_teta_inv*mE_eq;
163 
164  //Matrix mEq_abc2 = Matrix::Zero(3, 1);
165  //mEq_abc2 = inverseParkTransform(mThetaMech, mE_eq(0), mE_eq(1), 0);
166 
167  //Matrix mIdq2 = Matrix::Zero(2, 1);
168  //Matrix mVdq2 = Matrix::Zero(2, 1);
169  //Matrix mIabc2 = Matrix::Zero(3, 1);
170 
171  //mVdq2 << mVdq0(0),
172  // mVdq0(1);
173 
174  //mIdq2 = mR_eq.inverse()*(mVdq2 - mE_eq);
175 
176  //mIabc2 = mKrs_teta_inv*mIdq2;
177 
178  mIabc = -mG_eq_abc*mEq_abc;
179  mIdq = mKrs_teta*mIabc;
180 
181  //mVdq = mR_eq*mIdq + mE_eq;
182 
183  mId = (**mIdq)(0);
184  mIq = (**mIdq)(1);
185  mIfd = (mPsifd + mLmd*mId) / (mLlfd + mLmd);
186 
187  mCurrents(0, 0) = mIq;
188  mCurrents(1, 0) = mId;
189  mCurrents(2, 0) = mIfd;
190 
191  mFluxes = mInductanceMat*mCurrents;
192 
193  mPsiq = mFluxes(0, 0);
194  mPsid = mFluxes(1, 0);
195  mPsifd = mFluxes(2, 0);
196 
197  mIa = mBase_I * mIabc(0);
198  mIb = mBase_I * mIabc(1);
199  mIc = mBase_I * mIabc(2);
200 
201  mVoltages <<
202  mVq,
203  mVd,
204  mVfd;
205 }

231 Matrix EMT::Ph3::SynchronGeneratorDQSmpl::parkTransform(Real theta, Real a, Real b, Real c)
232 {
233  Matrix dq0vector(3, 1);
234 
235  // Park transform according to Kundur
236  Real d, q;
237 
238  d = 2. / 3. * cos(theta)*a + 2. / 3. * cos(theta - 2. * M_PI / 3.)*b + 2. / 3. * cos(theta + 2. * M_PI / 3.)*c;
239  q = -2. / 3. * sin(theta)*a - 2. / 3. * sin(theta - 2. * M_PI / 3.)*b - 2. / 3. * sin(theta + 2. * M_PI / 3.)*c;
240 
241  Vector zero;
242  zero << 1. / 3. * a, 1. / 3. * b, 1. / 3. * c;
243 
244  dq0vector <<
245  d,
246  q,
247  0;
248 
249  return dq0vector;
250 }

252 Matrix EMT::Ph3::SynchronGeneratorDQSmpl::inverseParkTransform(Real theta, Real d, Real q, Real zero)
253 {
254  Matrix abcVector(3, 1);
255 
256  // Park transform according to Kundur
257  Real a, b, c;
258 
259  a = cos(theta)*d - sin(theta)*q + 1.*zero;
260  b = cos(theta - 2. * M_PI / 3.)*d - sin(theta - 2. * M_PI / 3.)*q + 1.*zero;
261  c = cos(theta + 2. * M_PI / 3.)*d - sin(theta + 2. * M_PI / 3.)*q + 1.*zero;
262 
263  abcVector <<
264  a,
265  b,
266  c;
267 
268  return abcVector;
