39 void EMT::Ph3::SynchronGeneratorIdeal::initializeFromNodesAndTerminals(Real frequency) {
40 
41  if (mSourceType == CPS::GeneratorType::IdealVoltageSource) {
42  mSubVoltageSource = EMT::Ph3::VoltageSource::make(mName + "_vs", mLogLevel);
43  mSubComponents.push_back(mSubVoltageSource);
44  } else {
45  mSubCurrentSource = EMT::Ph3::CurrentSource::make(mName + "_cs", mLogLevel);
46  mSubComponents.push_back(mSubCurrentSource);
47  }
48 
49  mSubComponents[0]->connect({ SimNode::GND, node(0) });
50 
51  if (mSourceType == CPS::GeneratorType::IdealVoltageSource)
52  mSubComponents[0]->setVirtualNodeAt(mVirtualNodes[0], 0);
53 
54  if (mSourceType == CPS::GeneratorType::IdealCurrentSource)
55  mSubComponents[0]->setTerminalAt(terminal(0), 1);
56 
57  mSubComponents[0]->initialize(mFrequencies);
58  mSubComponents[0]->initializeFromNodesAndTerminals(frequency);
59 
60  if (mSourceType == CPS::GeneratorType::IdealVoltageSource)
61  setAttributeRef("V_ref", mSubComponents[0]->attribute<MatrixComp>("V_ref"));
62 
63  mSLog->info(
64  "\n--- Initialization from powerflow ---"
65  "\nTerminal 0 voltage: {:s}"
66  "\nTerminal 0 power: {:s}"
67  "\n--- Initialization from powerflow finished ---",
68  Logger::phasorToString(initialSingleVoltage(0)),
69  Logger::complexToString(terminal(0)->singlePower()));
70 }
