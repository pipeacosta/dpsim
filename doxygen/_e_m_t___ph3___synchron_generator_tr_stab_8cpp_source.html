21 Matrix EMT::Ph3::SynchronGeneratorTrStab::getParkTransformMatrixPowerInvariant(Real theta) {
22  // Return park matrix for theta
23  // Assumes that d-axis starts aligned with phase a
24  Matrix Tdq = Matrix::Zero(2, 3);
25  Real k = sqrt(2. / 3.);
26  Tdq <<
27  k * cos(theta), k * cos(theta - 2. * M_PI / 3.), k * cos(theta + 2. * M_PI / 3.),
28  -k * sin(theta), -k * sin(theta - 2. * M_PI / 3.), -k * sin(theta + 2. * M_PI / 3.);
29  return Tdq;
30 }

60 void EMT::Ph3::SynchronGeneratorTrStab::setFundamentalParametersPU(Real nomPower, Real nomVolt, Real nomFreq,
61  Real Ll, Real Lmd, Real Llfd, Real inertia, Real D) {
62  setBaseParameters(nomPower, nomVolt, nomFreq);
63 
64  // Input is in per unit but all values are converted to absolute values.
65  mParameterType = ParameterType::statorReferred;
66  mStateType = StateType::statorReferred;
67 
68  mLl = Ll;
69  mLmd = Lmd;
70  mLd = mLl + mLmd;
71  mLlfd = Llfd;
72  mLfd = mLlfd + mLmd;
73  // M = 2*H where H = inertia
74  mInertia = inertia;
75  // X'd in absolute values
76  mXpd = mNomOmega * (mLd - mLmd*mLmd / mLfd) * mBase_L;
77  mLpd = (mLd - mLmd*mLmd / mLfd) * mBase_L;
78 
79  mSLog->info("\n--- Parameters ---"
80  "\nimpedance: {:f}"
81  "\ninductance: {:f}", mXpd, mLpd);

132 void EMT::Ph3::SynchronGeneratorTrStab::initializeFromNodesAndTerminals(Real frequency) {
133 
134  // Initialize omega mech with nominal system frequency
135  mOmMech = mNomOmega;
136 
137  mInitElecPower = (mInitElecPower == Complex(0,0))
138  ? -terminal(0)->singlePower()
139  : mInitElecPower;
140  mInitMechPower = (mInitElecPower == Complex(0,0))
141  ? mInitElecPower.real()
142  : mInitMechPower;
143 
144  // use complex interface quantities for initialization calculations
145  MatrixComp intfVoltageComplex = MatrixComp::Zero(3, 1);
146  MatrixComp intfCurrentComplex = MatrixComp::Zero(3, 1);
147  // // derive complex threephase initialization from single phase initial values (only valid for balanced systems)
148  // intfVoltageComplex(0, 0) = RMS3PH_TO_PEAK1PH * initialSingleVoltage(0);
149  intfVoltageComplex(0, 0) = initialSingleVoltage(0);
150  intfVoltageComplex(1, 0) = intfVoltageComplex(0, 0) * SHIFT_TO_PHASE_B;
151  intfVoltageComplex(2, 0) = intfVoltageComplex(0, 0) * SHIFT_TO_PHASE_C;
152  intfCurrentComplex(0, 0) = std::conj(-2./3.*mInitElecPower / intfVoltageComplex(0, 0));
153  intfCurrentComplex(1, 0) = intfCurrentComplex(0, 0) * SHIFT_TO_PHASE_B;
154  intfCurrentComplex(2, 0) = intfCurrentComplex(0, 0) * SHIFT_TO_PHASE_C;
155 
156  //save real interface quantities calculated from complex ones
157  mIntfVoltage = intfVoltageComplex.real();
158  mIntfCurrent = intfCurrentComplex.real();
159 
160  mImpedance = Complex(mRs, mXpd);
161 
162  // Calculate initial emf behind reactance from power flow results
163  mEp = intfVoltageComplex(0, 0) - mImpedance * intfCurrentComplex(0, 0);
164 
165  // The absolute value of Ep is constant, only delta_p changes every step
166  mEp_abs = Math::abs(mEp);
167  // Delta_p is the angular position of mEp with respect to the synchronously rotating reference
168  mDelta_p= Math::phase(mEp);
169 
170  // // Update active electrical power that is compared with the mechanical power
171  mElecActivePower = ( 3./2. * intfVoltageComplex(0,0) * std::conj( - intfCurrentComplex(0,0)) ).real();
172  // mElecActivePower = ( (mEp - mIntfVoltage(0,0)) / mImpedance * mIntfVoltage(0,0) ).real();
173  // For infinite power bus
174  // mElecActivePower = (Math::abs(mEp) * Math::abs(mIntfVoltage(0,0)) / mXpd) * sin(mDelta_p);
175 
176  // Start in steady state so that electrical and mech. power are the same
177  // because of the initial condition mOmMech = mNomOmega the damping factor is not considered at the initialisation
178  mMechPower = mElecActivePower- mKd*(mOmMech - mNomOmega);
179 
180  // Initialize node between X'd and Ep
181  mVirtualNodes[0]->setInitialVoltage(PEAK1PH_TO_RMS3PH*mEp);
182 
183  MatrixComp vref = MatrixComp::Zero(3,1);
184  vref= CPS::Math::singlePhaseVariableToThreePhase(mEp);
185 
186  // Create sub voltage source for emf
187  mSubVoltageSource = EMT::Ph3::VoltageSource::make(mName + "_src", mLogLevel);
188  mSubVoltageSource->setParameters(vref,frequency);
189  mSubVoltageSource->connect({SimNode::GND, mVirtualNodes[0]});
190  mSubVoltageSource->setVirtualNodeAt(mVirtualNodes[1], 0);
191  mSubVoltageSource->initialize(mFrequencies);
192  mSubVoltageSource->initializeFromNodesAndTerminals(frequency);
193 
194  // Create sub inductor as Xpd
195  mSubInductor = EMT::Ph3::Inductor::make(mName + "_ind", mLogLevel);
196  mSubInductor->setParameters(CPS::Math::singlePhaseParameterToThreePhase(mLpd));
197  mSubInductor->connect({mVirtualNodes[0],terminal(0)->node()});
198  mSubInductor->initialize(mFrequencies);
199  mSubInductor->initializeFromNodesAndTerminals(frequency);
200 
201  mSLog->info("\n--- Initialize according to powerflow ---"
202  "\nTerminal 0 voltage: {:e}<{:e}"
203  "\nVoltage behind reactance: {:e}<{:e}"
204  "\ninitial electrical power: {:e}+j{:e}"
205  "\nactive electrical power: {:e}"
206  "\nmechanical power: {:e}"
207  "\n--- End of powerflow initialization ---",
208  Math::abs(mIntfVoltage(0,0)), Math::phaseDeg(mIntfVoltage(0,0)),
209  Math::abs(mEp), Math::phaseDeg(mEp),
210  mInitElecPower.real(), mInitElecPower.imag(),
211  mElecActivePower, mMechPower);
212 }

214 void EMT::Ph3::SynchronGeneratorTrStab::step(Real time) {
215  // #### Calculations on input of time step k ####
216  // Transform interface quantities to synchronously rotating DQ reference frame
217  Matrix intfVoltageDQ = parkTransformPowerInvariant(mThetaN, mIntfVoltage);
218  Matrix intfCurrentDQ = parkTransformPowerInvariant(mThetaN, mIntfCurrent);
219  // Update electrical power (minus sign to calculate generated power from consumed current)
220  mElecActivePower = - 1. * (intfVoltageDQ(0, 0)*intfCurrentDQ(0, 0) + intfVoltageDQ(1, 0)*intfCurrentDQ(1, 0));
221 
222  // The damping factor Kd is adjusted to obtain a damping ratio of 0.3
223  // Real MaxElecActivePower= Math::abs(mEp) * Math::abs(mIntfVoltage(0,0)) / mXpd;
224  // mKd=4*0.3*sqrt(mNomOmega*mInertia*MaxElecActivePower*0.5);
225  mKd=1*mNomPower;
226 
227  // #### Calculate state for time step k+1 ####
228  // semi-implicit Euler or symplectic Euler method for mechanical equations
229  Real dOmMech = mNomOmega / (2.*mInertia * mNomPower) * (mMechPower - mElecActivePower-mKd*(mOmMech - mNomOmega));
230  if (mBehaviour == Behaviour::MNASimulation)
231  mOmMech = mOmMech + mTimeStep * dOmMech;
232  Real dDelta_p = mOmMech - mNomOmega;
233  if (mBehaviour == Behaviour::MNASimulation)
234  mDelta_p = mDelta_p + mTimeStep * dDelta_p;
235  // Update emf - only phase changes
236  if (mBehaviour == Behaviour::MNASimulation)
237  mEp = Complex(mEp_abs * cos(mDelta_p), mEp_abs * sin(mDelta_p));
238 
239  // Update nominal system angle
240  mThetaN = mThetaN + mTimeStep * mNomOmega;
241 
242  // mStates << Math::abs(mEp), Math::phaseDeg(mEp), mElecActivePower, mMechPower,
243  // mDelta_p, mOmMech, dOmMech, dDelta_p, mIntfVoltage(0,0).real(), mIntfVoltage(0,0).imag();
244  // SPDLOG_LOGGER_DEBUG(mSLog, "\nStates, time {:f}: \n{:s}", time, Logger::matrixToString(mStates));
245 }
