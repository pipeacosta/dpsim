21 Matrix EMT::Ph3::SynchronGeneratorTrStab::getParkTransformMatrixPowerInvariant(Real theta) {
22  // Return park matrix for theta
23  // Assumes that d-axis starts aligned with phase a
24  Matrix Tdq = Matrix::Zero(2, 3);
25  Real k = sqrt(2. / 3.);
26  Tdq <<
27  k * cos(theta), k * cos(theta - 2. * M_PI / 3.), k * cos(theta + 2. * M_PI / 3.),
28  -k * sin(theta), -k * sin(theta - 2. * M_PI / 3.), -k * sin(theta + 2. * M_PI / 3.);
29  return Tdq;
30 }

139 void EMT::Ph3::SynchronGeneratorTrStab::initializeFromNodesAndTerminals(Real frequency) {
140 
141  // Initialize omega mech with nominal system frequency
142  **mOmMech = mNomOmega;
143 
144  mInitElecPower = (mInitElecPower == Complex(0,0))
145  ? -terminal(0)->singlePower()
146  : mInitElecPower;
147  mInitMechPower = (mInitElecPower == Complex(0,0))
148  ? mInitElecPower.real()
149  : mInitMechPower;
150 
151  // use complex interface quantities for initialization calculations
152  MatrixComp intfVoltageComplex = MatrixComp::Zero(3, 1);
153  MatrixComp intfCurrentComplex = MatrixComp::Zero(3, 1);
154  // // derive complex threephase initialization from single phase initial values (only valid for balanced systems)
155  // intfVoltageComplex(0, 0) = RMS3PH_TO_PEAK1PH * initialSingleVoltage(0);
156  intfVoltageComplex(0, 0) = initialSingleVoltage(0);
157  intfVoltageComplex(1, 0) = intfVoltageComplex(0, 0) * SHIFT_TO_PHASE_B;
158  intfVoltageComplex(2, 0) = intfVoltageComplex(0, 0) * SHIFT_TO_PHASE_C;
159  intfCurrentComplex(0, 0) = std::conj(-2./3.*mInitElecPower / intfVoltageComplex(0, 0));
160  intfCurrentComplex(1, 0) = intfCurrentComplex(0, 0) * SHIFT_TO_PHASE_B;
161  intfCurrentComplex(2, 0) = intfCurrentComplex(0, 0) * SHIFT_TO_PHASE_C;
162 
163  //save real interface quantities calculated from complex ones
164  **mIntfVoltage = intfVoltageComplex.real();
165  **mIntfCurrent = intfCurrentComplex.real();
166 
167  mImpedance = Complex(**mRs, mXpd);
168 
169  // Calculate initial emf behind reactance from power flow results
170  **mEp = intfVoltageComplex(0, 0) - mImpedance * intfCurrentComplex(0, 0);
171 
172  // The absolute value of Ep is constant, only delta_p changes every step
173  **mEp_abs = Math::abs(**mEp);
174  // Delta_p is the angular position of mEp with respect to the synchronously rotating reference
175  **mDelta_p= Math::phase(**mEp);
176 
177  // // Update active electrical power that is compared with the mechanical power
178  **mElecActivePower = ( 3./2. * intfVoltageComplex(0,0) * std::conj( - intfCurrentComplex(0,0)) ).real();
179  // mElecActivePower = ( (mEp - (**mIntfVoltage)(0,0)) / mImpedance * (**mIntfVoltage)(0,0) ).real();
180  // For infinite power bus
181  // mElecActivePower = (Math::abs(mEp) * Math::abs((**mIntfVoltage)(0,0)) / mXpd) * sin(mDelta_p);
182 
183  // Start in steady state so that electrical and mech. power are the same
184  // because of the initial condition mOmMech = mNomOmega the damping factor is not considered at the initialisation
185  **mMechPower = **mElecActivePower - mKd*(**mOmMech - mNomOmega);
186 
187  // Initialize node between X'd and Ep
188  mVirtualNodes[0]->setInitialVoltage(PEAK1PH_TO_RMS3PH* **mEp);
189 
190  MatrixComp vref = MatrixComp::Zero(3,1);
191  vref= CPS::Math::singlePhaseVariableToThreePhase(**mEp);
192 
193  // Create sub voltage source for emf
194  mSubVoltageSource = EMT::Ph3::VoltageSource::make(**mName + "_src", mLogLevel);
195  mSubVoltageSource->setParameters(vref,frequency);
196  mSubVoltageSource->connect({SimNode::GND, mVirtualNodes[0]});
197  mSubVoltageSource->setVirtualNodeAt(mVirtualNodes[1], 0);
198  mSubVoltageSource->initialize(mFrequencies);
199  mSubVoltageSource->initializeFromNodesAndTerminals(frequency);
200 
201  // Create sub inductor as Xpd
202  mSubInductor = EMT::Ph3::Inductor::make(**mName + "_ind", mLogLevel);
203  mSubInductor->setParameters(CPS::Math::singlePhaseParameterToThreePhase(mLpd));
204  mSubInductor->connect({mVirtualNodes[0],terminal(0)->node()});
205  mSubInductor->initialize(mFrequencies);
206  mSubInductor->initializeFromNodesAndTerminals(frequency);
207 
208  mSLog->info("\n--- Initialize according to powerflow ---"
209  "\nTerminal 0 voltage: {:e}<{:e}"
210  "\nVoltage behind reactance: {:e}<{:e}"
211  "\ninitial electrical power: {:e}+j{:e}"
212  "\nactive electrical power: {:e}"
213  "\nmechanical power: {:e}"
214  "\n--- End of powerflow initialization ---",
215  Math::abs((**mIntfVoltage)(0,0)), Math::phaseDeg((**mIntfVoltage)(0,0)),
216  Math::abs(**mEp), Math::phaseDeg(**mEp),
217  mInitElecPower.real(), mInitElecPower.imag(),
218  **mElecActivePower, **mMechPower);
219 }

221 void EMT::Ph3::SynchronGeneratorTrStab::step(Real time) {
222  // #### Calculations on input of time step k ####
223  // Transform interface quantities to synchronously rotating DQ reference frame
224  Matrix intfVoltageDQ = parkTransformPowerInvariant(mThetaN, **mIntfVoltage);
225  Matrix intfCurrentDQ = parkTransformPowerInvariant(mThetaN, **mIntfCurrent);
226  // Update electrical power (minus sign to calculate generated power from consumed current)
227  **mElecActivePower = - 1. * (intfVoltageDQ(0, 0)*intfCurrentDQ(0, 0) + intfVoltageDQ(1, 0)*intfCurrentDQ(1, 0));
228 
229  // The damping factor Kd is adjusted to obtain a damping ratio of 0.3
230  // Real MaxElecActivePower= Math::abs(mEp) * Math::abs((**mIntfVoltage)(0,0)) / mXpd;
231  // mKd=4*0.3*sqrt(mNomOmega*mInertia*MaxElecActivePower*0.5);
232  mKd=1*mNomPower;
233 
234  // #### Calculate state for time step k+1 ####
235  // semi-implicit Euler or symplectic Euler method for mechanical equations
236  Real dOmMech = mNomOmega / (2. * **mInertia * mNomPower) * (**mMechPower - **mElecActivePower-mKd*(**mOmMech - mNomOmega));
237  if (mBehaviour == Behaviour::MNASimulation)
238  **mOmMech = **mOmMech + mTimeStep * dOmMech;
239  Real dDelta_p = **mOmMech - mNomOmega;
240  if (mBehaviour == Behaviour::MNASimulation)
241  **mDelta_p = **mDelta_p + mTimeStep * dDelta_p;
242  // Update emf - only phase changes
243  if (mBehaviour == Behaviour::MNASimulation)
244  **mEp = Complex(**mEp_abs * cos(**mDelta_p), **mEp_abs * sin(**mDelta_p));
245 
246  // Update nominal system angle
247  mThetaN = mThetaN + mTimeStep * mNomOmega;
248 
249  // mStates << Math::abs(mEp), Math::phaseDeg(mEp), mElecActivePower, mMechPower,
250  // mDelta_p, mOmMech, dOmMech, dDelta_p, (**mIntfVoltage)(0,0).real(), (**mIntfVoltage)(0,0).imag();
251  // SPDLOG_LOGGER_DEBUG(mSLog, "\nStates, time {:f}: \n{:s}", time, Logger::matrixToString(mStates));
252 }
