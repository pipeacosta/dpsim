40 void EMT::Ph3::SynchronGeneratorVBR::initialize(Real om, Real dt,
41  Real initActivePower, Real initReactivePower,
42  Real initTerminalVolt, Real initVoltAngle, Real initFieldVoltage, Real initMechPower) {
43 
44  mSystemOmega = om;
45  mSystemTimeStep = dt;
46 
47  mResistanceMat = Matrix::Zero(3, 3);
48  mResistanceMat <<
49  mRs, 0, 0,
50  0, mRs, 0,
51  0, 0, mRs;
52 
53  //Dynamic mutual inductances
54  mDLmd = 1. / (1. / mLmd + 1. / mLlfd + 1. / mLlkd);
55 
56  if (mNumDampingWindings == 2)
57  mDLmq = 1. / (1. / mLmq + 1. / mLlkq1 + 1. / mLlkq2);
58  else
59  {
60  mDLmq = 1. / (1. / mLmq + 1. / mLlkq1);
61  K1a = Matrix::Zero(2, 1);
62  K1 = Matrix::Zero(2, 1);
63  }
64 
65  mLa = (mDLmq + mDLmd) / 3.;
66  mLb = (mDLmd - mDLmq) / 3.;
67 
68  // steady state per unit initial value
69  initStatesInPerUnit(initActivePower, initReactivePower, initTerminalVolt, initVoltAngle, initFieldVoltage, initMechPower);
70 
71 
72  // Correcting variables
73  mThetaMech = mThetaMech + PI / 2;
74  mMechTorque = -mMechTorque;
75  mIq = -mIq;
76  mId = -mId;
77 
78 
79  // #### VBR Model Dynamic variables #######################################
80  CalculateAuxiliarConstants(dt*mBase_OmElec);
81 
82  if (mNumDampingWindings == 2)
83  mPsimq = mDLmq*(mPsikq1 / mLlkq1 + mPsikq2 / mLlkq2 + mIq);
84  else
85  mPsimq = mDLmq*(mPsikq1 / mLlkq1 + mIq);
86 
87  mPsimd = mDLmd*(mPsifd / mLlfd + mPsikd / mLlkd + mId);
88 
89  mDqStatorCurrents <<
90  mIq,
91  mId;
92 
93  mPsikq1kq2 <<
94  mPsikq1,
95  mPsikq2;
96  mPsifdkd <<
97  mPsifd,
98  mPsikd;
99 
100  CalculateAuxiliarVariables();
101  K1K2 << K1, K2;
102  mDVqd = K1K2*mDqStatorCurrents + h_qdr;
103  mDVq = mDVqd(0);
104  mDVd = mDVqd(1);
105 
106  mDVa = inverseParkTransform(mThetaMech, mDVq, mDVd, 0)(0);
107  mDVb = inverseParkTransform(mThetaMech, mDVq, mDVd, 0)(1);
108  mDVc = inverseParkTransform(mThetaMech, mDVq, mDVd, 0)(2);
109 
110  mDVabc <<
111  mDVa,
112  mDVb,
113  mDVc;
114 
115  mVa = inverseParkTransform(mThetaMech, mVq, mVd, mV0)(0);
116  mVb = inverseParkTransform(mThetaMech, mVq, mVd, mV0)(1);
117  mVc = inverseParkTransform(mThetaMech, mVq, mVd, mV0)(2);
118 
119  mIa = inverseParkTransform(mThetaMech, mIq, mId, mI0)(0);
120  mIb = inverseParkTransform(mThetaMech, mIq, mId, mI0)(1);
121  mIc = inverseParkTransform(mThetaMech, mIq, mId, mI0)(2);
122 
123  CalculateL();
124 }

127  stepInPerUnit(mSystemOmega, mSystemTimeStep, time, mNumericalMethod);

160 void EMT::Ph3::SynchronGeneratorVBR::stepInPerUnit(Real om, Real dt, Real time, NumericalMethod numMethod) {
161  // Estimate mechanical variables with euler
162  if (WithTurbineGovernor == true) {
163  mMechTorque = -mTurbineGovernor.step(mOmMech, 1, 300e6 / 555e6, dt);
164 
165  }
166 
167  mElecTorque = (mPsimd*mIq - mPsimq*mId);
168  mOmMech = mOmMech + dt * (1. / (2. * mH) * (mElecTorque - mMechTorque));
169  mThetaMech = mThetaMech + dt * (mOmMech* mBase_OmMech);
170 
171 
172  // Calculate equivalent Resistance and current source
173  mVabc <<
174  mVa,
175  mVb,
176  mVc;
177 
178  mIabc <<
179  mIa,
180  mIb,
181  mIc;
182 
183  mEsh_vbr = (mResistanceMat - (2 / (dt*mBase_OmElec))*mDInductanceMat)*mIabc + mDVabc - mVabc;
184 
185  CalculateL();
186 
187  mPsikq1kq2 <<
188  mPsikq1,
189  mPsikq2;
190  mPsifdkd <<
191  mPsifd,
192  mPsikd;
193 
194  CalculateAuxiliarVariables();
195 
196  R_eq_vbr = mResistanceMat + (2 / (dt*mBase_OmElec))*mDInductanceMat + K;
197  E_eq_vbr = mEsh_vbr + E_r_vbr;
198 
199  mConductanceMat = (R_eq_vbr*mBase_Z).inverse();
200  mISourceEq = R_eq_vbr.inverse()*E_eq_vbr*mBase_I;
201 }

203 void EMT::Ph3::SynchronGeneratorVBR::mnaPostStep(Matrix& rightVector, Matrix& leftVector, Real time) {
204  if ( terminalNotGrounded(0) ) {
205  mVa = Math::realFromVectorElement(leftVector, matrixNodeIndex(0)) / mBase_V;
206  }
207  else {
208  mVa = 0;
209  }
210  if ( terminalNotGrounded(1) ) {
211  mVb = Math::realFromVectorElement(leftVector, matrixNodeIndex(1)) / mBase_V;
212  }
213  else {
214  mVb = 0;
215  }
216  if ( matrixNodeIndex(2) >= 0) {
217  mVc = Math::realFromVectorElement(leftVector, matrixNodeIndex(2)) / mBase_V;
218  }
219  else {
220  mVc = 0;
221  }
222 
223  // ################ Update machine stator and rotor variables ############################
224  mVabc <<
225  mVa,
226  mVb,
227  mVc;
228 
229  mVq = parkTransform(mThetaMech, mVa, mVb, mVc)(0);
230  mVd = parkTransform(mThetaMech, mVa, mVb, mVc)(1);
231  mV0 = parkTransform(mThetaMech, mVa, mVb, mVc)(2);
232 
233  if (WithExciter == true) {
234  mVfd = mExciter.step(mVd, mVq, 1, mSystemTimeStep);
235  }
236 
237  mIabc = R_eq_vbr.inverse()*(mVabc - E_eq_vbr);
238 
239  mIa = mIabc(0);
240  mIb = mIabc(1);
241  mIc = mIabc(2);
242 
243  mIq_hist = mIq;
244  mId_hist = mId;
245 
246  mIq = parkTransform(mThetaMech, mIa, mIb, mIc)(0);
247  mId = parkTransform(mThetaMech, mIa, mIb, mIc)(1);
248  mI0 = parkTransform(mThetaMech, mIa, mIb, mIc)(2);
249 
250  // Calculate rotor flux likanges
251  if (mNumDampingWindings == 2) {
252  mDqStatorCurrents <<
253  mIq,
254  mId;
255 
256  mPsikq1kq2 = E1*mIq + E2*mPsikq1kq2 + E1*mIq_hist;
257  mPsifdkd = F1*mId + F2*mPsifdkd + F1*mId_hist + F3*mVfd;
258 
259  mPsikq1 = mPsikq1kq2(0);
260  mPsikq2 = mPsikq1kq2(1);
261  mPsifd = mPsifdkd(0);
262  mPsikd = mPsifdkd(1);
263 
264  }
265  else {
266 
267  mDqStatorCurrents <<
268  mIq,
269  mId;
270 
271  mPsikq1 = E1_1d*mIq + E2_1d*mPsikq1 + E1_1d*mIq_hist;
272  mPsifdkd = F1*mId + F2*mPsifdkd + F1*mId_hist + F3*mVfd;
273 
274  mPsifd = mPsifdkd(0);
275  mPsikd = mPsifdkd(1);
276  }
277 
278 
279  // Calculate dynamic flux likages
280  if (mNumDampingWindings == 2) {
281  mPsimq = mDLmq*(mPsikq1 / mLlkq1 + mPsikq2 / mLlkq2 + mIq);
282  }
283  else {
284  mPsimq = mDLmq*(mPsikq1 / mLlkq1 + mIq);
285  }
286 
287  mPsimd = mDLmd*(mPsifd / mLlfd + mPsikd / mLlkd + mId);
288 
289  K1K2 << K1, K2;
290  mDVqd = K1K2*mDqStatorCurrents + h_qdr;
291  mDVq = mDVqd(0);
292  mDVd = mDVqd(1);
293 
294  mDVa = inverseParkTransform(mThetaMech, mDVq, mDVd, 0)(0);
295  mDVb = inverseParkTransform(mThetaMech, mDVq, mDVd, 0)(1);
296  mDVc = inverseParkTransform(mThetaMech, mDVq, mDVd, 0)(2);
297  mDVabc <<
298  mDVa,
299  mDVb,
300  mDVc;
301 }

303 void EMT::Ph3::SynchronGeneratorVBR::CalculateL() {
304  mDInductanceMat <<
305  mLl + mLa - mLb*cos(2 * mThetaMech), -mLa / 2 - mLb*cos(2 * mThetaMech - 2 * PI / 3), -mLa / 2 - mLb*cos(2 * mThetaMech + 2 * PI / 3),
306  -mLa / 2 - mLb*cos(2 * mThetaMech - 2 * PI / 3), mLl + mLa - mLb*cos(2 * mThetaMech - 4 * PI / 3), -mLa / 2 - mLb*cos(2 * mThetaMech),
307  -mLa / 2 - mLb*cos(2 * mThetaMech + 2 * PI / 3), -mLa / 2 - mLb*cos(2 * mThetaMech), mLl + mLa - mLb*cos(2 * mThetaMech + 4 * PI / 3);
308 }

310 void EMT::Ph3::SynchronGeneratorVBR::CalculateAuxiliarConstants(Real dt) {
311 
312  b11 = (mRkq1 / mLlkq1)*(mDLmq / mLlkq1 - 1);
313  b13 = mRkq1*mDLmq / mLlkq1;
314  b31 = (mRfd / mLlfd)*(mDLmd / mLlfd - 1);
315  b32 = mRfd*mDLmd / (mLlfd*mLlkd);
316  b33 = mRfd*mDLmd / mLlfd;
317  b41 = mRkd*mDLmd / (mLlfd*mLlkd);
318  b42 = (mRkd / mLlkd)*(mDLmd / mLlkd - 1);
319  b43 = mRkd*mDLmd / mLlkd;
320 
321  c23 = mDLmd*mRfd / (mLlfd*mLlfd)*(mDLmd / mLlfd - 1) + mDLmd*mDLmd*mRkd / (mLlkd*mLlkd*mLlfd);
322  c24 = mDLmd*mRkd / (mLlkd*mLlkd)*(mDLmd / mLlkd - 1) + mDLmd*mDLmd*mRfd / (mLlfd*mLlfd*mLlkd);
323  c25 = (mRfd / (mLlfd*mLlfd) + mRkd / (mLlkd*mLlkd))*mDLmd*mDLmd;
324  c26 = mDLmd / mLlfd;
325 
326  if (mNumDampingWindings == 2) {
327  b12 = mRkq1*mDLmq / (mLlkq1*mLlkq2);
328  b21 = mRkq2*mDLmq / (mLlkq1*mLlkq2);
329  b22 = (mRkq2 / mLlkq2)*(mDLmq / mLlkq2 - 1);
330  b23 = mRkq2*mDLmq / mLlkq2;
331  c11 = mDLmq*mRkq1 / (mLlkq1*mLlkq1)*(mDLmq / mLlkq1 - 1) + mDLmq*mDLmq*mRkq2 / (mLlkq2*mLlkq2*mLlkq1);
332  c12 = mDLmq*mRkq2 / (mLlkq2*mLlkq2)*(mDLmq / mLlkq2 - 1) + mDLmq*mDLmq*mRkq1 / (mLlkq1*mLlkq1*mLlkq2);
333  c15 = (mRkq1 / (mLlkq1*mLlkq1) + mRkq2 / (mLlkq2*mLlkq2))*mDLmq*mDLmq;
334 
335  Ea <<
336  2 - dt*b11, -dt*b12,
337  -dt*b21, 2 - dt*b22;
338  E1b <<
339  dt*b13,
340  dt*b23;
341  E1 = Ea.inverse() * E1b;
342 
343  E2b <<
344  2 + dt*b11, dt*b12,
345  dt*b21, 2 + dt*b22;
346  E2 = Ea.inverse() * E2b;
347  }
348  else {
349  c11 = mDLmq*mRkq1 / (mLlkq1*mLlkq1)*(mDLmq / mLlkq1 - 1);
350  c15 = (mRkq1 / (mLlkq1*mLlkq1))*mDLmq*mDLmq;
351 
352  E1_1d = (1 / (2 - dt*b11))*dt*b13;
353  E2_1d = (1 / (2 - dt*b11))*(2 + dt*b11);
354  }
355 
356  Fa <<
357  2 - dt*b31, -dt*b32,
358  -dt*b41, 2 - dt*b42;
359  F1b <<
360  dt*b33,
361  dt*b43;
362  F1 = Fa.inverse() * F1b;
363 
364  F2b <<
365  2 + dt*b31, dt*b32,
366  dt*b41, 2 + dt*b42;
367 
368  F2 = Fa.inverse() * F2b;
369 
370  F3b <<
371  2 * dt,
372  0;
373  F3 = Fa.inverse()* F3b;
374 
375  C26 <<
376  0,
377  c26;
378 }

380 void EMT::Ph3::SynchronGeneratorVBR::CalculateAuxiliarVariables() {
381 
382  if (mNumDampingWindings == 2) {
383  c21_omega = -mOmMech*mDLmq / mLlkq1;
384  c22_omega = -mOmMech*mDLmq / mLlkq2;
385  c13_omega = mOmMech*mDLmd / mLlfd;
386  c14_omega = mOmMech*mDLmd / mLlkd;
387 
388  K1a <<
389  c11, c12,
390  c21_omega, c22_omega;
391  K1b <<
392  c15,
393  0;
394  K1 = K1a*E1 + K1b;
395  }
396  else {
397  c21_omega = -mOmMech*mDLmq / mLlkq1;
398  c13_omega = mOmMech*mDLmd / mLlfd;
399  c14_omega = mOmMech*mDLmd / mLlkd;
400 
401  K1a <<
402  c11,
403  c21_omega;
404  K1b <<
405  c15,
406  0;
407  K1 = K1a*E1_1d + K1b;
408  }
409 
410  K2a <<
411  c13_omega, c14_omega,
412  c23, c24;
413  K2b <<
414  0,
415  c25;
416  K2 = K2a*F1 + K2b;
417 
418  K <<
419  K1, K2, Matrix::Zero(2, 1),
420  0, 0, 0;
421 
422  mKrs_teta <<
423  2. / 3. * cos(mThetaMech), 2. / 3. * cos(mThetaMech - 2. * M_PI / 3.), 2. / 3. * cos(mThetaMech + 2. * M_PI / 3.),
424  2. / 3. * sin(mThetaMech), 2. / 3. * sin(mThetaMech - 2. * M_PI / 3.), 2. / 3. * sin(mThetaMech + 2. * M_PI / 3.),
425  1. / 3., 1. / 3., 1. / 3.;
426 
427  mKrs_teta_inv <<
428  cos(mThetaMech), sin(mThetaMech), 1.,
429  cos(mThetaMech - 2. * M_PI / 3.), sin(mThetaMech - 2. * M_PI / 3.), 1,
430  cos(mThetaMech + 2. * M_PI / 3.), sin(mThetaMech + 2. * M_PI / 3.), 1.;
431 
432  K = mKrs_teta_inv*K*mKrs_teta;
433 
434  if (mNumDampingWindings == 2)
435  h_qdr = K1a*E2*mPsikq1kq2 + K1a*E1*mIq + K2a*F2*mPsifdkd + K2a*F1*mId + (K2a*F3 + C26)*mVfd;
436  else
437  h_qdr = K1a*E2_1d*mPsikq1 + K1a*E1_1d*mIq + K2a*F2*mPsifdkd + K2a*F1*mId + (K2a*F3 + C26)*mVfd;
438 
439  H_qdr << h_qdr,
440  0;
441 
442  E_r_vbr = mKrs_teta_inv*H_qdr;
443 }

445 Matrix EMT::Ph3::SynchronGeneratorVBR::parkTransform(Real theta, Real a, Real b, Real c) {
446 
447  Matrix dq0vector(3, 1);
448 
449  Real q, d, zero;
450 
451  q = 2. / 3. * cos(theta)*a + 2. / 3. * cos(theta - 2. * M_PI / 3.)*b + 2. / 3. * cos(theta + 2. * M_PI / 3.)*c;
452  d = 2. / 3. * sin(theta)*a + 2. / 3. * sin(theta - 2. * M_PI / 3.)*b + 2. / 3. * sin(theta + 2. * M_PI / 3.)*c;
453  zero = 1. / 3.*a + 1. / 3.*b + 1. / 3.*c;
454 
455  dq0vector << q,
456  d,
457  zero;
458 
459  return dq0vector;
460 }

462 Matrix EMT::Ph3::SynchronGeneratorVBR::inverseParkTransform(Real theta, Real q, Real d, Real zero) {
463 
464  Matrix abcVector(3, 1);
465 
466  Real a, b, c;
467 
468  a = cos(theta)*q + sin(theta)*d + 1.*zero;
469  b = cos(theta - 2. * M_PI / 3.)*q + sin(theta - 2. * M_PI / 3.)*d + 1.*zero;
470  c = cos(theta + 2. * M_PI / 3.)*q + sin(theta + 2. * M_PI / 3.)*d + 1.*zero;
471 
472  abcVector << a,
473  b,
474  c;
475 
476  return abcVector;
