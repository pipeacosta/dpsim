36 void EMT::Ph3::SynchronGeneratorVBRSmpl::initialize(Real om, Real dt,
37  Real initActivePower, Real initReactivePower,
38  Real initTerminalVolt, Real initVoltAngle, Real initMechPower) {
39 
40  mSystemOmega = om;
41  mSystemTimeStep = dt;
42 
43  mResistanceMat = Matrix::Zero(3, 3);
44  mResistanceMat <<
45  mRs, 0, 0,
46  0, mRs, 0,
47  0, 0, mRs;
48 
49  R_load <<
50  1037.8378 / mBase_Z, 0, 0,
51  0, 1037.8378 / mBase_Z, 0,
52  0, 0, 1037.8378 / mBase_Z;
53 
54  //Dynamic mutual inductances
55 
56  mDLmd = 1. / (1. / mLmd + 1. / mLlfd + 1. / mLlkd);
57 
58  mDLmq = 1. / (1. / mLmq + 1. / mLlkq1 + 1. / mLlkq2);
59 
60  A_flux <<
61  -mRkq1 / mLlkq1*(1 - mDLmq / mLlkq1), mRkq1 / mLlkq1*(mDLmq / mLlkq2), 0, 0,
62  mRkq2 / mLlkq2*(mDLmq / mLlkq1), -mRkq2 / mLlkq2*(1 - mDLmq / mLlkq2), 0, 0,
63  0, 0, -mRfd / mLlfd*(1 - mDLmd / mLlfd), mRfd / mLlfd*(mDLmd / mLlkd),
64  0, 0, mRkd / mLlkd*(mDLmd / mLlfd), -mRkd / mLlkd*(1 - mDLmd / mLlkd);
65  B_flux <<
66  mRkq1*mDLmq / mLlkq1, 0,
67  mRkq2*mDLmq / mLlkq2, 0,
68  0, mRfd*mDLmd / mLlfd,
69  0, mRkd*mDLmd / mLlkd;
70 
71  mLa = (mDLmq + mDLmd) / 3.;
72  mLb = (mDLmd - mDLmq) / 3.;
73 
74  mDLd = mLl + mDLmd;
75  mDLq = mLl + mDLmq;
76 
77  // steady state per unit initial value
78  initStatesInPerUnit(initActivePower, initReactivePower, initTerminalVolt, initVoltAngle, initMechPower);
79 
80  // Correcting variables
81  mThetaMech = mThetaMech + PI / 2;
82  mMechTorque = -mMechTorque;
83  mIq = -mIq;
84  mId = -mId;
85 
86  // VBR Model Dynamic variables
87 
88  mDPsid = mDLmd*(mPsifd / mLlfd) + mDLmd*(mPsikd / mLlkd);
89  mDPsiq = mDLmq*(mPsikq1 / mLlkq1) + mDLmq*(mPsikq2 / mLlkq2);
90 
91  mDVq = mOmMech*mDPsid;
92  mPsimq = mDLmq*(mPsikq1 / mLlkq1 + mPsikq2 / mLlkq2 + mIq);
93 
94  mDVd = -mOmMech*mDPsiq;
95  mPsimd = mDLmd*(mPsifd / mLlfd + mPsikd / mLlkd + mId);
96 
97  mDVqd <<
98  mDVq,
99  mDVd;
100 
101  mVa = inverseParkTransform(mThetaMech, mVq, mVd, mV0)(0);
102  mVb = inverseParkTransform(mThetaMech, mVq, mVd, mV0)(1);
103  mVc = inverseParkTransform(mThetaMech, mVq, mVd, mV0)(2);
104 
105  mIa = inverseParkTransform(mThetaMech, mIq, mId, mI0)(0);
106  mIb = inverseParkTransform(mThetaMech, mIq, mId, mI0)(1);
107  mIc = inverseParkTransform(mThetaMech, mIq, mId, mI0)(2);
108 
109 }

134 void EMT::Ph3::SynchronGeneratorVBRSmpl::stepInPerUnit(Real om, Real dt, Real time, NumericalMethod numMethod)
135 {
136  // Calculate mechanical variables with euler
137  //mElecTorque = (mPsimd*mIq - mPsimq*mId);
138  mElecTorque = mVd*mId + mVq*mIq + mRs*(mId*mId + mIq*mIq);
139  mOmMech = mOmMech + dt * (1. / (2. * mH) * (mElecTorque - mMechTorque));
140  mThetaMech = mThetaMech + dt * (mOmMech* mBase_OmMech);
141 
142  Matrix R(2, 2);
143  R <<
144  -(mRs + R_load(0, 0)), -mOmMech*mDLd,
145  mOmMech*mDLq, -(mRs + R_load(1, 1));
146 
147  mDqStatorCurrents = R.inverse()*mDVqd;
148 
149  mIq = mDqStatorCurrents(0, 0);
150  mId = mDqStatorCurrents(1, 0);
151 
152  mIa_hist = mIa;
153  mIb_hist = mIb;
154  mIc_hist = mIc;
155 
156  mIa = inverseParkTransform(mThetaMech, mIq, mId, 0)(0);
157  mIb = inverseParkTransform(mThetaMech, mIq, mId, 0)(1);
158  mIc = inverseParkTransform(mThetaMech, mIq, mId, 0)(2);
159 
160  mIabc <<
161  mIa*mBase_I,
162  mIb*mBase_I,
163  mIc*mBase_I;
164 
165  mVq = -mRs*mIq - mOmMech*mDLd*mId + mDVq;
166  mVd = -mRs*mId + mOmMech*mDLq*mIq + mDVd;
167 
168  if (mHasExciter == true) {
169  mVfd = mExciter.step(mVd, mVq, 1, dt);
170  }
171 
172  C_flux <<
173  0,
174  0,
175  mVfd,
176  0;
177 
178  mRotorFlux <<
179  mPsikq1,
180  mPsikq2,
181  mPsifd,
182  mPsikd;
183 
184  mRotorFlux = Math::StateSpaceTrapezoidal(mRotorFlux, A_flux, B_flux, C_flux, dt*mBase_OmElec, mDqStatorCurrents);
185 
186  mPsikq1 = mRotorFlux(0);
187  mPsikq2 = mRotorFlux(1);
188  mPsifd = mRotorFlux(2);
189  mPsikd = mRotorFlux(3);
190 
191  // Calculate dynamic flux likages
192  mDPsiq = mDLmq*(mPsikq1 / mLlkq1) + mDLmq*(mPsikq2 / mLlkq2);
193  mPsimq = mDLmq*(mPsikq1 / mLlkq1 + mPsikq2 / mLlkq2 + mIq);
194  mDPsid = mDLmd*(mPsifd / mLlfd) + mDLmd*(mPsikd / mLlkd);
195  mPsimd = mDLmd*(mPsifd / mLlfd + mPsikd / mLlkd + mId);
196 
197  // Calculate dynamic voltages
198  mDVq = mOmMech*mDPsid;
199  mDVd = -mOmMech*mDPsiq;
200 
201  mDVqd <<
202  mDVq,
203  mDVd;
204 }

206 void EMT::Ph3::SynchronGeneratorVBRSmpl::CalculateLandpL()
207 {
208  mDInductanceMat <<
209  mLl + mLa - mLb*cos(2 * mThetaMech), -mLa / 2 - mLb*cos(2 * mThetaMech - 2 * PI / 3), -mLa / 2 - mLb*cos(2 * mThetaMech + 2 * PI / 3),
210  -mLa / 2 - mLb*cos(2 * mThetaMech - 2 * PI / 3), mLl + mLa - mLb*cos(2 * mThetaMech - 4 * PI / 3), -mLa / 2 - mLb*cos(2 * mThetaMech),
211  -mLa / 2 - mLb*cos(2 * mThetaMech + 2 * PI / 3), -mLa / 2 - mLb*cos(2 * mThetaMech), mLl + mLa - mLb*cos(2 * mThetaMech + 4 * PI / 3);
212  pmDInductanceMat <<
213  mLb*sin(2 * mThetaMech), mLb*sin(2 * mThetaMech - 2 * PI / 3), mLb*sin(2 * mThetaMech + 2 * PI / 3),
214  mLb*sin(2 * mThetaMech - 2 * PI / 3), mLb*sin(2 * mThetaMech - 4 * PI / 3), mLb*sin(2 * mThetaMech),
215  mLb*sin(2 * mThetaMech + 2 * PI / 3), mLb*sin(2 * mThetaMech), mLb*sin(2 * mThetaMech + 4 * PI / 3);
216  pmDInductanceMat = pmDInductanceMat * 2 * mOmMech;

219 Matrix EMT::Ph3::SynchronGeneratorVBRSmpl::parkTransform(Real theta, Real a, Real b, Real c)
220 {
221  Matrix dq0vector(3, 1);
222 
223  Real q, d;
224 
225  q = 2. / 3. * cos(theta)*a + 2. / 3. * cos(theta - 2. * M_PI / 3.)*b + 2. / 3. * cos(theta + 2. * M_PI / 3.)*c;
226  d = 2. / 3. * sin(theta)*a + 2. / 3. * sin(theta - 2. * M_PI / 3.)*b + 2. / 3. * sin(theta + 2. * M_PI / 3.)*c;
227 
228  dq0vector << q,
229  d,
230  0;
231 
232  return dq0vector;
