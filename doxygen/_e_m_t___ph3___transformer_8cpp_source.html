50 void EMT::Ph3::Transformer::initializeFromNodesAndTerminals(Real frequency) {
51 
52  // Component parameters are referred to higher voltage side.
53  // Switch terminals to have terminal 0 at higher voltage side
54  // if transformer is connected the other way around.
55  if (Math::abs(**mRatio) < 1.) {
56  **mRatio = 1. / **mRatio;
57  std::shared_ptr<SimTerminal<Real>> tmp = mTerminals[0];
58  mTerminals[0] = mTerminals[1];
59  mTerminals[1] = tmp;
60  Real tmpVolt = mNominalVoltageEnd1;
61  mNominalVoltageEnd1 = mNominalVoltageEnd2;
62  mNominalVoltageEnd2 = tmpVolt;
63  mSLog->info("Switching terminals to have first terminal at higher voltage side. Updated parameters: ");
64  mSLog->info("Nominal Voltage End 1 = {} [V] Nominal Voltage End 2 = {} [V]", mNominalVoltageEnd1, mNominalVoltageEnd2);
65  mSLog->info("Tap Ratio = {} [ ] Phase Shift = {} [deg]", std::abs(**mRatio), std::arg(**mRatio));
66  }
67 
68  // Set initial voltage of virtual node in between
69  mVirtualNodes[0]->setInitialVoltage(initialSingleVoltage(1) * **mRatio);
70 
71  // Static calculations from load flow data
72  Real omega = 2. * PI * frequency;
73  MatrixComp impedance = MatrixComp::Zero(3, 3);
74  impedance <<
75  Complex(mResistance(0, 0), omega * mInductance(0, 0)), Complex(mResistance(0, 1), omega * mInductance(0, 1)), Complex(mResistance(0, 2), omega * mInductance(0, 2)),
76  Complex(mResistance(1, 0), omega * mInductance(1, 0)), Complex(mResistance(1, 1), omega * mInductance(1, 1)), Complex(mResistance(1, 2), omega * mInductance(1, 2)),
77  Complex(mResistance(2, 0), omega * mInductance(2, 0)), Complex(mResistance(2, 1), omega * mInductance(2, 1)), Complex(mResistance(2, 2), omega * mInductance(2, 2));
78 
79  mSLog->info("Resistance (referred to higher voltage side) = {} [Ohm]", Logger::matrixToString(mResistance));
80  mSLog->info("Inductance (referred to higher voltage side) = {} [H]", Logger::matrixToString(mInductance));
81  mSLog->info("Reactance (referred to higher voltage side) = {} [Ohm]", Logger::matrixToString(omega * mInductance));
82 
83  MatrixComp vInitABC = MatrixComp::Zero(3, 1);
84  vInitABC(0, 0) = mVirtualNodes[0]->initialSingleVoltage() - RMS3PH_TO_PEAK1PH * initialSingleVoltage(0);
85  vInitABC(1, 0) = vInitABC(0, 0) * SHIFT_TO_PHASE_B;
86  vInitABC(2, 0) = vInitABC(0, 0) * SHIFT_TO_PHASE_C;
87 
88  MatrixComp iInit = impedance.inverse() * vInitABC;
89  **mIntfCurrent = iInit.real();
90  **mIntfVoltage = vInitABC.real();
91 
92  // Create series sub components
93  mSubInductor = std::make_shared<EMT::Ph3::Inductor>(**mName + "_ind", mLogLevel);
94  mSubComponents.push_back(mSubInductor);
95  mSubInductor->setParameters(mInductance);
96 
97  if (mNumVirtualNodes == 3) {
98  mVirtualNodes[2]->setInitialVoltage(initialSingleVoltage(0));
99  mSubResistor = std::make_shared<EMT::Ph3::Resistor>(**mName + "_res", mLogLevel);
100  mSubComponents.push_back(mSubResistor);
101  mSubResistor->setParameters(mResistance);
102  mSubResistor->connect({ node(0), mVirtualNodes[2] });
103  mSubInductor->connect({ mVirtualNodes[2], mVirtualNodes[0] });
104  }
105  else {
106  mSubInductor->connect({ node(0), mVirtualNodes[0] });
107  }
108 
109  // Create parallel sub components (three-phase power)
110  Real pSnub = P_SNUB_TRANSFORMER*mRatedPower;
111  Real qSnub = Q_SNUB_TRANSFORMER*mRatedPower;
112 
113  // A snubber conductance is added on the higher voltage side
114  Real snubberResistance1 = std::pow(std::abs(mNominalVoltageEnd1),2) / pSnub;
115  mSnubberResistance1 = Math::singlePhaseParameterToThreePhase(snubberResistance1);
116  mSubSnubResistor1 = std::make_shared<EMT::Ph3::Resistor>(**mName + "_snub_res1", mLogLevel);
117  mSubSnubResistor1->setParameters(mSnubberResistance1);
118  mSubSnubResistor1->connect({ node(0), EMT::SimNode::GND });
119  mSLog->info("Snubber Resistance 1 (connected to higher voltage side {}) = {} [Ohm]", node(0)->name(), Logger::matrixToString(mSnubberResistance1));
120  mSubComponents.push_back(mSubSnubResistor1);
121 
122  // A snubber conductance is added on the lower voltage side
123  Real snubberResistance2 = std::pow(std::abs(mNominalVoltageEnd2),2) / pSnub;
124  mSnubberResistance2 = Math::singlePhaseParameterToThreePhase(snubberResistance2);
125  mSubSnubResistor2 = std::make_shared<EMT::Ph3::Resistor>(**mName + "_snub_res2", mLogLevel);
126  mSubSnubResistor2->setParameters(mSnubberResistance2);
127  mSubSnubResistor2->connect({ node(1), EMT::SimNode::GND });
128  mSLog->info("Snubber Resistance 2 (connected to lower voltage side {}) = {} [Ohm]", node(1)->name(), Logger::matrixToString(mSnubberResistance2));
129  mSubComponents.push_back(mSubSnubResistor2);
130 
131  // // A snubber capacitance is added to higher voltage side (not used as capacitor at high voltage side made it worse)
132  // Real snubberCapacitance1 = qSnub / std::pow(std::abs(mNominalVoltageEnd1),2) / omega;
133  // mSnubberCapacitance1 = Math::singlePhaseParameterToThreePhase*(snubberCapacitance1);
134  // mSubSnubCapacitor1 = std::make_shared<EMT::Ph3::Capacitor>(**mName + "_snub_cap1", mLogLevel);
135  // mSubSnubCapacitor1->setParameters(mSnubberCapacitance1);
136  // mSubSnubCapacitor1->connect({ node(0), EMT::SimNode::GND });
137  // mSLog->info("Snubber Capacitance 1 (connected to higher voltage side {}) = \n{} [F] \n ", node(0)->name(), Logger::matrixToString(mSnubberCapacitance1));
138  // mSubComponents.push_back(mSubSnubCapacitor1);
139 
140  // A snubber capacitance is added to lower voltage side
141  Real snubberCapacitance2 = qSnub / std::pow(std::abs(mNominalVoltageEnd2),2) / omega;
142  mSnubberCapacitance2 = Math::singlePhaseParameterToThreePhase(snubberCapacitance2);
143  mSubSnubCapacitor2 = std::make_shared<EMT::Ph3::Capacitor>(**mName + "_snub_cap2", mLogLevel);
144  mSubSnubCapacitor2->setParameters(mSnubberCapacitance2);
145  mSubSnubCapacitor2->connect({ node(1), EMT::SimNode::GND });
146  mSLog->info("Snubber Capacitance 2 (connected to lower voltage side {}) = {} [F]", node(1)->name(), Logger::matrixToString(mSnubberCapacitance2));
147  mSubComponents.push_back(mSubSnubCapacitor2);
148 
149  // Initialize electrical subcomponents
150  mSLog->info("Electrical subcomponents: ");
151  for (auto subcomp: mSubComponents) {
152  mSLog->info("- {}", subcomp->name());
153  subcomp->initialize(mFrequencies);
154  subcomp->initializeFromNodesAndTerminals(frequency);
155  }
156 
157  mSLog->info(
158  "\n--- Initialization from powerflow ---"
159  "\nVoltage across: {:s}"
160  "\nCurrent: {:s}"
161  "\nTerminal 0 voltage: {:s}"
162  "\nTerminal 1 voltage: {:s}"
163  "\nVirtual Node 1 voltage: {:s}"
164  "\n--- Initialization from powerflow finished ---",
165  Logger::matrixToString(**mIntfVoltage),
166  Logger::matrixToString(**mIntfCurrent),
167  Logger::phasorToString(RMS3PH_TO_PEAK1PH * initialSingleVoltage(0)),
168  Logger::phasorToString(RMS3PH_TO_PEAK1PH * initialSingleVoltage(1)),
169  Logger::phasorToString(RMS3PH_TO_PEAK1PH * mVirtualNodes[0]->initialSingleVoltage()));
170 }

192 void EMT::Ph3::Transformer::mnaApplySystemMatrixStamp(Matrix& systemMatrix) {
193  // Ideal transformer equations
194  if (terminalNotGrounded(0)) {
195  Math::setMatrixElement(systemMatrix, mVirtualNodes[0]->matrixNodeIndex(PhaseType::A), mVirtualNodes[1]->matrixNodeIndex(PhaseType::A), -1.);
196  Math::setMatrixElement(systemMatrix, mVirtualNodes[0]->matrixNodeIndex(PhaseType::B), mVirtualNodes[1]->matrixNodeIndex(PhaseType::B), -1.);
197  Math::setMatrixElement(systemMatrix, mVirtualNodes[0]->matrixNodeIndex(PhaseType::C), mVirtualNodes[1]->matrixNodeIndex(PhaseType::C), -1.);
198 
199  Math::setMatrixElement(systemMatrix, mVirtualNodes[1]->matrixNodeIndex(PhaseType::A), mVirtualNodes[0]->matrixNodeIndex(PhaseType::A), 1.);
200  Math::setMatrixElement(systemMatrix, mVirtualNodes[1]->matrixNodeIndex(PhaseType::B), mVirtualNodes[0]->matrixNodeIndex(PhaseType::B), 1.);
201  Math::setMatrixElement(systemMatrix, mVirtualNodes[1]->matrixNodeIndex(PhaseType::C), mVirtualNodes[0]->matrixNodeIndex(PhaseType::C), 1.);
202 
203  }
204  if (terminalNotGrounded(1)) {
205  Math::setMatrixElement(systemMatrix, matrixNodeIndex(1, 0), mVirtualNodes[1]->matrixNodeIndex(PhaseType::A), (**mRatio).real());
206  Math::setMatrixElement(systemMatrix, matrixNodeIndex(1, 1), mVirtualNodes[1]->matrixNodeIndex(PhaseType::B), (**mRatio).real());
207  Math::setMatrixElement(systemMatrix, matrixNodeIndex(1, 2), mVirtualNodes[1]->matrixNodeIndex(PhaseType::C), (**mRatio).real());
208  Math::setMatrixElement(systemMatrix, mVirtualNodes[1]->matrixNodeIndex(PhaseType::A), matrixNodeIndex(1, 0), -(**mRatio).real());
209  Math::setMatrixElement(systemMatrix, mVirtualNodes[1]->matrixNodeIndex(PhaseType::B), matrixNodeIndex(1, 1), -(**mRatio).real());
210  Math::setMatrixElement(systemMatrix, mVirtualNodes[1]->matrixNodeIndex(PhaseType::C), matrixNodeIndex(1, 2), -(**mRatio).real());
211  }
212 
213  // Add subcomps to system matrix
214  for (auto subcomp: mSubComponents)
215  if (auto mnasubcomp = std::dynamic_pointer_cast<MNAInterface>(subcomp))
216  mnasubcomp->mnaApplySystemMatrixStamp(systemMatrix);
217 
218  if (terminalNotGrounded(0)) {
219  mSLog->info("Add {:s} to system at ({:d},{:d})", Logger::complexToString(Complex(-1.0, 0)),
220  mVirtualNodes[0]->matrixNodeIndex(PhaseType::A), mVirtualNodes[1]->matrixNodeIndex(PhaseType::A));
221  mSLog->info("Add {:s} to system at ({:d},{:d})", Logger::complexToString(Complex(-1.0, 0)),
222  mVirtualNodes[0]->matrixNodeIndex(PhaseType::B), mVirtualNodes[1]->matrixNodeIndex(PhaseType::B));
223  mSLog->info("Add {:s} to system at ({:d},{:d})", Logger::complexToString(Complex(-1.0, 0)),
224  mVirtualNodes[0]->matrixNodeIndex(PhaseType::C), mVirtualNodes[1]->matrixNodeIndex(PhaseType::C));
225 
226  mSLog->info("Add {:s} to system at ({:d},{:d})", Logger::complexToString(Complex(1.0, 0)),
227  mVirtualNodes[1]->matrixNodeIndex(PhaseType::A), mVirtualNodes[0]->matrixNodeIndex(PhaseType::A));
228  mSLog->info("Add {:s} to system at ({:d},{:d})", Logger::complexToString(Complex(1.0, 0)),
229  mVirtualNodes[1]->matrixNodeIndex(PhaseType::B), mVirtualNodes[0]->matrixNodeIndex(PhaseType::B));
230  mSLog->info("Add {:s} to system at ({:d},{:d})", Logger::complexToString(Complex(1.0, 0)),
231  mVirtualNodes[1]->matrixNodeIndex(PhaseType::C), mVirtualNodes[0]->matrixNodeIndex(PhaseType::C));
232  }
233  if (terminalNotGrounded(1)) {
234  mSLog->info("Add {:s} to system at ({:d},{:d})", Logger::complexToString(**mRatio),
235  matrixNodeIndex(1, 0), mVirtualNodes[1]->matrixNodeIndex(PhaseType::A));
236  mSLog->info("Add {:s} to system at ({:d},{:d})", Logger::complexToString(**mRatio),
237  matrixNodeIndex(1, 1), mVirtualNodes[1]->matrixNodeIndex(PhaseType::B));
238  mSLog->info("Add {:s} to system at ({:d},{:d})", Logger::complexToString(**mRatio),
239  matrixNodeIndex(1, 2), mVirtualNodes[1]->matrixNodeIndex(PhaseType::C));
240 
241  mSLog->info("Add {:s} to system at ({:d},{:d})", Logger::complexToString(- **mRatio),
242  mVirtualNodes[1]->matrixNodeIndex(PhaseType::A), matrixNodeIndex(1, 0));
243  mSLog->info("Add {:s} to system at ({:d},{:d})", Logger::complexToString(- **mRatio),
244  mVirtualNodes[1]->matrixNodeIndex(PhaseType::B), matrixNodeIndex(1, 1));
245  mSLog->info("Add {:s} to system at ({:d},{:d})", Logger::complexToString(- **mRatio),
246  mVirtualNodes[1]->matrixNodeIndex(PhaseType::C), matrixNodeIndex(1, 2));
247  }

302 void EMT::Ph3::Transformer::mnaUpdateVoltage(const Matrix& leftVector) {
303  // v1 - v0
304  **mIntfVoltage = Matrix::Zero(3, 1);
305  if (terminalNotGrounded(1)) {
306  (**mIntfVoltage)(0, 0) = Math::realFromVectorElement(leftVector, matrixNodeIndex(1, 0));
307  (**mIntfVoltage)(1, 0) = Math::realFromVectorElement(leftVector, matrixNodeIndex(1, 1));
308  (**mIntfVoltage)(2, 0) = Math::realFromVectorElement(leftVector, matrixNodeIndex(1, 2));
309  }
310  if (terminalNotGrounded(0)) {
311  (**mIntfVoltage)(0, 0) = (**mIntfVoltage)(0, 0) - Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 0));
312  (**mIntfVoltage)(1, 0) = (**mIntfVoltage)(1, 0) - Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 1));
313  (**mIntfVoltage)(2, 0) = (**mIntfVoltage)(2, 0) - Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 2));
314  }
