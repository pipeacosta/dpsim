69 void EMT::Ph3::VoltageSource::initializeFromNodesAndTerminals(Real frequency) {
70  mSLog->info("\n--- Initialization from node voltages ---");
71  // TODO: this approach currently overwrites voltage reference set from outside, when not using setParameters
72  if (!mParametersSet) {
73  auto srcSigSine = Signal::SineWaveGenerator::make(mName + "_sw");
74  // Complex(1,0) is used as initialPhasor for signal generator as only phase is used
75  srcSigSine->setParameters(Complex(1,0), frequency);
76  mSrcSig = srcSigSine;

81  mSLog->info("\nReference voltage: {:s}"

85  Logger::phasorToString(initialSingleVoltage(0)),

154 void EMT::Ph3::VoltageSource::updateVoltage(Real time) {
155  if(mSrcSig != nullptr) {
156  mSrcSig->step(time);
157  for(int i = 0; i < 3; i++) {
158  mIntfVoltage(i, 0) = RMS3PH_TO_PEAK1PH * Math::abs(attribute<MatrixComp>("V_ref")->get()(i, 0))
159  * cos(Math::phase(mSrcSig->getSignal()) + Math::phase(attribute<MatrixComp>("V_ref")->get()(i, 0)));
160  }
161  } else {
162  mIntfVoltage = RMS3PH_TO_PEAK1PH * attribute<MatrixComp>("V_ref")->get().real();
163  }
164  mSLog->debug(
165  "\nUpdate Voltage: {:s}",
166  Logger::matrixToString(mIntfVoltage)
167  );
168 }
