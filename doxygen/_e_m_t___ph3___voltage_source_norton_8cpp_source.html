54 void EMT::Ph3::VoltageSourceNorton::mnaApplySystemMatrixStamp(Matrix& systemMatrix) {
55  // Apply matrix stamp for equivalent resistance
56  if (terminalNotGrounded(0)){
57  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0, 0), matrixNodeIndex(0, 0), mConductance);
58  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0, 1), matrixNodeIndex(0, 1), mConductance);
59  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0, 2), matrixNodeIndex(0, 2), mConductance);
60  }
61  if (terminalNotGrounded(1)){
62  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1, 0), matrixNodeIndex(1, 0), mConductance);
63  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1, 1), matrixNodeIndex(1, 1), mConductance);
64  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1, 2), matrixNodeIndex(1, 2), mConductance);
65  }
66  if (terminalNotGrounded(0) && terminalNotGrounded(1)) {
67  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0, 0), matrixNodeIndex(1, 0), -mConductance);
68  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1, 0), matrixNodeIndex(0, 0), -mConductance);
69 
70  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0, 1), matrixNodeIndex(1, 1), -mConductance);
71  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1, 1), matrixNodeIndex(0, 1), -mConductance);
72 
73  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0, 2), matrixNodeIndex(1, 2), -mConductance);
74  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1, 2), matrixNodeIndex(0, 2), -mConductance);
75  }
76 }

92 void EMT::Ph3::VoltageSourceNorton::updateState(Real time) {
93  // Check if set source was called
94  if (Math::abs(**mVoltageRef) > 0) {
95  (**mIntfVoltage)(0, 0) = Math::abs(**mVoltageRef) * cos(2. * PI * **mSrcFreq * time + Math::phase(**mVoltageRef));
96  (**mIntfVoltage)(1, 0) = Math::abs(**mVoltageRef) * cos(2. * PI * **mSrcFreq * time + Math::phase(**mVoltageRef) - 2. / 3. * M_PI);
97  (**mIntfVoltage)(2, 0) = Math::abs(**mVoltageRef) * cos(2. * PI * **mSrcFreq * time + Math::phase(**mVoltageRef) + 2. / 3. * M_PI);
98  }
99 
100  mEquivCurrent = **mIntfVoltage / **mResistance;
101 }

104 void EMT::Ph3::VoltageSourceNorton::mnaUpdateVoltage(const Matrix& leftVector) {
105  // Calculate v1 - v0
106  if (terminalNotGrounded(1)) {
107  (**mIntfVoltage)(0, 0) = Math::realFromVectorElement(leftVector, matrixNodeIndex(1, 0));
108  (**mIntfVoltage)(1, 0) = Math::realFromVectorElement(leftVector, matrixNodeIndex(1, 1));
109  (**mIntfVoltage)(2, 0) = Math::realFromVectorElement(leftVector, matrixNodeIndex(1, 2));
110  }
111  if (terminalNotGrounded(0)) {
112  (**mIntfVoltage)(0, 0) = (**mIntfVoltage)(0, 0) - Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 0));
113  (**mIntfVoltage)(1, 0) = (**mIntfVoltage)(1, 0) - Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 1));
114  (**mIntfVoltage)(2, 0) = (**mIntfVoltage)(2, 0) - Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 2));
115  }
