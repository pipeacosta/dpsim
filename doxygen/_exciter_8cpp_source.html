68 Real Signal::Exciter::step(Real mVd, Real mVq, Real Vref, Real dt) {
69  // Voltage magnitude calculation
70  mVh = sqrt(pow(mVd, 2.) + pow(mVq, 2.));
71 
72  // Voltage Transducer equation
73  mVm = Math::StateSpaceEuler(mVm, -1 / mTr, 1 / mTr, dt, mVh);
74 
75  // Stabilizing feedback equation
76  mVis = Math::StateSpaceEuler(mVis, -1 / mTf, mKf / mTe / mTf, dt, mVr - mVse - mVf*mKe);
77 
78  // Voltage regulator equation
79  mVr = Math::StateSpaceEuler(mVr, -1 / mTa, mKa / mTa, dt, Vref - mVm - mVis + mVf_init);
80 
81  // Voltage regulator limiter
82  // Vr,max and Vr,min parameters
83  // from M. Eremia, "Handbook of Electrical Power System Dynamics", 2013, p.96
84  // TODO: fix hard-coded limiter values
85  if (mVr > 1)
86  mVr = 1;
87  else if (mVr < -0.9)
88  mVr = -0.9;
89 
90  // Exciter saturation
91  // Se(Ef1), Ef1, Se(Ef2) and Ef2
92  // from M. Eremia, "Handbook of Electrical Power System Dynamics", 2013, p.96
93  // TODO: fix hard-coded saturation values
94  if (mVf <= 2.3)
95  mVse = (0.1 / 2.3)*mVf;
96  else
97  mVse = (0.33 / 3.1)*mVf;
98  mVse = mVse*mVf;
99 
100  // Exciter equation
101  mVf = Math::StateSpaceEuler(mVf, - mKe / mTe, 1 / mTe, dt, mVr - mVse);
102 
103  return mVf;
104 }
