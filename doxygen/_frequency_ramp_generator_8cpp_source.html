31 void Signal::FrequencyRampGenerator::step(Real time) {
32  if(mUseAbsoluteCalc) {
33  stepAbsolute(time);
34  return;
35  }
36 
37  Real currPhase;
38  Real currFreq;
39  Real timestep = time - mOldTime;
40  mOldTime = time;
41 
42  currPhase = Math::phase(attribute<Complex>("sigOut")->get());
43 
44  if(time <= mTimeStart) {
45  currFreq = mFreqStart;
46  } else if(time <= mTimeStart + mDuration) {
47  currFreq = mFreqStart + mRocof * (time - mTimeStart);
48  } else {
49  currFreq = mFreqEnd;
50  }
51  currPhase += 2. * PI * currFreq * timestep;
52 
53  attribute<Complex>("sigOut")->set(mMagnitude * Complex(cos(currPhase), sin(currPhase)));
54  attribute<Real>("freq")->set(currFreq);
55 }

57 void Signal::FrequencyRampGenerator::stepAbsolute(Real time) {
58  Real currPhase = mInitialPhase + 2 * PI * time * mFreqStart;
59  Real currFreq = mFreqStart;
60 
61  if(time > mTimeStart + mDuration) {
62  currPhase += 2 * PI * mRocof / 2 * pow(mDuration, 2);
63  currPhase += 2 * PI * mRocof * mDuration * (time - (mTimeStart + mDuration));
64  currFreq = mFreqEnd;
65  } else if(time > mTimeStart) {
66  if(mSmooth) { // cos shape
67  // the phase is calculated as 2pi times the integral of the frequency term below
68  currPhase += 2 * PI * (mFreqEnd - mFreqStart) / 2 * ((time-mTimeStart) - 2 * mDuration / (2 * PI) * sin(2 * PI / (2 * mDuration) * (time-mTimeStart)));
69  currFreq += (mFreqEnd - mFreqStart) / 2 * (1 - cos(2 * PI / (2 * mDuration) * (time-mTimeStart)));
70  } else { // linear
71  currPhase += 2 * PI * mRocof * pow(time - mTimeStart, 2) / 2;
72  currFreq += mRocof * (time - mTimeStart);
73  }
74  }
75 
76  attribute<Complex>("sigOut")->set(mMagnitude * Complex(cos(currPhase), sin(currPhase)));
77  attribute<Real>("freq")->set(currFreq);
78 }
