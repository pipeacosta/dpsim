62 void Integrator::signalStep(Real time, Int timeStepCount) {
63  mInputCurr = attribute<Real>("input_ref")->get();
64 
65  mSLog->info("Time {}:", time);
66  mSLog->info("Input values: inputCurr = {}, inputPrev = {}, statePrev = {}", mInputCurr, mInputPrev, mStatePrev);
67 
68  mStateCurr = mStatePrev + mTimeStep/2.0*mInputCurr + mTimeStep/2.0*mInputPrev;
69  mOutputCurr = mStateCurr;
70 
71  mSLog->info("State values: stateCurr = {}", mStateCurr);
72  mSLog->info("Output values: outputCurr = {}:", mOutputCurr);
