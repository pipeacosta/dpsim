34 PyObject* Python::LoadCim(PyObject* self, PyObject* args, PyObject *kwargs) {
35 #ifdef WITH_CIM
36  CPS::Real frequency = 50;
37  PyObject *filenames;
38  PyBytesObject *filename;
39  std::list<fs::path> cimFiles;
40  int logLevel = (int) CPS::Logger::Level::info;
41  const char *name;
42 
43  const char *kwlist[] = {"name", "files", "frequency", "log_level", nullptr};
44 
45  if (!PyArg_ParseTupleAndKeywords(args, kwargs, "sO|di", (char **) kwlist, &name, &filenames, &frequency, &logLevel))
46  return nullptr;
47 
48  if (PyList_Check(filenames)) {
49  for (int i = 0; i < PyList_Size(filenames); i++) {
50  if (!PyUnicode_FSConverter(PyList_GetItem(filenames, i), &filename)) {
51  PyErr_SetString(PyExc_TypeError, "First argument must be filename or list of filenames");
52  return nullptr;
53  }
54  cimFiles.emplace_back(PyBytes_AsString((PyObject*) filename));
55  Py_DECREF(filename);
56  }
57  }
58  else if (PyBytes_Check(filenames)) {
59  if (!PyUnicode_FSConverter(filenames, &filename)) {
60  PyErr_SetString(PyExc_TypeError, "First argument must be filename or list of filenames");
61  return nullptr;
62  }
63  cimFiles.push_back(PyBytes_AsString((PyObject*) filename));
64  }
65  else {
66  PyErr_SetString(PyExc_TypeError, "First argument must be filename or list of filenames");
67  return nullptr;
68  }
69 
70  if (cimFiles.size() == 0) {
71  PyErr_SetString(PyExc_TypeError, "First argument must be filename or list of filenames");
72  return nullptr;
73  }
74 
75  CPS::CIM::Reader reader(name, (CPS::Logger::Level) logLevel, (CPS::Logger::Level) logLevel);
76 
77  DPsim::Python::SystemTopology *pySys = PyObject_New(DPsim::Python::SystemTopology, &DPsim::Python::SystemTopology::type);
78 
79  using SharedSysPtr = std::shared_ptr<CPS::SystemTopology>;
80 
81  new (&pySys->sys) SharedSysPtr();
82  pySys->pyComponentDict = PyDict_New();
83  pySys->pyNodeDict = PyDict_New();
84 
85  try {
86  pySys->sys = std::make_shared<CPS::SystemTopology>(reader.loadCIM(frequency, cimFiles));
87  }
88  catch (const CPS::CIM::InvalidTopology &) {
89  PyErr_SetString(PyExc_TypeError, "The topology of the CIM model is invalid");
90  return nullptr;
91  }
92 
93  // Fill pyComponentDict, pyNodeDict
94  for (auto comp : pySys->sys->mComponents) {
95  PyObject *pyName = PyUnicode_FromString(comp->uid().c_str());
96 
97  if (PyDict_Contains(pySys->pyComponentDict, pyName)) {
98  PyErr_Format(PyExc_AttributeError, "Duplicated Component name");
99  return nullptr;
100  }
101 
102  Component *pyComp = PyObject_New(Component, &Component::type);
103  Component::init(pyComp);
104 
105  pyComp->comp = comp;
106 
107  PyDict_SetItem(pySys->pyComponentDict, pyName, (PyObject *) pyComp);
108  }
109 
110  for (auto node : pySys->sys->mNodes) {
111  PyObject *pyName = PyUnicode_FromString(node->name().c_str());
112 
113  if (PyDict_Contains(pySys->pyNodeDict, pyName)) {
114  PyErr_Format(PyExc_AttributeError, "Duplicated Node name");
115  return nullptr;
116  }
117 
118  auto dpNode = std::dynamic_pointer_cast<CPS::SimNode<CPS::Complex>>(node);
119  if (dpNode) {
120  Python::Node<CPS::Complex> *pyNode = PyObject_New(Python::Node<CPS::Complex>, &Python::Node<CPS::Complex>::type);
121 
122  using SharedNodePtr = std::shared_ptr<CPS::SimNode<CPS::Complex>>;
123  new (&pyNode->node) SharedNodePtr();
124 
125  pyNode->node = dpNode;
126 
127  PyDict_SetItem(pySys->pyNodeDict, pyName, (PyObject*) pyNode);
128  continue;
129  }
130 
131  auto emtNode = std::dynamic_pointer_cast<CPS::SimNode<CPS::Real>>(node);
132  if (emtNode) {
133  Python::Node<CPS::Real> *pyNode = PyObject_New(Python::Node<CPS::Real>, &Python::Node<CPS::Real>::type);
134 
135  using SharedNodePtr = std::shared_ptr<CPS::SimNode<CPS::Real>>;
136  new (&pyNode->node) SharedNodePtr();
137 
138  pyNode->node = emtNode;
139 
140  PyDict_SetItem(pySys->pyNodeDict, pyName, (PyObject*) pyNode);
141  continue;
142  }
143 
144  return nullptr;
145  }
146 
147  Py_INCREF(pySys);
148  return (PyObject *) pySys;
149 #else
150  PyErr_SetString(PyExc_NotImplementedError, "not implemented on this platform");
151  return nullptr;
152 #endif
153 }
