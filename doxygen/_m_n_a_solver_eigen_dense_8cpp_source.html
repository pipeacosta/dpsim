32 template <typename VarType>
33 void MnaSolverEigenDense<VarType>::switchedMatrixStamp(std::size_t index, MNAInterface::List& components) {
34 
35  auto bit = std::bitset<SWITCH_NUM>(index);
36  auto& sys = mSwitchedMatrices[bit][0];
37 
38  for (auto comp : components)
39  comp->mnaApplySystemMatrixStamp(sys);
40 
41  for (UInt i = 0; i < mSwitches.size(); ++i)
42  mSwitches[i]->mnaApplySwitchSystemMatrixStamp(bit[i], sys, 0);
43 
44  // Compute LU-factorization for system matrix

64 template <>
65 void MnaSolverEigenDense<Real>::createEmptySystemMatrix() {
66  if (mSwitches.size() > SWITCH_NUM)
67  throw SystemError("Too many Switches.");
68 
69  for (std::size_t i = 0; i < (1ULL << mSwitches.size()); i++) {
70  auto bit = std::bitset<SWITCH_NUM>(i);
71  mSwitchedMatrices[bit].push_back(Matrix::Zero(mNumMatrixNodeIndices, mNumMatrixNodeIndices));
72  mLuFactorizations[bit].push_back(Eigen::PartialPivLU<Matrix>());
73  }
74 }

76 template <>
77 void MnaSolverEigenDense<Complex>::createEmptySystemMatrix() {
78  if (mSwitches.size() > SWITCH_NUM)
79  throw SystemError("Too many Switches.");
80 
81  if (mFrequencyParallel) {
82  for (UInt i = 0; i < std::pow(2,mSwitches.size()); ++i) {
83  for(Int freq = 0; freq < mSystem.mFrequencies.size(); ++freq) {
84  auto bit = std::bitset<SWITCH_NUM>(i);
85  mSwitchedMatrices[bit].push_back(Matrix::Zero(2*(mNumMatrixNodeIndices), 2*(mNumMatrixNodeIndices)));
86  mLuFactorizations[bit].push_back(Eigen::PartialPivLU<Matrix>());
87  }
88  }
89  }
90  else {
91  for (std::size_t i = 0; i < (1ULL << mSwitches.size()); i++) {
92  auto bit = std::bitset<SWITCH_NUM>(i);
93  mSwitchedMatrices[bit].push_back(Matrix::Zero(2*(mNumTotalMatrixNodeIndices), 2*(mNumTotalMatrixNodeIndices)));
94  mLuFactorizations[bit].push_back(Eigen::PartialPivLU<Matrix>());
95  }
96  }
97 }

118 void MnaSolverEigenDense<VarType>::solve(Real time, Int timeStepCount) {
119  // Reset source vector
120  mRightSideVector.setZero();
121 
122  // Add together the right side vector (computed by the components'
123  // pre-step tasks)
124  for (auto stamp : mRightVectorStamps)
125  mRightSideVector += *stamp;
126 
127  if (!mIsInInitialization)
128  MnaSolver<VarType>::updateSwitchStatus();
129 
130  if (mSwitchedMatrices.size() > 0)
131  mLeftSideVector = mLuFactorizations[mCurrentSwitchStatus][0].solve(mRightSideVector);
132 
133  // TODO split into separate task? (dependent on x, updating all v attributes)
134  for (UInt nodeIdx = 0; nodeIdx < mNumNetNodes; ++nodeIdx)
135  mNodes[nodeIdx]->mnaUpdateVoltage(mLeftSideVector);

140 template <typename VarType>
141 void MnaSolverEigenDense<VarType>::solveWithHarmonics(Real time, Int timeStepCount, Int freqIdx) {
142  mRightSideVectorHarm[freqIdx].setZero();
143 
144  // Sum of right side vectors (computed by the components' pre-step tasks)
145  for (auto stamp : mRightVectorStamps)
146  mRightSideVectorHarm[freqIdx] += stamp->col(freqIdx);
147 
148  mLeftSideVectorHarm[freqIdx] = mLuFactorizations[mCurrentSwitchStatus][freqIdx].solve(mRightSideVectorHarm[freqIdx]);
