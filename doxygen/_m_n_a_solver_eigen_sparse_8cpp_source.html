33 template <typename VarType>
34 void MnaSolverEigenSparse<VarType>::switchedMatrixStamp(std::size_t index, std::vector<std::shared_ptr<CPS::MNAInterface>>& comp)
35 {
36  auto bit = std::bitset<SWITCH_NUM>(index);
37  auto& sys = mSwitchedMatrices[bit][0];
38  for (auto comp : comp) {
39  comp->mnaApplySystemMatrixStamp(sys);
40  }
41  for (UInt i = 0; i < mSwitches.size(); ++i)
42  mSwitches[i]->mnaApplySwitchSystemMatrixStamp(bit[i], sys, 0);
43  // Compute LU-factorization for system matrix
44  mLuFactorizations[bit][0]->analyzePattern(sys);
45  mLuFactorizations[bit][0]->factorize(sys);
46 }

59 template <>
60 void MnaSolverEigenSparse<Complex>::createEmptySystemMatrix() {
61  if (mSwitches.size() > SWITCH_NUM)
62  throw SystemError("Too many Switches.");
63 
64  if (mFrequencyParallel) {
65  for (UInt i = 0; i < std::pow(2,mSwitches.size()); ++i) {
66  for(Int freq = 0; freq < mSystem.mFrequencies.size(); ++freq) {
67  auto bit = std::bitset<SWITCH_NUM>(i);
68  mSwitchedMatrices[bit].push_back(SparseMatrix(2*(mNumMatrixNodeIndices), 2*(mNumMatrixNodeIndices)));
69  mLuFactorizations[bit].push_back(std::make_shared<LUFactorizedSparse>());
70  }
71  }
72  }
73  else {
74  for (std::size_t i = 0; i < (1ULL << mSwitches.size()); i++) {
75  auto bit = std::bitset<SWITCH_NUM>(i);
76  mSwitchedMatrices[bit].push_back(SparseMatrix(2*(mNumTotalMatrixNodeIndices), 2*(mNumTotalMatrixNodeIndices)));
77  mLuFactorizations[bit].push_back(std::make_shared<LUFactorizedSparse>());
78  }
79  mBaseSystemMatrix.resize(2 * (mNumTotalMatrixNodeIndices), 2 * (mNumTotalMatrixNodeIndices));
80  }
81 }

112  MnaSolver<VarType>::updateSwitchStatus();
113 
114  if (mSwitchedMatrices.size() > 0)

125 template <typename VarType>
126 void MnaSolverEigenSparse<VarType>::solveWithHarmonics(Real time, Int timeStepCount, Int freqIdx) {
127  mRightSideVectorHarm[freqIdx].setZero();
128 
129  // Sum of right side vectors (computed by the components' pre-step tasks)
130  for (auto stamp : mRightVectorStamps)
131  mRightSideVectorHarm[freqIdx] += stamp->col(freqIdx);
132 
133  mLeftSideVectorHarm[freqIdx] = mLuFactorizations[mCurrentSwitchStatus][freqIdx]->solve(mRightSideVectorHarm[freqIdx]);
134 }
