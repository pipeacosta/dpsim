33 template <typename VarType>
34 void MnaSolverEigenSparse<VarType>::switchedMatrixStamp(std::size_t index, std::vector<std::shared_ptr<CPS::MNAInterface>>& comp)
35 {
36  auto bit = std::bitset<SWITCH_NUM>(index);
37  auto& sys = mSwitchedMatrices[bit][0];
38  for (auto comp : comp) {
39  comp->mnaApplySystemMatrixStamp(sys);
40  }
41  for (UInt i = 0; i < mSwitches.size(); ++i)
42  mSwitches[i]->mnaApplySwitchSystemMatrixStamp(bit[i], sys, 0);
43 
44  // Compute LU-factorization for system matrix
45  mLuFactorizations[bit][0]->analyzePattern(sys);
46  mLuFactorizations[bit][0]->factorize(sys);
47 }

50 void MnaSolverEigenSparse<VarType>::stampVariableSystemMatrix() {
51 
52  mSLog->info("Number of variable Elements: {}"
53  "\nNumber of MNA components: {}",
54  mVariableComps.size(),
55  mMNAComponents.size());
56 
57  // Build base matrix with only static elements
58  mBaseSystemMatrix.setZero();
59  for (auto statElem : mMNAComponents)
60  statElem->mnaApplySystemMatrixStamp(mBaseSystemMatrix);
61  mSLog->info("Base matrix with only static elements: {}", Logger::matrixToString(mBaseSystemMatrix));
62  mSLog->flush();
63 
64  // Use matrix with only static elements as basis for variable system matrix
65  mVariableSystemMatrix = mBaseSystemMatrix;
66 
67  // Now stamp switches into matrix
68  mSLog->info("Stamping switches");
69  for (auto sw : mSwitches)
70  sw->mnaApplySystemMatrixStamp(mVariableSystemMatrix);
71 
72  // Now stamp initial state of variable elements into matrix
73  mSLog->info("Stamping variable elements");
74  for (auto varElem : mMNAIntfVariableComps)
75  varElem->mnaApplySystemMatrixStamp(mVariableSystemMatrix);
76 
77  mSLog->info("Initial system matrix with variable elements {}", Logger::matrixToString(mVariableSystemMatrix));
78  mSLog->flush();
79 
80  // Calculate factorization of current matrix
81  mLuFactorizationVariableSystemMatrix.analyzePattern(mVariableSystemMatrix);
82  mLuFactorizationVariableSystemMatrix.factorize(mVariableSystemMatrix);

109 template <typename VarType>
110 void MnaSolverEigenSparse<VarType>::recomputeSystemMatrix(Real time) {
111  // Start from base matrix
112  mVariableSystemMatrix = mBaseSystemMatrix;
113 
114  // Now stamp switches into matrix
115  for (auto sw : mSwitches)
116  sw->mnaApplySystemMatrixStamp(mVariableSystemMatrix);
117 
118  // Now stamp variable elements into matrix
119  for (auto comp : mMNAIntfVariableComps)
120  comp->mnaApplySystemMatrixStamp(mVariableSystemMatrix);
121 
122  // Refactorization of matrix assuming that structure remained
123  // constant by omitting analyzePattern
124  mLuFactorizationVariableSystemMatrix.factorize(mVariableSystemMatrix);
125  ++mNumRecomputations;
126 }

128 template <>
129 void MnaSolverEigenSparse<Real>::createEmptySystemMatrix() {
130  if (mSwitches.size() > SWITCH_NUM)
131  throw SystemError("Too many Switches.");
132 
133  if (mSystemMatrixRecomputation) {
134  mBaseSystemMatrix = SparseMatrix(mNumMatrixNodeIndices, mNumMatrixNodeIndices);
135  mVariableSystemMatrix = SparseMatrix(mNumMatrixNodeIndices, mNumMatrixNodeIndices);
136  } else {
137  for (std::size_t i = 0; i < (1ULL << mSwitches.size()); i++){
138  auto bit = std::bitset<SWITCH_NUM>(i);
139  mSwitchedMatrices[bit].push_back(SparseMatrix(mNumMatrixNodeIndices, mNumMatrixNodeIndices));
140  mLuFactorizations[bit].push_back(std::make_shared<LUFactorizedSparse>());
141  }
142  }
143 }

145 template <>
146 void MnaSolverEigenSparse<Complex>::createEmptySystemMatrix() {
147  if (mSwitches.size() > SWITCH_NUM)
148  throw SystemError("Too many Switches.");
149 
150  if (mFrequencyParallel) {
151  for (UInt i = 0; i < std::pow(2,mSwitches.size()); ++i) {
152  for(Int freq = 0; freq < mSystem.mFrequencies.size(); ++freq) {
153  auto bit = std::bitset<SWITCH_NUM>(i);
154  mSwitchedMatrices[bit].push_back(SparseMatrix(2*(mNumMatrixNodeIndices), 2*(mNumMatrixNodeIndices)));
155  mLuFactorizations[bit].push_back(std::make_shared<LUFactorizedSparse>());
156  }
157  }
158  } else if (mSystemMatrixRecomputation) {
159  mBaseSystemMatrix = SparseMatrix(2*(mNumMatrixNodeIndices), 2*(mNumMatrixNodeIndices));
160  mVariableSystemMatrix = SparseMatrix(2*(mNumMatrixNodeIndices), 2*(mNumMatrixNodeIndices));
161  } else {
162  for (std::size_t i = 0; i < (1ULL << mSwitches.size()); i++) {
163  auto bit = std::bitset<SWITCH_NUM>(i);
164  mSwitchedMatrices[bit].push_back(SparseMatrix(2*(mNumTotalMatrixNodeIndices), 2*(mNumTotalMatrixNodeIndices)));
165  mLuFactorizations[bit].push_back(std::make_shared<LUFactorizedSparse>());
166  }
167  }

220  mRightSideVectorHarm[freqIdx].setZero();
