71  template <typename VarType>
72  static std::shared_ptr<MnaSolver<VarType>> factory(String name,
73  CPS::Domain domain = CPS::Domain::DP,
74  CPS::Logger::Level logLevel = CPS::Logger::Level::info,
75  MnaSolverImpl implementation = mSupportedSolverImpls().back(),
76  String pluginName = "plugin.so")
77  {
78  //To avoid regression we use EigenDense in case of undefined implementation
79  if (implementation == MnaSolverImpl::Undef) {
80  implementation = MnaSolverImpl::EigenDense;
81  }
82  CPS::Logger::Log log = CPS::Logger::get("MnaSolverFactory", CPS::Logger::Level::info, CPS::Logger::Level::info);
83 
84  switch(implementation) {
85  case MnaSolverImpl::EigenDense:
86  log->info("creating EigenDense solver implementation");
87  return std::make_shared<MnaSolverEigenDense<VarType>>(name, domain, logLevel);
88 #ifdef WITH_SPARSE
89  case MnaSolverImpl::EigenSparse:
90  log->info("creating EigenSparse solver implementation");
91  return std::make_shared<MnaSolverEigenSparse<VarType>>(name, domain, logLevel);
92 #endif
93 #ifdef WITH_CUDA
94  case MnaSolverImpl::CUDADense:
95  log->info("creating CUDADense solver implementation");
96  return std::make_shared<MnaSolverGpuDense<VarType>>(name, domain, logLevel);
97 #ifdef WITH_SPARSE
98  case MnaSolverImpl::CUDASparse:
99  log->info("creating CUDASparse solver implementation");
100  return std::make_shared<MnaSolverGpuSparse<VarType>>(name, domain, logLevel);
101 #ifdef WITH_MAGMA
102  case MnaSolverImpl::CUDAMagma:
103  log->info("creating CUDAMagma solver implementation");
104  return std::make_shared<MnaSolverGpuMagma<VarType>>(name, domain, logLevel);
105 #endif
106 #endif
107 #endif
108 #ifdef WITH_MNASOLVERPLUGIN
109  case MnaSolverImpl::Plugin:
110  log->info("creating Plugin solver implementation");
111  return std::make_shared<MnaSolverPlugin<VarType>>(pluginName, name, domain, logLevel);
112 #endif
113  default:
114  throw CPS::SystemError("unsupported MNA implementation.");
115 
116  }
117  }
