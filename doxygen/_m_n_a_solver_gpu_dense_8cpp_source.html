88 
89  //Workspace
90  int workSpaceSize = 0;
91  cusolverStatus_t status = CUSOLVER_STATUS_SUCCESS;
92  if((status =

100  ) != CUSOLVER_STATUS_SUCCESS)
101  std::cerr << "cusolverDnDgetrf_bufferSize() failed (calculating required space for LU-factorization)" << std::endl;
102  CUDA_ERROR_HANDLER(cudaMalloc((void**)&mDeviceCopy.workSpace, workSpaceSize))

163 template <typename VarType>
164 void MnaSolverGpuDense<VarType>::solve(Real time, Int timeStepCount) {
165  // Reset source vector
166  this->mRightSideVector.setZero();
167 
168  // Add together the right side vector (computed by the components'
169  // pre-step tasks)
170  for (const auto &stamp : this->mRightVectorStamps)
171  this->mRightSideVector += *stamp;
172 
173  if (!this->mIsInInitialization)
174  this->updateSwitchStatus();
175 
176  //Copy right vector to device
177  CUDA_ERROR_HANDLER(cudaMemcpy(mDeviceCopy.vector, &this->mRightSideVector(0), mDeviceCopy.size * sizeof(Real), cudaMemcpyHostToDevice))
178 
179  // Solve
180  if (this->mSwitchedMatrices.size() > 0) {
181  cusolverStatus_t status = cusolverDnDgetrs(
182  mCusolverHandle,
183  CUBLAS_OP_N,
184  mDeviceCopy.size,
185  1, /* nrhs */
186  mDeviceCopy.matrix,
187  mDeviceCopy.size,
188  mDeviceCopy.pivSeq,
189  mDeviceCopy.vector,
190  mDeviceCopy.size,
191  mDeviceCopy.errInfo);
192 
193  CUDA_ERROR_HANDLER(cudaDeviceSynchronize())
194 
195  if(status != CUSOLVER_STATUS_SUCCESS)
196  std::cerr << "cusolverDnDgetrs() failed (Solving A*x = b)" << std::endl;
197  int info;
198  CUDA_ERROR_HANDLER(cudaMemcpy(&info, mDeviceCopy.errInfo, sizeof(int), cudaMemcpyDeviceToHost))
199  if(0 > info) {
200  std::cerr << -info << "-th parameter is wrong" << std::endl;
201  }
202  }
203 
204  //Copy Solution back
205  CUDA_ERROR_HANDLER(cudaMemcpy(&this->mLeftSideVector(0), mDeviceCopy.vector, mDeviceCopy.size * sizeof(Real), cudaMemcpyDeviceToHost))
206 
207  // TODO split into separate task? (dependent on x, updating all v attributes)
208  for (UInt nodeIdx = 0; nodeIdx < this->mNumNetNodes; ++nodeIdx)
209  this->mNodes[nodeIdx]->mnaUpdateVoltage(this->mLeftSideVector);
210 
211  // Components' states will be updated by the post-step tasks
212 }
