46 template <typename VarType>
47 void MnaSolverGpuMagma<VarType>::initialize() {
48  MnaSolver<VarType>::initialize();
49 
50  int size = this->mRightSideVector.rows();
51  int p_nnz = 0;
52  int p[size];
53  auto hMat = this->mSwitchedMatrices[std::bitset<SWITCH_NUM>(0)];
54  size_t nnz = hMat[0].nonZeros();
55  cusparseMatDescr_t descr_M = 0;
56 
57  cusolverSpHandle_t mCusolverhandle;
58  if (cusolverSpCreate(&mCusolverhandle) != CUSOLVER_STATUS_SUCCESS) {
59  mSLog->error("cusolverSpCreate returend an error");
60  return;
61  }
62  if(cusparseCreateMatDescr(&descr_M) != CUSPARSE_STATUS_SUCCESS) {
63  mSLog->error("cusolver returend an error");
64  return;
65  }
66 
67  if(cusparseSetMatIndexBase(descr_M, CUSPARSE_INDEX_BASE_ZERO) != CUSPARSE_STATUS_SUCCESS) {
68  mSLog->error("cusolver returend an error");
69  return;
70  }
71  if(cusparseSetMatType(descr_M, CUSPARSE_MATRIX_TYPE_GENERAL) != CUSPARSE_STATUS_SUCCESS) {
72  mSLog->error("cusolver returend an error");
73  return;
74  }
75 
76  if (cusolverSpDcsrzfdHost(
77  mCusolverhandle, size, nnz, descr_M,
78  hMat[0].valuePtr(),
79  hMat[0].outerIndexPtr(),
80  hMat[0].innerIndexPtr(),
81  p, &p_nnz) != CUSOLVER_STATUS_SUCCESS) {
82  mSLog->error("cusolverSpDcsrzfdHost returend an error");
83  return;
84  }
85 
86  // create Eigen::PermutationMatrix from the p
87  mTransp = std::unique_ptr<Eigen::PermutationMatrix<Eigen::Dynamic> >(
88  new Eigen::PermutationMatrix<Eigen::Dynamic>(
89  Eigen::Map< Eigen::Matrix<int, Eigen::Dynamic, 1> >(p, size, 1)));
90 
91  // apply permutation
92  //std::cout << "Before System Matrix:" << std::endl << hMat[0] << std::endl;
93  hMat[0] = *mTransp * hMat[0];
94  //std::cout << "permutation:" << std::endl << mTransp->toDenseMatrix() << std::endl;
95  //std::cout << "inverse permutation:" << std::endl << mTransp->inverse().toDenseMatrix() << std::endl;
96  //std::cout << "System Matrix:" << std::endl << hMat[0] << std::endl;
97  magma_dcsrset(size, size,
98  hMat[0].outerIndexPtr(),
99  hMat[0].innerIndexPtr(),
100  hMat[0].valuePtr(),
101  &mHostSysMat,
102  mMagmaQueue);
103 
104  mMagmaOpts.solver_par.solver = Magma_PIDRMERGE;
105  mMagmaOpts.solver_par.restart = 8;
106  mMagmaOpts.solver_par.maxiter = 1000;
107  mMagmaOpts.solver_par.rtol = 1e-10;
108  mMagmaOpts.solver_par.maxiter = 1000;
109  mMagmaOpts.precond_par.solver = Magma_ILU;
110  mMagmaOpts.precond_par.levels = 0;
111  mMagmaOpts.precond_par.trisolver = Magma_CUSOLVE;
112 
113  magma_dsolverinfo_init(&mMagmaOpts.solver_par,
114  &mMagmaOpts.precond_par,
115  mMagmaQueue);
116 
117  magma_dvinit(&mDevRhsVec, Magma_DEV, size, 1, 0.0, mMagmaQueue);
118  magma_dmtransfer(mHostSysMat, &mDevSysMat, Magma_CPU, Magma_DEV, mMagmaQueue);
119  magma_d_precondsetup(mDevSysMat, mDevRhsVec, &mMagmaOpts.solver_par, &mMagmaOpts.precond_par, mMagmaQueue);
120 
121  cusolverSpDestroy(mCusolverhandle);
122  cusparseDestroyMatDescr(descr_M);
123 
124 }

126 template <typename VarType>
127 Task::List MnaSolverGpuMagma<VarType>::getTasks() {
128  Task::List l;
129 
130  for (auto comp : this->mMNAComponents) {
131  for (auto task : comp->mnaTasks()) {
132  l.push_back(task);
133  }
134  }
135  for (auto node : this->mNodes) {
136  for (auto task : node->mnaTasks())
137  l.push_back(task);
138  }
139  // TODO signal components should be moved out of MNA solver
140  for (auto comp : this->mSimSignalComps) {
141  for (auto task : comp->getTasks()) {
142  l.push_back(task);
143  }
144  }
145  l.push_back(std::make_shared<MnaSolverGpuMagma<VarType>::SolveTask>(*this));
146  l.push_back(std::make_shared<MnaSolverGpuMagma<VarType>::LogTask>(*this));
147  return l;

150 template <typename VarType>
151 void MnaSolverGpuMagma<VarType>::solve(Real time, Int timeStepCount) {
152  int size = this->mRightSideVector.rows();
153  int one = 0;
154  // Reset source vector
155  this->mRightSideVector.setZero();
156 
157  // Add together the right side vector (computed by the components'
158  // pre-step tasks)
159  for (const auto &stamp : this->mRightVectorStamps)
160  this->mRightSideVector += *stamp;
161 
162  if (!this->mIsInInitialization)
163  this->updateSwitchStatus();
164 
165  this->mRightSideVector = *mTransp * this->mRightSideVector;
166  //Copy right vector to device
167  magma_dvset(size, 1, this->mRightSideVector.data(), &mHostRhsVec, mMagmaQueue);
168  magma_dmtransfer(mHostRhsVec, &mDevRhsVec, Magma_CPU, Magma_DEV, mMagmaQueue);
169  magma_dvinit(&mDevLhsVec, Magma_DEV, mHostRhsVec.num_rows, mHostRhsVec.num_cols, 0.0, mMagmaQueue);
170 
171  // Solve
172  magma_d_solver(mDevSysMat, mDevRhsVec, &mDevLhsVec, &mMagmaOpts, mMagmaQueue);
173 
174  //Copy Solution back
175  magma_dmtransfer(mDevLhsVec, &mHostLhsVec, Magma_DEV, Magma_CPU, mMagmaQueue);
176  magma_dvcopy(mDevLhsVec, &size, &one, this->leftSideVector().data(), mMagmaQueue);
177 
178  mSLog->debug("result has size ({},{})", size, one);
179 
180  //Apply inverse Permutation: TODO: This seems to be wrong, but why?
181  //this->mLeftSideVector = mTransp->inverse() * this->mLeftSideVector;
182 
183  // TODO split into separate task? (dependent on x, updating all v attributes)
184  for (UInt nodeIdx = 0; nodeIdx < this->mNumNetNodes; ++nodeIdx)
185  this->mNodes[nodeIdx]->mnaUpdateVoltage(this->mLeftSideVector);
186 
187 
188  // Components' states will be updated by the post-step tasks
189 }
