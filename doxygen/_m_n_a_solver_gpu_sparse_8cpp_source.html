54 template <typename VarType>
55 void MnaSolverGpuSparse<VarType>::initialize() {
56  MnaSolver<VarType>::initialize();
57 
58  cusparseStatus_t csp_status;
59  cusolverStatus_t cso_status;
60  csp_status = cusparseCreate(&mCusparsehandle);
61  checkCusparseStatus(csp_status, "cuSparse initialization failed:");
62 
63  if ((cso_status = cusolverSpCreate(&mCusolverhandle)) != CUSOLVER_STATUS_SUCCESS) {
64  mSLog->error("cuSolver initialization failed: Error code {}", cso_status);
65  throw SolverException();
66  }
67 
68  size_t N = this->mRightSideVector.rows();
69  auto hMat = this->mSwitchedMatrices[std::bitset<SWITCH_NUM>(0)];
70 
71  mGpuRhsVec = cuda::Vector<double>(N);
72  mGpuLhsVec = cuda::Vector<double>(N);
73  mGpuIntermediateVec = cuda::Vector<double>(N);
74 
75  cusparseMatDescr_t descr_M = 0;
76  csrilu02Info_t info_M = 0;
77  int structural_zero;
78  int numerical_zero;
79 
80  size_t nnz = hMat[0].nonZeros();
81 
82  // step 1: create a descriptor which contains
83  // - matrix M is base-1
84  // - matrix L is base-1
85  // - matrix L is lower triangular
86  // - matrix L has unit diagonal
87  // - matrix U is base-1
88  // - matrix U is upper triangular
89  // - matrix U has non-unit diagonal
90  checkCusparseStatus(cusparseCreateMatDescr(&descr_M));
91  checkCusparseStatus(cusparseSetMatIndexBase(descr_M, CUSPARSE_INDEX_BASE_ZERO));
92  checkCusparseStatus(cusparseSetMatType(descr_M, CUSPARSE_MATRIX_TYPE_GENERAL));
93 
94  checkCusparseStatus(cusparseCreateMatDescr(&descr_L));
95  checkCusparseStatus(cusparseSetMatIndexBase(descr_L, CUSPARSE_INDEX_BASE_ZERO));
96  checkCusparseStatus(cusparseSetMatType(descr_L, CUSPARSE_MATRIX_TYPE_GENERAL));
97  checkCusparseStatus(cusparseSetMatFillMode(descr_L, CUSPARSE_FILL_MODE_LOWER));
98  checkCusparseStatus(cusparseSetMatDiagType(descr_L, CUSPARSE_DIAG_TYPE_UNIT));
99 
100  checkCusparseStatus(cusparseCreateMatDescr(&descr_U));
101  checkCusparseStatus(cusparseSetMatIndexBase(descr_U, CUSPARSE_INDEX_BASE_ZERO));
102  checkCusparseStatus(cusparseSetMatType(descr_U, CUSPARSE_MATRIX_TYPE_GENERAL));
103  checkCusparseStatus(cusparseSetMatFillMode(descr_U, CUSPARSE_FILL_MODE_UPPER));
104  checkCusparseStatus(cusparseSetMatDiagType(descr_U, CUSPARSE_DIAG_TYPE_NON_UNIT));
105 
106  // step 2: create a empty info structure
107  // we need one info for csrilu02 and two info's for csrsv2
108  checkCusparseStatus(cusparseCreateCsrilu02Info(&info_M));
109  checkCusparseStatus(cusparseCreateCsrsv2Info(&info_L));
110  checkCusparseStatus(cusparseCreateCsrsv2Info(&info_U));
111 
112  // step 2a: permutate Matrix M' = P*M
113  int p_nnz = 0;
114  int p[N];
115 
116  cso_status = cusolverSpDcsrzfdHost(
117  mCusolverhandle, N,nnz, descr_M,
118  hMat[0].valuePtr(),
119  hMat[0].outerIndexPtr(),
120  hMat[0].innerIndexPtr(),
121  p, &p_nnz);
122 
123  if (cso_status != CUSOLVER_STATUS_SUCCESS) {
124  mSLog->error("cusolverSpDcsrzfdHost returend an error");
125  }
126  // create Eigen::PermutationMatrix from the p
127  mTransp = std::unique_ptr<Eigen::PermutationMatrix<Eigen::Dynamic> >(
128  new Eigen::PermutationMatrix<Eigen::Dynamic>(
129  Eigen::Map< Eigen::Matrix<int, Eigen::Dynamic, 1> >(p, N, 1)));
130 
131  // apply permutation
132  hMat[0] = *mTransp * hMat[0];
133 
134  // copy P' to GPU
135  mSysMat = std::unique_ptr<cuda::CudaMatrix<double, int>>(new cuda::CudaMatrix<double, int>(hMat[0], N));
136 
137  double *d_csrVal = mSysMat->val.data();
138  int *d_csrRowPtr = mSysMat->row.data();
139  int *d_csrColInd = mSysMat->col.data();
140 
141 
142  // step 3: query how much memory used in csrilu02 and csrsv2, and allocate the buffer
143  int pBufferSize_M, pBufferSize_L, pBufferSize_U;
144  int pBufferSize;
145  csp_status = cusparseDcsrilu02_bufferSize(mCusparsehandle, N, nnz, descr_M,
146  d_csrVal, d_csrRowPtr, d_csrColInd,
147  info_M, &pBufferSize_M);
148  checkCusparseStatus(csp_status, "failed to get cusparse bufferSize:");
149 
150  csp_status = cusparseDcsrsv2_bufferSize(mCusparsehandle, CUSPARSE_OPERATION_NON_TRANSPOSE,
151  N, nnz, descr_L, d_csrVal, d_csrRowPtr,
152  d_csrColInd, info_L, &pBufferSize_L);
153  checkCusparseStatus(csp_status, "failed to get cusparse bufferSize:");
154 
155  csp_status = cusparseDcsrsv2_bufferSize(mCusparsehandle, CUSPARSE_OPERATION_NON_TRANSPOSE,
156  N, nnz, descr_U, d_csrVal, d_csrRowPtr,
157  d_csrColInd, info_U, &pBufferSize_U);
158  checkCusparseStatus(csp_status, "failed to get cusparse bufferSize:");
159 
160  // Buffer
161  pBufferSize = std::max({pBufferSize_M, pBufferSize_L, pBufferSize_U});
162  pBuffer = cuda::Vector<char>(pBufferSize);
163 
164  // step 4: perform analysis of incomplete Cholesky on M
165  // perform analysis of triangular solve on L
166  // perform analysis of triangular solve on U
167  // The lower(upper) triangular part of M has the same sparsity pattern as L(U),
168  // we can do analysis of csrilu0 and csrsv2 simultaneously.
169  csp_status = cusparseDcsrilu02_analysis(mCusparsehandle, N, nnz, descr_M,
170  d_csrVal, d_csrRowPtr, d_csrColInd,
171  info_M, CUSPARSE_SOLVE_POLICY_NO_LEVEL,
172  pBuffer.data());
173  checkCusparseStatus(csp_status, "failed to analyse cusparse problem:");
174 
175  csp_status = cusparseXcsrilu02_zeroPivot(mCusparsehandle, info_M, &structural_zero);
176  if (csp_status == CUSPARSE_STATUS_ZERO_PIVOT){
177  mSLog->error("A({},{}) is missing", structural_zero, structural_zero);
178  checkCusparseStatus(csp_status);
179  throw SolverException();
180  }
181 
182  csp_status = cusparseDcsrsv2_analysis(mCusparsehandle, CUSPARSE_OPERATION_NON_TRANSPOSE,
183  N, nnz, descr_L, d_csrVal, d_csrRowPtr,
184  d_csrColInd, info_L, CUSPARSE_SOLVE_POLICY_NO_LEVEL,
185  pBuffer.data());
186  checkCusparseStatus(csp_status, "failed to analyse cusparse problem:");
187 
188  csp_status = cusparseDcsrsv2_analysis(mCusparsehandle, CUSPARSE_OPERATION_NON_TRANSPOSE,
189  N, nnz, descr_U, d_csrVal, d_csrRowPtr,
190  d_csrColInd, info_U, CUSPARSE_SOLVE_POLICY_USE_LEVEL,
191  pBuffer.data());
192  checkCusparseStatus(csp_status, "failed to analyse cusparse problem:");
193 
194  // step 5: M = L * U
195 
196  csp_status = cusparseDcsrilu02(mCusparsehandle, N, nnz, descr_M, d_csrVal,
197  d_csrRowPtr, d_csrColInd, info_M, CUSPARSE_SOLVE_POLICY_NO_LEVEL,
198  pBuffer.data());
199  checkCusparseStatus(csp_status, "failed to perform cusparse ILU:");
200 
201  csp_status = cusparseXcsrilu02_zeroPivot(mCusparsehandle, info_M, &numerical_zero);
202  if (csp_status == CUSPARSE_STATUS_ZERO_PIVOT){
203  mSLog->error("U({},{}) is zero\n", numerical_zero, numerical_zero);
204  checkCusparseStatus(csp_status);
205  throw SolverException();
206  }
207 
208  //std::cout << *mSysMat.get() << std::endl;
209 
210  csp_status = cusparseDestroyCsrilu02Info(info_M);
211  checkCusparseStatus(csp_status, "failed to destroy info:");
212  csp_status = cusparseDestroyMatDescr(descr_M);
213  checkCusparseStatus(csp_status, "failed to destroy MatDescr:");
214 }

216 template <typename VarType>
217 Task::List MnaSolverGpuSparse<VarType>::getTasks() {
218  Task::List l;
219 
220  for (auto comp : this->mMNAComponents) {
221  for (auto task : comp->mnaTasks()) {
222  l.push_back(task);
223  }
224  }
225  for (auto node : this->mNodes) {
226  for (auto task : node->mnaTasks())
227  l.push_back(task);
228  }
229  // TODO signal components should be moved out of MNA solver
230  for (auto comp : this->mSimSignalComps) {
231  for (auto task : comp->getTasks()) {
232  l.push_back(task);
233  }
234  }
235  l.push_back(std::make_shared<MnaSolverGpuSparse<VarType>::SolveTask>(*this));
236  l.push_back(std::make_shared<MnaSolverGpuSparse<VarType>::LogTask>(*this));
237  return l;
238 }

240 template <typename VarType>
241 void MnaSolverGpuSparse<VarType>::solve(Real time, Int timeStepCount) {
242  cudaError_t status;
243  cusparseStatus_t csp_status;
244  int size = this->mRightSideVector.rows();
245  // Reset source vector
246  this->mRightSideVector.setZero();
247 
248  // Add together the right side vector (computed by the components'
249  // pre-step tasks)
250  for (const auto &stamp : this->mRightVectorStamps)
251  this->mRightSideVector += *stamp;
252 
253  if (!this->mIsInInitialization)
254  this->updateSwitchStatus();
255 
256  //Copy right vector to device
257  //Permutate right side: R' = P * R
258  this->mRightSideVector = *mTransp * this->mRightSideVector;
259  status = cudaMemcpy(mGpuRhsVec.data(), &this->mRightSideVector(0), size * sizeof(Real), cudaMemcpyHostToDevice);
260  if (status != cudaSuccess) {
261  mSLog->error("Cuda Error: {}", cudaGetErrorString(status));
262  throw SolverException();
263  }
264 
265  const double alpha = 1.;
266  // Solve
267  // step 6: solve L*z = x
268  csp_status = cusparseDcsrsv2_solve(mCusparsehandle, CUSPARSE_OPERATION_NON_TRANSPOSE,
269  size, mSysMat->non_zero, &alpha, descr_L,
270  mSysMat->val.data(), mSysMat->row.data(),
271  mSysMat->col.data(), info_L, mGpuRhsVec.data(),
272  mGpuIntermediateVec.data(), CUSPARSE_SOLVE_POLICY_NO_LEVEL,
273  pBuffer.data());
274  checkCusparseStatus(csp_status, "failed to solve L*z=x:");
275 
276  // step 7: solve U*y = z
277  csp_status = cusparseDcsrsv2_solve(mCusparsehandle, CUSPARSE_OPERATION_NON_TRANSPOSE,
278  size, mSysMat->non_zero, &alpha, descr_U,
279  mSysMat->val.data(), mSysMat->row.data(),
280  mSysMat->col.data(), info_U, mGpuIntermediateVec.data(),
281  mGpuLhsVec.data(), CUSPARSE_SOLVE_POLICY_USE_LEVEL,
282  pBuffer.data());
283  checkCusparseStatus(csp_status, "failed to solve U*y=z:");
284 
285  //Copy Solution back
286  status = cudaMemcpy(&(**this->mLeftSideVector)(0), mGpuLhsVec.data(), size * sizeof(Real), cudaMemcpyDeviceToHost);
287  if (status != cudaSuccess) {
288  mSLog->error("Cuda Error: {}", cudaGetErrorString(status));
289  throw SolverException();
290  }
291

293  //**this->mLeftSideVector = mTransp->inverse() * **this->mLeftSideVector;
294 
295  // TODO split into separate task? (dependent on x, updating all v attributes)
296  for (UInt nodeIdx = 0; nodeIdx < this->mNumNetNodes; ++nodeIdx)
297  this->mNodes[nodeIdx]->mnaUpdateVoltage(**this->mLeftSideVector);
298 
299 
300  // Components' states will be updated by the post-step tasks
301 }
