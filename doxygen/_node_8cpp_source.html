13 CPS::TopologicalNode::List Python::nodesFromPython(PyObject* list) {
14  CPS::TopologicalNode::List nodes;
15 
16  if (!PyList_Check(list))
17  throw std::invalid_argument( "argument is not a list" );
18 
19  for (int i = 0; i < PyList_Size(list); i++) {
20  PyObject* obj = PyList_GetItem(list, i);
21  if (PyObject_TypeCheck(obj, &Python::Node<CPS::Real>::type)) {
22  Node<CPS::Real>* pyNode = (Node<CPS::Real>*) obj;
23  nodes.push_back(std::dynamic_pointer_cast<CPS::TopologicalNode>(pyNode->node));
24  }
25  else if (PyObject_TypeCheck(obj, &Python::Node<CPS::Complex>::type)) {
26  Node<CPS::Complex>* pyNode = (Node<CPS::Complex>*) obj;
27  nodes.push_back(std::dynamic_pointer_cast<CPS::TopologicalNode>(pyNode->node));
28  }
29  else {
30  throw std::invalid_argument( "list element is not a dpsim.Node" );
31  }
32  }
33 
34  return nodes;
35  }

40  typename CPS::SimNode<VarType>::List nodes;

42  if (!PyList_Check(list))

46  PyObject* obj = PyList_GetItem(list, i);

72 template<typename VarType>
73 int Python::Node<VarType>::init(Python::Node<VarType> *self, PyObject *args, PyObject *kwds)
74 {
75  static const char *kwlist[] = {"uid", "initial_voltage", nullptr};
76 
77  const char *uid;
78  CPS::Complex initialVoltage;
79 
80  if (!PyArg_ParseTupleAndKeywords(args, kwds, "s|D", (char **) kwlist, &uid, &initialVoltage)) {
81  return -1;
82  }
83 
84  self->node = std::make_shared<CPS::SimNode<VarType>>(uid, CPS::PhaseType::Single, std::vector<CPS::Complex>({ initialVoltage }));
85 
86  return 0;
87 };

89 template<typename VarType>
90 void Python::Node<VarType>::dealloc(Python::Node<VarType> *self)
91 {
92  // Since this is not a C++ destructor which would automatically call the
93  // destructor of its members, we have to manually call the destructor of
94  // the vectors here to free the associated memory.
95 
96  // This is a workaround for a compiler bug: https://stackoverflow.com/a/42647153/8178705
97  using SharedNodePtr = std::shared_ptr<CPS::SimNode<VarType>>;
98 
99  if (self->node)
100  self->node.~SharedNodePtr();
101 
102  Py_TYPE(self)->tp_free((PyObject*) self);
103 }
