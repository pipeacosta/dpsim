89 void PLL::signalStep(Real time, Int timeStepCount) {
90  mInputCurr(1,0) = attribute<Real>("input_ref")->get();
91 
92  mSLog->info("Time {}:", time);
93  mSLog->info("Input values: inputCurr = ({}, {}), inputPrev = ({}, {}), stateCurr = ({}, {}), statePrev = ({}, {})", mInputCurr(0,0), mInputCurr(1,0), mInputPrev(0,0), mInputPrev(1,0), mStateCurr(0,0), mStateCurr(1,0), mStatePrev(0,0), mStatePrev(1,0));
94 
95  mStateCurr = Math::StateSpaceTrapezoidal(mStatePrev, mA, mB, mTimeStep, mInputCurr, mInputPrev);
96  mOutputCurr = mC * mStateCurr + mD * mInputCurr;
97 
98  mSLog->info("State values: stateCurr = ({}, {})", mStateCurr(0,0), mStateCurr(1,0));
99  mSLog->info("Output values: outputCurr = ({}, {}):", mOutputCurr(0,0), mOutputCurr(1,0));
100 }
