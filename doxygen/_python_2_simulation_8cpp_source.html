38 void Python::Simulation::threadFunction(Python::Simulation *self)
39 {
40  Real time, finalTime;
41 
42  self->sim->initialize();
43 
44  Timer timer(Timer::Flags::fail_on_overrun);
45 
46  if (self->realTime) {
47  timer.setStartTime(self->startTime);
48  timer.setInterval(self->realTimeStep);
49  timer.start();
50 
51  std::cout << "Starting simulation at " << self->startTime << " (delta_T = " << self->startTime - Timer::StartClock::now() << " seconds)" << std::endl;
52  }
53 
54  finalTime = self->sim->finalTime();
55 
56  time = 0;
57  while (time < finalTime) {
58  time = self->sim->step();
59 
60  if (self->realTime) {
61  try {
62  timer.sleep();
63  }
64  catch (Timer::OverrunException) {
65  std::unique_lock<std::mutex> lk(*self->mut);
66 
67  if (self->failOnOverrun) {
68  newState(self, Simulation::State::overrun);
69  self->cond->notify_one();
70  }
71  }
72  }
73 
74  if (self->sim->timeStepCount() == 1) {
75  std::unique_lock<std::mutex> lk(*self->mut);
76  newState(self, Simulation::State::running);
77  self->cond->notify_one();
78  }
79 
80  if (self->state == State::pausing || self->singleStepping) {
81  std::unique_lock<std::mutex> lk(*self->mut);
82 
83  newState(self, Simulation::State::paused);
84  self->cond->notify_one();
85 
86  while (self->state == State::paused)
87  self->cond->wait(lk);
88 
89  if (self->state == State::resuming) {
90  newState(self, Simulation::State::running);
91  self->cond->notify_one();
92  }
93  }
94 
95  if (self->state == State::stopping) {
96  std::unique_lock<std::mutex> lk(*self->mut);
97  newState(self, State::stopped);
98  self->cond->notify_one();
99  return;
100  }
101  }
102 
103  self->sim->scheduler()->stop();
104 
105  for (auto lg : self->sim->loggers())
106  lg->close();
107 
108  {
109  std::unique_lock<std::mutex> lk(*self->mut);
110  newState(self, State::done);
111  self->cond->notify_one();
112  }
113 }

115 PyObject* Python::Simulation::newfunc(PyTypeObject* subtype, PyObject *args, PyObject *kwds)
116 {
117  Python::Simulation *self;
118 
119  self = (Python::Simulation*) subtype->tp_alloc(subtype, 0);
120  if (self) {
121  // since mutex, thread etc. have no copy-constructor, but we can't use
122  // our own C++ constructor that could be called from python, we need to
123  // implement them as pointers
124  self->cond = new std::condition_variable();
125  self->mut = new std::mutex();
126 
127  using SharedSimPtr = std::shared_ptr<DPsim::Simulation>;
128  using PyObjectsList = std::vector<PyObject *>;
129 
130  new (&self->sim) SharedSimPtr();
131  new (&self->refs) PyObjectsList();
132  }
133 
134  return (PyObject*) self;
135 }

137 int Python::Simulation::init(Simulation* self, PyObject *args, PyObject *kwds)
138 {
139  static const char *kwlist[] = {"name", "system", "timestep", "duration",
140  "start_time", "start_time_us", "sim_type", "solver_type", "single_stepping",
141  "rt", "rt_factor", "start_sync", "init_steady_state", "log_level",
142  "fail_on_overrun", "split_subnets", "tear_components", nullptr};
143  double timestep = 1e-3, duration = DBL_MAX, rtFactor = 1;
144  const char *name = nullptr;
145  int t = 0, s = 0, rt = 0, ss = 0, st = 0, log_level = 2, initSteadyState = 0, splitSubnets = 1;
146  int failOnOverrun = 0;
147 
148  unsigned long startTime = -1;
149  unsigned long startTimeUs = 0;
150 
151  enum Solver::Type solverType;
152  enum Domain domain;
153  CPS::Logger::Level logLevel = CPS::Logger::Level::info;
154 
155  CPS::IdentifiedObject::List tearComponents;
156  PyObject* pyTearComponents = nullptr;
157 
158  if (!PyArg_ParseTupleAndKeywords(args, kwds, "sO|ddkkiippdppippO", (char **) kwlist,
159  &name, &self->pySys, &timestep, &duration,
160  &startTime, &startTimeUs, &s, &t, &ss,
161  &rt, &rtFactor, &st, &initSteadyState, &log_level,
162  &failOnOverrun, &splitSubnets, &pyTearComponents)) {
163  return -1;
164  }
165 
166  self->state = State::stopped;
167  self->realTime = rt;
168  self->startSync = ss;
169  self->singleStepping = st;
170  self->failOnOverrun = failOnOverrun;
171  self->realTimeStep = timestep / rtFactor;
172 
173  if (startTime > 0) {
174  self->startTime = Timer::StartClock::from_time_t(startTime) + std::chrono::microseconds(startTimeUs);
175  }
176  else
177  self->startTime = Timer::StartClock::now();
178 
179  switch (s) {
180  case 0: domain = Domain::DP; break;
181  case 1: domain = Domain::EMT; break;
182  case 2: domain = Domain::SP; break;
183  default:
184  PyErr_SetString(PyExc_TypeError, "Invalid sim_type argument (must be one of 0, 1, 2)");
185  return -1;
186  }
187 
188  switch (t) {
189  case 0: solverType = DPsim::Solver::Type::MNA; break;
190  case 1: solverType = DPsim::Solver::Type::DAE; break;
191  case 2: solverType = DPsim::Solver::Type::NRP; break;
192  default:
193  PyErr_SetString(PyExc_TypeError, "Invalid solver_type argument (must be one of 0, 1, 2)");
194  return -1;
195  }
196 
197  switch (log_level) {
198  case 0: logLevel = CPS::Logger::Level::trace; break;
199  case 1: logLevel = CPS::Logger::Level::debug; break;
200  case 2: logLevel = CPS::Logger::Level::info; break;
201  case 3: logLevel = CPS::Logger::Level::warn; break;
202  case 4: logLevel = CPS::Logger::Level::err; break;
203  default:
204  PyErr_SetString(PyExc_TypeError, "Invalid log_level argument (must be one of 0, 1, 2, 3, 4)");
205  return -1;
206  }
207 
208  if (!PyObject_TypeCheck(self->pySys, &Python::SystemTopology::type)) {
209  PyErr_SetString(PyExc_TypeError, "Argument system must be dpsim.SystemTopology");
210  return -1;
211  }
212 
213  if (pyTearComponents) {
214  try {
215  tearComponents = compsFromPython(pyTearComponents);
216  } catch (...) {
217  PyErr_SetString(PyExc_TypeError, "tear_components must be a list of components");
218  return -1;
219  }
220  }
221 
222  Py_INCREF(self->pySys);
223 
224  if (self->realTime) {
225  self->sim = std::make_shared<DPsim::RealTimeSimulation>(name, logLevel);
226  self->sim->setSystem(*self->pySys->sys);
227  self->sim->setTimeStep(timestep);
228  self->sim->setFinalTime(duration);
229  self->sim->setDomain(domain);
230  self->sim->setSolverType(solverType);
231  self->sim->doSteadyStateInit(initSteadyState);
232  }
233  else {
234  self->sim = std::make_shared<DPsim::Simulation>(name, logLevel);
235  self->sim->setSystem(*self->pySys->sys);
236  self->sim->setTimeStep(timestep);
237  self->sim->setFinalTime(duration);
238  self->sim->setDomain(domain);
239  self->sim->setSolverType(solverType);
240  self->sim->doSteadyStateInit(initSteadyState);
241  self->sim->doSplitSubnets(splitSubnets);
242  self->sim->setTearingComponents(tearComponents);
243  }
244  self->channel = new EventChannel();
245 
246  return 0;
247 }

249 void Python::Simulation::dealloc(Python::Simulation* self)
250 {
251  if (self->thread) {
252  // We have to cancel the running thread here, because otherwise self can't
253  // be freed.
254  Python::Simulation::stop(self, nullptr);
255  self->thread->join();
256  delete self->thread;
257  }
258 
259  if (self->sim) {
260  using SimSharedPtr = std::shared_ptr<DPsim::Simulation>;
261  self->sim.~SimSharedPtr();
262  }
263 
264  delete self->mut;
265  delete self->cond;
266  delete self->channel;
267 
268  for (auto it : self->refs) {
269  Py_DECREF(it);
270  }
271 
272  // Since this is not a C++ destructor which would automatically call the
273  // destructor of its members, we have to manually call the destructor of
274  // the vectors here to free the associated memory.
275 
276  // This is a workaround for a compiler bug: https://stackoverflow.com/a/42647153/8178705
277  using PyObjectsList = std::vector<PyObject *>;
278 
279  self->refs.~PyObjectsList();
280 
281  Py_XDECREF(self->pySys);
282  Py_TYPE(self)->tp_free((PyObject*) self);
283 }

285 const char* Python::Simulation::docAddEvent =
286 "add_switch_event(sw, time, state)\n"
287 "Add a switch event to the simulation.\n"
288 "\n"
289 ":param sw: The Switch `Component` which should perform the switch action.\n"
290 ":param time: The time at which the switch action should occur.\n"
291 ":param state: Wether to open or close the switch.";
292 PyObject* Python::Simulation::addEvent(Simulation* self, PyObject* args)
293 {
294  double eventTime;
295  PyObject *pyObj, *pyVal;
296  Python::Component *pyComp;
297  const char *name;
298 
299  if (!PyArg_ParseTuple(args, "dOsO", &eventTime, &pyObj, &name, &pyVal))
300  return nullptr;
301 
302  if (!PyObject_TypeCheck(pyObj, &Python::Component::type)) {
303  PyErr_SetString(PyExc_TypeError, "First argument must be of type dpsim.Component");
304  return nullptr;
305  }
306 
307  pyComp = (Python::Component *) pyObj;
308 
309  try {
310  auto attr = pyComp->comp->attribute(name);
311  }
312  catch (InvalidAttributeException &) {
313  PyErr_SetString(PyExc_TypeError, "Invalid attribute");
314  return nullptr;
315  }
316 
317  if (PyBool_Check(pyVal)) {
318  Bool val = PyObject_IsTrue(pyVal);
319 
320  auto attr = pyComp->comp->attribute<Bool>(name);
321  if (!attr)
322  goto fail;
323 
324  auto evt = AttributeEvent<Bool>::make(eventTime, attr, val);
325  self->sim->addEvent(evt);
326  }
327  else if (PyLong_Check(pyVal)) {
328  Int val = PyLong_AsLong(pyVal);
329 
330  auto intAttr = pyComp->comp->attribute<Int>(name);
331  auto uintAttr = pyComp->comp->attribute<UInt>(name);
332  if (!intAttr && !uintAttr)
333  goto fail;
334 
335  if (intAttr) {
336  auto evt = AttributeEvent<Int>::make(eventTime, intAttr, val);
337  self->sim->addEvent(evt);
338  }
339 
340  if (uintAttr) {
341  auto evt = AttributeEvent<UInt>::make(eventTime, uintAttr, val);
342  self->sim->addEvent(evt);
343  }
344  }
345  else if (PyFloat_Check(pyVal)) {
346  double val = PyFloat_AsDouble(pyVal);
347 
348  auto attr = pyComp->comp->attribute<Real>(name);
349  if (!attr)
350  goto fail;
351 
352  auto evt = AttributeEvent<Real>::make(eventTime, attr, val);
353  self->sim->addEvent(evt);
354  }
355  else if (PyComplex_Check(pyVal)) {
356  Complex val(
357  PyComplex_RealAsDouble(pyVal),
358  PyComplex_ImagAsDouble(pyVal)
359  );
360 
361  auto attr = pyComp->comp->attribute<Complex>(name);
362  if (!attr)
363  goto fail;
364 
365  auto evt = AttributeEvent<Complex>::make(eventTime, attr, val);
366  self->sim->addEvent(evt);
367  }
368 
369  Py_RETURN_NONE;
370 
371 fail:
372  PyErr_SetString(PyExc_TypeError, "Invalid attribute or type");
373  return nullptr;
374 }
