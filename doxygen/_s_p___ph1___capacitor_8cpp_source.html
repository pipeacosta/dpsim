29 void SP::Ph1::Capacitor::initializeFromNodesAndTerminals(Real frequency) {
30 
31  Real omega = 2 * PI * frequency;
32  mSusceptance = Complex(0, omega * **mCapacitance);
33  mImpedance = Complex(1, 0) / mSusceptance;
34  mAdmittance = Complex(1, 0) / mImpedance;
35  (**mIntfVoltage)(0, 0) = initialSingleVoltage(1) - initialSingleVoltage(0);
36  **mIntfCurrent = mSusceptance * **mIntfVoltage;
37 
38  mSLog->info("\nCapacitance [F]: {:s}"
39  "\nImpedance [Ohm]: {:s}"
40  "\nAdmittance [S]: {:s}",
41  Logger::realToString(**mCapacitance),
42  Logger::complexToString(mImpedance),
43  Logger::complexToString(mAdmittance));
44  mSLog->info(
45  "\n--- Initialization from powerflow ---"
46  "\nVoltage across: {:s}"
47  "\nCurrent: {:s}"
48  "\nTerminal 0 voltage: {:s}"
49  "\nTerminal 1 voltage: {:s}"
50  "\n--- Initialization from powerflow finished ---",
51  Logger::phasorToString((**mIntfVoltage)(0, 0)),
52  Logger::phasorToString((**mIntfCurrent)(0, 0)),
53  Logger::phasorToString(initialSingleVoltage(0)),
54  Logger::phasorToString(initialSingleVoltage(1)));
55 }

74 void SP::Ph1::Capacitor::mnaApplySystemMatrixStamp(Matrix& systemMatrix) {
75 
76  if (terminalNotGrounded(0)) {
77  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0), matrixNodeIndex(0), mSusceptance);
78  }
79  if (terminalNotGrounded(1)) {
80  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1), matrixNodeIndex(1), mSusceptance);
81  }
82  if (terminalNotGrounded(0) && terminalNotGrounded(1)) {
83  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0), matrixNodeIndex(1), -mSusceptance);
84  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1), matrixNodeIndex(0), -mSusceptance);
85  }
86 
87  mSLog->info("-- Matrix Stamp ---");
88  if (terminalNotGrounded(0))
89  mSLog->info("Add {:e}+j{:e} to system at ({:d},{:d})",
90  mSusceptance.real(), mSusceptance.imag(), matrixNodeIndex(0), matrixNodeIndex(0));
91  if (terminalNotGrounded(1))
92  mSLog->info("Add {:e}+j{:e} to system at ({:d},{:d})",
93  mSusceptance.real(), mSusceptance.imag(), matrixNodeIndex(1), matrixNodeIndex(1));
94  if (terminalNotGrounded(0) && terminalNotGrounded(1)) {
95  mSLog->info("Add {:e}+j{:e} to system at ({:d},{:d})",
96  -mSusceptance.real(), -mSusceptance.imag(), matrixNodeIndex(0), matrixNodeIndex(1));
97  mSLog->info("Add {:e}+j{:e} to system at ({:d},{:d})",
98  -mSusceptance.real(), -mSusceptance.imag(), matrixNodeIndex(1), matrixNodeIndex(0));
99  }
100 }

113 void SP::Ph1::Capacitor::mnaUpdateVoltage(const Matrix& leftVector) {
114  // v1 - v0
115  **mIntfVoltage = Matrix::Zero(3, 1);
116  if (terminalNotGrounded(1)) {
117  (**mIntfVoltage)(0, 0) = Math::complexFromVectorElement(leftVector, matrixNodeIndex(1));
118  }
119  if (terminalNotGrounded(0)) {
120  (**mIntfVoltage)(0, 0) = (**mIntfVoltage)(0, 0) - Math::complexFromVectorElement(leftVector, matrixNodeIndex(0));
121  }

134 void SP::Ph1::Capacitor::calculatePerUnitParameters(Real baseApparentPower){
135  mSLog->info("#### Calculate Per Unit Parameters for {}", **mName);
136  mBaseApparentPower = baseApparentPower;
137  mSLog->info("Base Power={} [VA]", baseApparentPower);
138 
139  mBaseImpedance = mBaseVoltage * mBaseVoltage / mBaseApparentPower;
140  mBaseAdmittance = 1.0 / mBaseImpedance;
141  mBaseCurrent = baseApparentPower / (mBaseVoltage*sqrt(3)); // I_base=(S_threephase/3)/(V_line_to_line/sqrt(3))
142  mSLog->info("Base Voltage={} [V] Base Impedance={} [Ohm]", mBaseVoltage, mBaseImpedance);
143 
144  mImpedancePerUnit = mImpedance / mBaseImpedance;
145  mAdmittancePerUnit = 1. / mImpedancePerUnit;
146  mSLog->info("Impedance={} [pu] Admittance={} [pu]", Logger::complexToString(mImpedancePerUnit), Logger::complexToString(mAdmittancePerUnit));
147 }
