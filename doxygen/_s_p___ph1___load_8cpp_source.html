101 void SP::Ph1::Load::initializeFromNodesAndTerminals(Real frequency) {
102 
103  if(!mParametersSet){
104  setParameters(
105  mTerminals[0]->singleActivePower(),
106  mTerminals[0]->singleReactivePower(),
107  std::abs(mTerminals[0]->initialSingleVoltage()));
108  }
109 
110  // instantiate subResistor for active power consumption
111  if (attribute<Real>("P")->get() != 0) {
112  mResistance = std::pow(mNomVoltage, 2) / attribute<Real>("P")->get();
113  mConductance = 1.0 / mResistance;
114  mSubResistor = std::make_shared<SP::Ph1::Resistor>(mUID + "_res", mName + "_res", Logger::Level::off);
115  mSubResistor->setParameters(mResistance);
116  mSubResistor->connect({ SimNode::GND, mTerminals[0]->node() });
117  mSubResistor->initialize(mFrequencies);
118  mSubResistor->initializeFromNodesAndTerminals(frequency);
119  }
120 
121  if (attribute<Real>("Q")->get() != 0)
122  mReactance = std::pow(mNomVoltage, 2) / attribute<Real>("Q")->get();
123  else
124  mReactance = 0;
125 
126  // instantiate subInductor or subCapacitor for reactive power consumption
127  if (mReactance > 0) {
128  mInductance = mReactance / (2 * PI * frequency);
129  mSubInductor = std::make_shared<SP::Ph1::Inductor>(mUID + "_res", mName + "_ind", Logger::Level::off);
130  mSubInductor->setParameters(mInductance);
131  mSubInductor->connect({ SimNode::GND, mTerminals[0]->node() });
132  mSubInductor->initialize(mFrequencies);
133  mSubInductor->initializeFromNodesAndTerminals(frequency);
134  } else if (mReactance < 0) {
135  mCapacitance = -1 / (2 * PI * frequency) / mReactance;
136  mSubCapacitor = std::make_shared<SP::Ph1::Capacitor>(mUID + "_res", mName + "_cap", Logger::Level::off);
137  mSubCapacitor->setParameters(mCapacitance);
138  mSubCapacitor->connect({ SimNode::GND, mTerminals[0]->node() });
139  mSubCapacitor->initialize(mFrequencies);
140  mSubCapacitor->initializeFromNodesAndTerminals(frequency);
141  }
142 
143  mIntfVoltage(0, 0) = mTerminals[0]->initialSingleVoltage();
144  mIntfCurrent(0, 0) = std::conj(Complex(attribute<Real>("P")->get(), attribute<Real>("Q")->get()) / mIntfVoltage(0, 0));
145 
146  mSLog->info(
147  "\n--- Initialization from powerflow ---"
148  "\nVoltage across: {:s}"
149  "\nCurrent: {:s}"
150  "\nTerminal 0 voltage: {:s}"
151  "\n--- Initialization from powerflow finished ---",
152  Logger::phasorToString(mIntfVoltage(0, 0)),
153  Logger::phasorToString(mIntfCurrent(0, 0)),
154  Logger::phasorToString(initialSingleVoltage(0)));
155  mSLog->info(
156  "Updated parameters according to powerflow:\n"
157  "Active Power={} [W] Reactive Power={} [VAr]", attribute<Real>("P")->get(), attribute<Real>("Q")->get());
158  mSLog->flush();
159 }

163 void SP::Ph1::Load::mnaInitialize(Real omega, Real timeStep, Attribute<Matrix>::Ptr leftVector) {
164  MNAInterface::mnaInitialize(omega, timeStep);
165  updateMatrixNodeIndices();
166  mRightVector = Matrix::Zero(leftVector->get().rows(), 1);
167  if (mSubResistor) {
168  mSubResistor->mnaInitialize(omega, timeStep, leftVector);
169  for (auto task : mSubResistor->mnaTasks()) {
170  mMnaTasks.push_back(task);
171  }
172  }
173  if (mSubInductor) {
174  mSubInductor->mnaInitialize(omega, timeStep, leftVector);
175  for (auto task : mSubInductor->mnaTasks()) {
176  mMnaTasks.push_back(task);
177  }
178  }
179  if (mSubCapacitor) {
180  mSubCapacitor->mnaInitialize(omega, timeStep, leftVector);
181  for (auto task : mSubCapacitor->mnaTasks()) {
182  mMnaTasks.push_back(task);
183  }
184  }
185  mMnaTasks.push_back(std::make_shared<MnaPostStep>(*this, leftVector));
186 }
