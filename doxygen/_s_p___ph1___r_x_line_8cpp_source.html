55 void SP::Ph1::RXLine::setPerUnitSystem(Real baseApparentPower, Real baseOmega) {
56  mBaseApparentPower = baseApparentPower;
57  mBaseOmega = baseOmega;
58  mBaseImpedance = (mBaseVoltage * mBaseVoltage) / mBaseApparentPower;
59  mBaseAdmittance = 1.0 / mBaseImpedance;
60  mBaseInductance = mBaseImpedance / mBaseOmega;
62  mBaseCurrent = baseApparentPower / (mBaseVoltage * sqrt(3));
63 /*
64  mLog.Log(Logger::Level::INFO) << "#### Set Per Unit System for " << mName << std::endl;
65  mLog.Log(Logger::Level::INFO) << " Base Voltage= " << mBaseVoltage << " [V] " << " Base Impedance= " << mBaseImpedance << " [Ohm] " << std::endl;
66  */
67 
68  mSeriesResPerUnit = mSeriesRes / mBaseImpedance;
69  mSeriesIndPerUnit = mInductance / mBaseInductance;
70 /*
71  mLog.Log(Logger::Level::INFO) << "Series Resistance Per Unit= " << " " << mSeriesResPerUnit << " [Ohm] "
72  << " Series Inductance Per Unit= " << " " << mSeriesIndPerUnit << " [H] "
73  << std::endl;
74  mLog.Log(Logger::Level::INFO) << "r " << mSeriesResPerUnit << std::endl << " x: " << mBaseOmega * mInductance / mBaseImpedance<<std::endl;*/

77 void SP::Ph1::RXLine::pfApplyAdmittanceMatrixStamp(SparseMatrixCompRow & Y) {
78  updateMatrixNodeIndices();
79  int bus1 = this->matrixNodeIndex(0);
80  int bus2 = this->matrixNodeIndex(1);
81 
82  //dimension check
83  /* TODO: FIX
84  if (bus1 > (n - 1) || bus2 > (n - 1)) {
85  std::stringstream ss;
86  ss << "Line>>" << this->getName() << ": Wrong Y dimension: " << n;
87  throw std::invalid_argument(ss.str());
88  //std::cout << "Line>>" << Name << ": Wrong Y dimension: " << n << endl;
89  return;
90  }
91  */
92 
93  //create the element admittance matrix
94  Complex y = Complex(1, 0) / Complex(mSeriesResPerUnit, 1. * mSeriesIndPerUnit);
95 
96  //Fill the internal matrix
97  mY_element = MatrixComp(2, 2);
98  mY_element(0, 0) = y;
99  mY_element(0, 1) = -y;
100  mY_element(1, 0) = -y;
101  mY_element(1, 1) = y;
102 
103  //check for inf or nan
104  for (int i = 0; i < 2; i++)
105  for (int j = 0; j < 2; j++)
106  if (std::isinf(mY_element.coeff(i, j).real()) || std::isinf(mY_element.coeff(i, j).imag())) {
107  std::cout << mY_element << std::endl;
108  std::stringstream ss;
109  ss << "Line>>" << this->name() << ": infinite or nan values in the element Y at: " << i << "," << j;
110  throw std::invalid_argument(ss.str());
111  std::cout << "Line>>" << this->name() << ": infinite or nan values in the element Y at: " << i << "," << j << std::endl;
112  }
113 
114  //set the circuit matrix values
115  Y.coeffRef(bus1, bus1) += mY_element.coeff(0, 0);
116  Y.coeffRef(bus1, bus2) += mY_element.coeff(0, 1);
117  Y.coeffRef(bus2, bus1) += mY_element.coeff(1, 0);
118  Y.coeffRef(bus2, bus2) += mY_element.coeff(1, 1);
119 
120  //mLog.Log(Logger::Level::INFO) << "#### Y matrix stamping: " << std::endl;
121  //mLog.Log(Logger::Level::INFO) << mY_element << std::endl;
122 }

149 void SP::Ph1::RXLine::initializeFromNodesAndTerminals(Real frequency) {
150 
151  mIntfVoltage(0, 0) = initialSingleVoltage(1) - initialSingleVoltage(0);
152  Complex impedance = { mSeriesRes, mSeriesInd * 2. * PI * frequency };
153  mIntfCurrent(0, 0) = mIntfVoltage(0, 0) / impedance;
154  mVirtualNodes[0]->setInitialVoltage(initialSingleVoltage(0) + mIntfCurrent(0, 0) * mSeriesRes);
155 
156  // Default model with virtual node in between
157  mSubResistor = std::make_shared<SP::Ph1::Resistor>(mName + "_res", mLogLevel);
158  mSubResistor->setParameters(mSeriesRes);
159  mSubResistor->connect({ mTerminals[0]->node(), mVirtualNodes[0] });
160  mSubResistor->initializeFromNodesAndTerminals(frequency);
161 
162  mSubInductor = std::make_shared<SP::Ph1::Inductor>(mName + "_ind", mLogLevel);
163  mSubInductor->setParameters(mSeriesInd);
164  mSubInductor->connect({ mVirtualNodes[0], mTerminals[1]->node() });
165  mSubInductor->initializeFromNodesAndTerminals(frequency);
166 
167  mInitialResistor = std::make_shared<SP::Ph1::Resistor>(mName + "_snubber_res", mLogLevel);
168  mInitialResistor->setParameters(1e6);
169  mInitialResistor->connect({ SimNode::GND, mTerminals[1]->node() });
170  mInitialResistor->initializeFromNodesAndTerminals(frequency);
171 
172  mSLog->info(
173  "\n--- Initialization from powerflow ---"
174  "\nVoltage across: {:s}"
175  "\nCurrent: {:s}"
176  "\nTerminal 0 voltage: {:s}"
177  "\nTerminal 1 voltage: {:s}"
178  "\n--- Initialization from powerflow finished ---",
179  Logger::phasorToString(mIntfVoltage(0, 0)),
180  Logger::phasorToString(mIntfCurrent(0, 0)),
181  Logger::phasorToString(initialSingleVoltage(0)),
182  Logger::phasorToString(initialSingleVoltage(1)));
183 }
