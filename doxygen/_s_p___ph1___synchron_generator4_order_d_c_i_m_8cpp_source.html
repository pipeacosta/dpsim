59 void SP::Ph1::SynchronGenerator4OrderDCIM::calculateStateMatrix() {
60  Real Td_t = mTd0_t * (mLd_t / mLd);
61  Real Tq_t = mTq0_t * (mLq_t / mLq);
62  mA << -1. / Tq_t, 0,
63  0 , -1 / Td_t;
64  mB << (1. / Tq_t) * (mLq-mLq_t) / mLq, 0.0,
65  0.0, (1. / Td_t) * (mLd-mLd_t) / mLd;
66  mC << 0,
67  (1. / Td_t) * mEf * (mLd_t / mLd);
68 }

73 void SP::Ph1::SynchronGenerator4OrderDCIM::stepInPerUnit() {
74  if (mSimTime>0.0) {
75  // calculate mechanical variables at t=k+1 with forward euler
76  **mOmMech = **mOmMech + mTimeStep * (1. / (2. * mH) * (mMechTorque - **mElecTorque));
77  **mThetaMech = **mThetaMech + mTimeStep * (**mOmMech * mBase_OmMech);
78  **mDelta = **mDelta + mTimeStep * (**mOmMech - 1.) * mBase_OmMech;
79  **mElecTorque = ((**mVdq)(0,0) * (**mIdq)(0,0) + (**mVdq)(1,0) * (**mIdq)(1,0));
80  }
81 
82  // get transformation matrix
83  mDqToComplexA = get_DqToComplexATransformMatrix();
84  mComplexAToDq = mDqToComplexA.transpose();
85 
86  // calculate Edq at t=k+1. Assumption: Vdq(k) = Vdq(k+1)
87  (**mEdq_t) = Math::StateSpaceTrapezoidal(**mEdq_t, mA, mB, mC, mTimeStep, **mVdq);
88 
89  // armature currents for at t=k+1
90  (**mIdq)(0,0) = ((**mEdq_t)(1,0) - (**mVdq)(1,0) ) / mLd_t;
91  (**mIdq)(1,0) = ((**mVdq)(0,0) - (**mEdq_t)(0,0) ) / mLq_t;
92 
93  // convert currents into the abc domain
94  Matrix Ia = mDqToComplexA * **mIdq;
95  (**mIntfCurrent)(0,0) = Complex(Ia(0,0), Ia(1,0)) * mBase_I_RMS;
96 }

102 void SP::Ph1::SynchronGenerator4OrderDCIM::mnaPostStep(const Matrix& leftVector) {
103  // update armature voltage
104  (**mIntfVoltage)(0, 0) = Math::complexFromVectorElement(leftVector, matrixNodeIndex(0));
105  Matrix Vabc = Matrix::Zero(2,1);
106  Vabc << (**mIntfVoltage)(0, 0).real(), (**mIntfVoltage)(0, 0).imag();
107  **mVdq = mComplexAToDq * Vabc / mBase_V_RMS;
108 
109  mSimTime = mSimTime + mTimeStep;

112 Matrix SP::Ph1::SynchronGenerator4OrderDCIM::get_DqToComplexATransformMatrix() {
113  Matrix dqToComplexA(2, 2);
114  dqToComplexA <<
115  cos(**mThetaMech - mBase_OmMech * mSimTime), -sin(**mThetaMech - mBase_OmMech * mSimTime),
116  sin(**mThetaMech - mBase_OmMech * mSimTime), cos(**mThetaMech - mBase_OmMech * mSimTime);
