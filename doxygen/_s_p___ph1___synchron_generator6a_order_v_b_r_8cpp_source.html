38 void SP::Ph1::SynchronGenerator6aOrderVBR::specificInitialization() {
39 
40  // auxiliar VBR constants
41  calculateAuxiliarConstants();
42 
43  // initial voltage behind the transient reactance in the dq reference frame
44  mEdq_t(0,0) = (mLq - mLq_t - mYq) * mIdq(1,0);
45  mEdq_t(1,0) = (1 - mTaa / mTd0_t) * mEf - (mLd - mLd_t - mYd) * mIdq(0,0);
46 
47  // initial dq behind the subtransient reactance in the dq reference frame
48  mEdq_s(0,0) = mVdq(0,0) - mLq_s * mIdq(1,0);
49  mEdq_s(1,0) = mVdq(1,0) + mLd_s * mIdq(0,0);
50 
51  // initialize conductance matrix
52  mConductanceMatrix = Matrix::Zero(2,2);
53 
54  // calculate resistance matrix in dq reference frame
55  mResistanceMatrixDq = Matrix::Zero(2,2);
56  mResistanceMatrixDq << 0.0, -mLq_s - mAd_s,
57  mLd_s - mAq_s, 0.0;
58 
59  mSLog->info(
60  "\n--- Model specific initialization ---"
61  "\nInitial Ed_t (per unit): {:f}"
62  "\nInitial Eq_t (per unit): {:f}"
63  "\nInitial Ed_s (per unit): {:f}"
64  "\nInitial Eq_s (per unit): {:f}"
65  "\n--- Model specific initialization finished ---",
66 
67  mEdq_t(0,0),
68  mEdq_t(1,0),
69  mEdq_s(0,0),
70  mEdq_s(1,0)
71  );

73 }

75 void SP::Ph1::SynchronGenerator6aOrderVBR::calculateAuxiliarConstants() {
76  mYd = (mTd0_s / mTd0_t) * (mLd_s / mLd_t) * (mLd - mLd_t);
77  mYq = (mTq0_s / mTq0_t) * (mLq_s / mLq_t) * (mLq - mLq_t);
78  Real Zq_t = mLd - mLd_t - mYd;
79  Real Zd_t = mLq - mLq_t - mYq;
80  Real Zq_s = mLd_t - mLd_s + mYd;
81  Real Zd_s = mLq_t - mLq_s + mYq;
82  Real Tf = mTaa / mTd0_t;
83 
84  mAd_t = mTimeStep * Zd_t / (2 * mTq0_t + mTimeStep);
85  mBd_t = (2 * mTq0_t - mTimeStep) / (2 * mTq0_t + mTimeStep);
86  mAq_t = - mTimeStep * Zq_t / (2 * mTd0_t + mTimeStep);
87  mBq_t = (2 * mTd0_t - mTimeStep) / (2 * mTd0_t + mTimeStep);
88  mDq_t = 2 * mTimeStep * (1 - Tf) / (2 * mTd0_t + mTimeStep);
89 
90  mAd_s = (mTimeStep * Zd_s + mTimeStep * mAd_t) / (2 * mTq0_s + mTimeStep);
91  mBd_s = (mTimeStep * mBd_t + mTimeStep) / (2 * mTq0_s + mTimeStep);
92  mCd_s = (2 * mTq0_s - mTimeStep) / (2 * mTq0_s + mTimeStep);
93  mAq_s = (-mTimeStep * Zq_s + mTimeStep * mAq_t ) / (2 * mTd0_s + mTimeStep);
94  mBq_s = (mTimeStep * mBq_t + mTimeStep) / (2 * mTd0_s + mTimeStep);
95  mCq_s = (2 * mTd0_s - mTimeStep) / (2 * mTd0_s + mTimeStep);
96  mDq_s = (mTimeStep * mDq_t + 2 * Tf * mTimeStep) / (2 * mTd0_s + mTimeStep);
97 }

99 void SP::Ph1::SynchronGenerator6aOrderVBR::stepInPerUnit() {
100  if (mSimTime>0.0) {
101  // calculate Edq_t at t=k
102  mEdq_t(0,0) = mAd_t * mIdq(1,0) + mEh_t(0,0);
103  mEdq_t(1,0) = mAq_t * mIdq(0,0) + mEh_t(1,0);
104 
105  // calculate Edq_s at t=k
106  mEdq_s(0,0) = -mIdq(1,0) * mLq_s + mVdq(0,0);
107  mEdq_s(1,0) = mIdq(0,0) * mLd_s + mVdq(1,0);
108 
109  // calculate mechanical variables at t=k+1 with forward euler
110  mElecTorque = (mVdq(0,0) * mIdq(0,0) + mVdq(1,0) * mIdq(1,0));
111  mOmMech = mOmMech + mTimeStep * (1. / (2. * mH) * (mMechTorque - mElecTorque));
112  mThetaMech = mThetaMech + mTimeStep * (mOmMech * mBase_OmMech);
113  mDelta = mDelta + mTimeStep * (mOmMech - 1.) * mBase_OmMech;
114  }
115 
116  mDqToComplexA = get_DqToComplexATransformMatrix();
117  mComplexAToDq = mDqToComplexA.transpose();
118 
119  // calculate resistance matrix at t=k+1
120  calculateResistanceMatrix();
121 
122  // calculate history term behind the transient reactance
123  mEh_t(0,0) = mAd_t * mIdq(1,0) + mBd_t * mEdq_t(0,0);
124  mEh_t(1,0) = mAq_t * mIdq(0,0) + mBq_t * mEdq_t(1,0) + mDq_t * mEf;
125 
126  // calculate history term behind the subtransient reactance
127  mEh_s(0,0) = mAd_s * mIdq(1,0) + mBd_s * mEdq_t(0,0) + mCd_s * mEdq_s(0,0);
128  mEh_s(1,0) = mAq_s * mIdq(0,0) + mBq_s * mEdq_t(1,0) + mCq_s * mEdq_s(1,0) + mDq_s * mEf;
129 
130  // convert Edq_t into the abc reference frame
131  mEh_s = mDqToComplexA * mEh_s;
132  Evbr = Complex(mEh_s(0,0), mEh_s(1,0)) * mBase_V_RMS;
133 }
