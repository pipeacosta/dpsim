47 void SP::Ph1::SynchronGeneratorTrStab::setFundamentalParametersPU(Real nomPower, Real nomVolt, Real nomFreq,
48  Real Ll, Real Lmd, Real Llfd, Real inertia, Real D) {
49  setBaseParameters(nomPower, nomVolt, nomFreq);
50  mSLog->info("\n--- Base Parameters ---"
51  "\nnomPower: {:f}"
52  "\nnomVolt: {:f}"
53  "\nnomFreq: {:f}", mNomPower, mNomVolt, mNomFreq);
54 
55  // Input is in per unit but all values are converted to absolute values.
56  mParameterType = ParameterType::statorReferred;
57  mStateType = StateType::statorReferred;
58 
59  **mLl = Ll;
60  mLmd = Lmd;
61  **mLd = **mLl + mLmd;
62  mLlfd = Llfd;
63  mLfd = mLlfd + mLmd;
64  // M = 2*H where H = inertia
65  **mInertia = inertia;
66  // X'd in absolute values
67  mXpd = mNomOmega * (**mLd - mLmd*mLmd / mLfd) * mBase_L;
68  mLpd = (**mLd - mLmd*mLmd / mLfd) * mBase_L;
69 
70  //The units of D are per unit power divided by per unit speed deviation.
71  // D is transformed to an absolute value to obtain Kd, which will be used in the swing equation
72  mKd= D*mNomPower/mNomOmega;
73 
74  mSLog->info("\n--- Parameters ---"
75  "\nimpedance: {:f}"
76  "\ninductance: {:f}"
77  "\ninertia: {:f}"
78  "\ndamping: {:f}", mXpd, mLpd, **mInertia, mKd);
79 }

153 void SP::Ph1::SynchronGeneratorTrStab::initializeFromNodesAndTerminals(Real frequency) {
154 
155  // Initialize omega mech with nominal system frequency
156  **mOmMech = mNomOmega;
157 
158  // Static calculation based on load flow
159  (**mIntfVoltage)(0,0) = initialSingleVoltage(0);
160  mInitElecPower = (mInitElecPower == Complex(0,0))
161  ? -terminal(0)->singlePower()
162  : mInitElecPower;
163  mInitMechPower = (mInitElecPower == Complex(0,0))
164  ? mInitElecPower.real()
165  : mInitMechPower;
166 
167  //I_intf is the current which is flowing into the Component, while mInitElecPower is flowing out of it
168  (**mIntfCurrent)(0,0) = std::conj( - mInitElecPower / (**mIntfVoltage)(0,0) );
169 
170  mImpedance = Complex(**mRs, mXpd);
171 
172  // Calculate initial emf behind reactance from power flow results
173  **mEp = (**mIntfVoltage)(0,0) - mImpedance * (**mIntfCurrent)(0,0);
174 
175  // The absolute value of Ep is constant, only delta_p changes every step
176  **mEp_abs = Math::abs(**mEp);
177  // Delta_p is the angular position of mEp with respect to the synchronously rotating reference
178  **mDelta_p= Math::phase(**mEp);
179 
180  // Update electrical power
181  // TODO: review for Rs != 0
182  **mElecActivePower = ( (**mIntfVoltage)(0,0) * std::conj( -(**mIntfCurrent)(0,0)) ).real();
183  **mElecReactivePower = ( (**mIntfVoltage)(0,0) * std::conj( -(**mIntfCurrent)(0,0)) ).imag();
184 
185  // Start in steady state so that electrical and mech. power are the same
186  // because of the initial condition mOmMech = mNomOmega the damping factor is not considered at the initialisation
187  // TODO: review for Rs != 0
188  **mMechPower = **mElecActivePower;
189 
190  // Initialize node between X'd and Ep
191  mVirtualNodes[0]->setInitialVoltage(**mEp);
192 
193  // Create sub voltage source for emf
194  mSubVoltageSource = SP::Ph1::VoltageSource::make(**mName + "_src", mLogLevel);
195  mSubVoltageSource->setParameters(**mEp);
196  mSubVoltageSource->connect({SimNode::GND, mVirtualNodes[0]});
197  mSubVoltageSource->setVirtualNodeAt(mVirtualNodes[1], 0);
198  mSubVoltageSource->initialize(mFrequencies);
199  mSubVoltageSource->initializeFromNodesAndTerminals(frequency);
200 
201  // Create sub inductor as Xpd
202  mSubInductor = SP::Ph1::Inductor::make(**mName + "_ind", mLogLevel);
203  mSubInductor->setParameters(mLpd);
204  mSubInductor->connect({mVirtualNodes[0], terminal(0)->node()});
205  mSubInductor->initialize(mFrequencies);
206  mSubInductor->initializeFromNodesAndTerminals(frequency);
207 
208  mSLog->info("\n--- Initialize according to powerflow ---"
209  "\nTerminal 0 voltage: {:e}<{:e}"
210  "\nVoltage behind reactance: {:e}<{:e}"
211  "\ninitial electrical power: {:e}+j{:e}"
212  "\nactive electrical power: {:e}"
213  "\nreactive electrical power: {:e}"
214  "\nmechanical power: {:e}"
215  "\n--- End of powerflow initialization ---",
216  Math::abs((**mIntfVoltage)(0,0)), Math::phaseDeg((**mIntfVoltage)(0,0)),
217  Math::abs(**mEp), Math::phaseDeg(**mEp),
218  mInitElecPower.real(), mInitElecPower.imag(),
219  **mElecActivePower, **mElecReactivePower, **mMechPower);
220 }

222 void SP::Ph1::SynchronGeneratorTrStab::step(Real time) {
223 
224  // #### Calculations based on values from time step k ####
225  // Electrical power at time step k
226  // TODO: review for Rs != 0
227  **mElecActivePower = ((**mIntfVoltage)(0,0) * std::conj( -(**mIntfCurrent)(0,0)) ).real();
228  **mElecReactivePower = ((**mIntfVoltage)(0,0) * std::conj( -(**mIntfCurrent)(0,0)) ).imag();
229 
230  // Mechanical speed derivative at time step k
231  // convert torque to power with actual rotor angular velocity or nominal omega
232  Real dOmMech;
233  if (mConvertWithOmegaMech)
234  dOmMech = mNomOmega*mNomOmega / (2.* **mInertia * mNomPower* **mOmMech) * (**mMechPower - **mElecActivePower - mKd*(**mOmMech - mNomOmega));
235  else
236  dOmMech = mNomOmega / (2.* **mInertia * mNomPower) * (**mMechPower - **mElecActivePower - mKd*(**mOmMech - mNomOmega));
237 
238  // #### Calculate states for time step k+1 applying semi-implicit Euler ####
239  // Mechanical speed at time step k+1 applying Euler forward
240  if (mBehaviour == Behaviour::MNASimulation)
241  **mOmMech = **mOmMech + mTimeStep * dOmMech;
242 
243  // Derivative of rotor angle at time step k + 1
244  // if reference omega is set, calculate delta with respect to reference
245  Real refOmega;
246  Real refDelta;
247  if (mUseOmegaRef) {
248  refOmega = attribute<Real>("w_ref")->get();
249  refDelta = attribute<Real>("delta_ref")->get();
250  } else {
251  refOmega = mNomOmega;
252  refDelta = 0;
253  }
254  Real dDelta_p = **mOmMech - refOmega;
255 
256  // Rotor angle at time step k + 1 applying Euler backward
257  // Update emf - only phase changes
258  if (mBehaviour == Behaviour::MNASimulation) {
259  **mDelta_p = **mDelta_p + mTimeStep * dDelta_p;
260  **mEp = Complex(**mEp_abs * cos(**mDelta_p), **mEp_abs * sin(**mDelta_p));
261  }
262 
263  mStates << Math::abs(**mEp), Math::phaseDeg(**mEp), **mElecActivePower, **mMechPower,
264  **mDelta_p, **mOmMech, dOmMech, dDelta_p, (**mIntfVoltage)(0,0).real(), (**mIntfVoltage)(0,0).imag();
265  SPDLOG_LOGGER_DEBUG(mSLog, "\nStates, time {:f}: \n{:s}", time, Logger::matrixToString(mStates));
266 }

268 void SP::Ph1::SynchronGeneratorTrStab::mnaInitialize(Real omega, Real timeStep, Attribute<Matrix>::Ptr leftVector) {
269  MNAInterface::mnaInitialize(omega, timeStep);
