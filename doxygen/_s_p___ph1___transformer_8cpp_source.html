98 void SP::Ph1::Transformer::initializeFromNodesAndTerminals(Real frequency) {
99  mNominalOmega = 2. * PI * frequency;
100  mReactance = mNominalOmega * **mInductance;
101  mSLog->info("Reactance={} [Ohm] (referred to primary side)", mReactance);
102 
103  // Component parameters are referred to higher voltage side.
104  // Switch terminals to have terminal 0 at higher voltage side
105  // if transformer is connected the other way around.
106  if (Math::abs(**mRatio) < 1.) {
107  **mRatio = 1. / **mRatio;
108  mRatioAbs = std::abs(**mRatio);
109  mRatioPhase = std::arg(**mRatio);
110  std::shared_ptr<SimTerminal<Complex>> tmp = mTerminals[0];
111  mTerminals[0] = mTerminals[1];
112  mTerminals[1] = tmp;
113  Real tmpVolt = **mNominalVoltageEnd1;
114  **mNominalVoltageEnd1 = **mNominalVoltageEnd2;
115  **mNominalVoltageEnd2 = tmpVolt;
116  mSLog->info("Switching terminals to have first terminal at higher voltage side. Updated parameters: ");
117  mSLog->info("Nominal Voltage End 1 = {} [V] Nominal Voltage End 2 = {} [V]", **mNominalVoltageEnd1, **mNominalVoltageEnd2);
118  mSLog->info("Tap Ratio = {} [ ] Phase Shift = {} [deg]", mRatioAbs, mRatioPhase);
119  }
120 
121  // Set initial voltage of virtual node in between
122  mVirtualNodes[0]->setInitialVoltage(initialSingleVoltage(1) * **mRatio);
123 
124  // Static calculations from load flow data
125  Complex impedance = { **mResistance, mReactance };
126  (**mIntfVoltage)(0, 0) = mVirtualNodes[0]->initialSingleVoltage() - initialSingleVoltage(0);
127  (**mIntfCurrent)(0, 0) = (**mIntfVoltage)(0, 0) / impedance;
128 
129  // Create series sub components
130  mSubInductor = std::make_shared<SP::Ph1::Inductor>(**mUID + "_ind", **mName + "_ind", Logger::Level::off);
131  mSubInductor->setParameters(**mInductance);
132  mSubComponents.push_back(mSubInductor);
133 
134  if (mNumVirtualNodes == 3) {
135  mVirtualNodes[2]->setInitialVoltage(initialSingleVoltage(0));
136  mSubResistor = std::make_shared<SP::Ph1::Resistor>(**mUID + "_res", **mName + "_res", Logger::Level::off);
137  mSubResistor->setParameters(**mResistance);
138  mSubResistor->connect({ node(0), mVirtualNodes[2] });
139  mSubInductor->connect({ mVirtualNodes[2], mVirtualNodes[0] });
140  mSubComponents.push_back(mSubResistor);
141  } else {
142  mSubInductor->connect({ node(0), mVirtualNodes[0] });
143  }
144 
145  // Create parallel sub components for init and mna behaviour
146  if (mBehaviour == TopologicalPowerComp::Behaviour::Initialization || mBehaviour == TopologicalPowerComp::Behaviour::MNASimulation) {
147 
148  Real pSnub = P_SNUB_TRANSFORMER * **mRatedPower;
149  Real qSnub = Q_SNUB_TRANSFORMER * **mRatedPower;
150 
151  // A snubber conductance is added on the higher voltage side
152  mSnubberResistance1 = std::pow(std::abs(**mNominalVoltageEnd1),2) / pSnub;
153  mSubSnubResistor1 = std::make_shared<SP::Ph1::Resistor>(**mName + "_snub_res1", mLogLevel);
154  mSubSnubResistor1->setParameters(mSnubberResistance1);
155  mSubSnubResistor1->connect({ node(0), SP::SimNode::GND });
156  mSLog->info("Snubber Resistance 1 (connected to higher voltage side {}) = {} [Ohm]", node(0)->name(), Logger::realToString(mSnubberResistance1));
157  mSubSnubResistor1->setBaseVoltage(**mNominalVoltageEnd1);
158  mSubComponents.push_back(mSubSnubResistor1);
159 
160  // A snubber conductance is added on the lower voltage side
161  mSnubberResistance2 = std::pow(std::abs(**mNominalVoltageEnd2),2) / pSnub;
162  mSubSnubResistor2 = std::make_shared<SP::Ph1::Resistor>(**mName + "_snub_res2", mLogLevel);
163  mSubSnubResistor2->setParameters(mSnubberResistance2);
164  mSubSnubResistor2->connect({ node(1), SP::SimNode::GND });
165  mSLog->info("Snubber Resistance 2 (connected to lower voltage side {}) = {} [Ohm]", node(1)->name(), Logger::realToString(mSnubberResistance2));
166  mSubSnubResistor2->setBaseVoltage(**mNominalVoltageEnd2);
167  mSubComponents.push_back(mSubSnubResistor2);
168 
169  // // A snubber capacitance is added to higher voltage side (not used as capacitor at high voltage side made it worse)
170  // mSnubberCapacitance1 = qSnub / std::pow(std::abs(mNominalVoltageEnd1),2) / mNominalOmega;
171  // mSubSnubCapacitor1 = std::make_shared<SP::Ph1::Capacitor>(**mName + "_snub_cap1", mLogLevel);
172  // mSubSnubCapacitor1->setParameters(mSnubberCapacitance1);
173  // mSubSnubCapacitor1->connect({ node(0), SP::SimNode::GND });
174  // mSLog->info("Snubber Capacitance 1 (connected to higher voltage side {}) = \n{} [F] \n ", node(0)->name(), Logger::realToString(mSnubberCapacitance1));
175  // mSubSnubCapacitor1->setBaseVoltage(mNominalVoltageEnd1);
176  // mSubComponents.push_back(mSubSnubCapacitor1);
177 
178  // A snubber capacitance is added to lower voltage side
179  mSnubberCapacitance2 = qSnub / std::pow(std::abs(**mNominalVoltageEnd2),2) / mNominalOmega;
180  mSubSnubCapacitor2 = std::make_shared<SP::Ph1::Capacitor>(**mName + "_snub_cap2", mLogLevel);
181  mSubSnubCapacitor2->setParameters(mSnubberCapacitance2);
182  mSubSnubCapacitor2->connect({ node(1), SP::SimNode::GND });
183  mSLog->info("Snubber Capacitance 2 (connected to lower voltage side {}) = {} [F]", node(1)->name(), Logger::realToString(mSnubberCapacitance2));
184  mSubSnubCapacitor2->setBaseVoltage(**mNominalVoltageEnd2);
185  mSubComponents.push_back(mSubSnubCapacitor2);
186  }
187 
188  // Initialize electrical subcomponents
189  mSLog->info("Electrical subcomponents: ");
190  for (auto subcomp: mSubComponents) {
191  mSLog->info("- {}", subcomp->name());
192  subcomp->initialize(mFrequencies);
193  subcomp->initializeFromNodesAndTerminals(frequency);
194  }
195 
196  mSLog->info(
197  "\n--- Initialization from powerflow ---"
198  "\nVoltage across: {:s}"
199  "\nCurrent: {:s}"
200  "\nTerminal 0 voltage: {:s}"
201  "\nTerminal 1 voltage: {:s}"
202  "\nVirtual Node 1 voltage: {:s}"
203  "\n--- Initialization from powerflow finished ---",
204  Logger::phasorToString((**mIntfVoltage)(0, 0)),
205  Logger::phasorToString((**mIntfCurrent)(0, 0)),
206  Logger::phasorToString(initialSingleVoltage(0)),
207  Logger::phasorToString(initialSingleVoltage(1)),
208  Logger::phasorToString(mVirtualNodes[0]->initialSingleVoltage()));
209 }

220 void SP::Ph1::Transformer::calculatePerUnitParameters(Real baseApparentPower, Real baseOmega) {
221  mSLog->info("#### Calculate Per Unit Parameters for {}", **mName);
222  mBaseApparentPower = baseApparentPower;
223  mBaseOmega = baseOmega;
224  mSLog->info("Base Power={} [VA] Base Omega={} [1/s]", baseApparentPower, baseOmega);
225 
226  mBaseImpedance = **mBaseVoltage * **mBaseVoltage / mBaseApparentPower;
227  mBaseAdmittance = 1.0 / mBaseImpedance;
228  mBaseCurrent = baseApparentPower / (**mBaseVoltage * sqrt(3)); // I_base=(S_threephase/3)/(V_line_to_line/sqrt(3))
229  mSLog->info("Base Voltage={} [V] Base Impedance={} [Ohm]", **mBaseVoltage, mBaseImpedance);
230 
231  mResistancePerUnit = **mResistance / mBaseImpedance;
232  mReactancePerUnit = mReactance / mBaseImpedance;
233  mSLog->info("Resistance={} [pu] Reactance={} [pu]", mResistancePerUnit, mReactancePerUnit);
234 
235  mBaseInductance = mBaseImpedance / mBaseOmega;
236  mInductancePerUnit = **mInductance / mBaseInductance;
237  mMagnetizing = mMagnetizingReactance / mBaseImpedance;
238  // omega per unit=1, hence 1.0*mInductancePerUnit.
239  mLeakagePerUnit = Complex(mResistancePerUnit,1.*mInductancePerUnit);
240  mMagnetizingPerUnit = mMagnetizing / mBaseImpedance;
241  mSLog->info("Leakage Impedance Per Unit={} [Ohm] ", mLeakagePerUnit);
242 
243  mRatioAbsPerUnit = mRatioAbs / **mNominalVoltageEnd1 * **mNominalVoltageEnd2;
244  mSLog->info("Tap Ratio={} [pu]", mRatioAbsPerUnit);
245 
246  // Calculate per unit parameters of subcomps
247  if (mSubSnubResistor1)
248  mSubSnubResistor1->calculatePerUnitParameters(mBaseApparentPower);
249  if (mSubSnubResistor2)
250  mSubSnubResistor2->calculatePerUnitParameters(mBaseApparentPower);
251  if (mSubSnubCapacitor1)
252  mSubSnubCapacitor1->calculatePerUnitParameters(mBaseApparentPower);
253  if (mSubSnubCapacitor2)
254  mSubSnubCapacitor2->calculatePerUnitParameters(mBaseApparentPower);
255 }

257 void SP::Ph1::Transformer::pfApplyAdmittanceMatrixStamp(SparseMatrixCompRow & Y) {
258  // calculate matrix stamp
259  mY_element = MatrixComp(2, 2);
260  Complex y = Complex(1, 0) / mLeakagePerUnit;
261  Complex ys = Complex(1, 0) / mMagnetizingPerUnit;
262  mY_element(0, 0) = (y + ys);
263  mY_element(0, 1) = -y*mRatioAbsPerUnit;
264  mY_element(1, 0) = -y*mRatioAbsPerUnit;
265  mY_element(1, 1) = (y + ys)*std::pow(mRatioAbsPerUnit, 2);
266 
267  //check for inf or nan
268  for (int i = 0; i < 2; i++)
269  for (int j = 0; j < 2; j++)
270  if (std::isinf(mY_element.coeff(i, j).real()) || std::isinf(mY_element.coeff(i, j).imag())) {
271  std::cout << mY_element << std::endl;
272  std::cout << "Zm:" << mMagnetizing << std::endl;
273  std::cout << "Zl:" << mLeakage << std::endl;
274  std::cout << "tap:" << mRatioAbsPerUnit << std::endl;
275  std::stringstream ss;
276  ss << "Transformer>>" << this->name() << ": infinite or nan values in the element Y at: " << i << "," << j;
277  throw std::invalid_argument(ss.str());
278  }
279 
280  //set the circuit matrix values
281  Y.coeffRef(this->matrixNodeIndex(0), this->matrixNodeIndex(0)) += mY_element.coeff(0, 0);
282  Y.coeffRef(this->matrixNodeIndex(0), this->matrixNodeIndex(1)) += mY_element.coeff(0, 1);
283  Y.coeffRef(this->matrixNodeIndex(1), this->matrixNodeIndex(1)) += mY_element.coeff(1, 1);
284  Y.coeffRef(this->matrixNodeIndex(1), this->matrixNodeIndex(0)) += mY_element.coeff(1, 0);
285 
286  mSLog->info("#### Y matrix stamping: {}", mY_element);
287 
288  if (mSubSnubResistor1)
289  mSubSnubResistor1->pfApplyAdmittanceMatrixStamp(Y);
290  if (mSubSnubResistor2)
291  mSubSnubResistor2->pfApplyAdmittanceMatrixStamp(Y);
292  if (mSubSnubCapacitor1)
293  mSubSnubCapacitor1->pfApplyAdmittanceMatrixStamp(Y);
294  if (mSubSnubCapacitor2)
295  mSubSnubCapacitor2->pfApplyAdmittanceMatrixStamp(Y);
296 }

340  // Ideal transformer equations

342  Math::setMatrixElement(systemMatrix, mVirtualNodes[0]->matrixNodeIndex(), mVirtualNodes[1]->matrixNodeIndex(), Complex(-1.0, 0));
343  Math::setMatrixElement(systemMatrix, mVirtualNodes[1]->matrixNodeIndex(), mVirtualNodes[0]->matrixNodeIndex(), Complex(1.0, 0));

346  Math::setMatrixElement(systemMatrix, matrixNodeIndex(1), mVirtualNodes[1]->matrixNodeIndex(), **mRatio);

353  mnasubcomp->mnaApplySystemMatrixStamp(systemMatrix);
