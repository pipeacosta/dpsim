65 void SP::Ph1::VoltageSource::initializeFromNodesAndTerminals(Real frequency) {
66  Complex voltageRef = **mVoltageRef;
67 
68  if (voltageRef == Complex(0, 0))
69  voltageRef = initialSingleVoltage(1) - initialSingleVoltage(0);
70 
71  if (mSrcSig == nullptr) {
72  Signal::SineWaveGenerator srcSigSine(**mName);
73  srcSigSine.setParameters(voltageRef);
74  mSrcSig = std::make_shared<Signal::SineWaveGenerator>(srcSigSine);
75 
76  mVoltageRef->setReference(mSrcSig->mSigOut);
77  mSrcFreq->setReference(mSrcSig->mFreq);
78  } else {
79  **mVoltageRef = voltageRef;
80  }
81 
82  mSLog->info(
83  "\n--- Initialization from node voltages ---"
84  "\nVoltage across: {:s}"
85  "\nTerminal 0 voltage: {:s}"
86  "\nTerminal 1 voltage: {:s}"
87  "\n--- Initialization from node voltages ---",
88  Logger::phasorToString(mSrcSig->getSignal()),
89  Logger::phasorToString(initialSingleVoltage(0)),
90  Logger::phasorToString(initialSingleVoltage(1)));
91 }

124 void SP::Ph1::VoltageSource::mnaApplySystemMatrixStamp(Matrix& systemMatrix) {
125  for (UInt freq = 0; freq < mNumFreqs; freq++) {
126  if (terminalNotGrounded(0)) {
127  Math::setMatrixElement(systemMatrix, mVirtualNodes[0]->matrixNodeIndex(), matrixNodeIndex(0), Complex(-1, 0), mNumFreqs, freq);
128  Math::setMatrixElement(systemMatrix, matrixNodeIndex(0), mVirtualNodes[0]->matrixNodeIndex(), Complex(-1, 0), mNumFreqs, freq);
129  }
130  if (terminalNotGrounded(1)) {
131  Math::setMatrixElement(systemMatrix, mVirtualNodes[0]->matrixNodeIndex(), matrixNodeIndex(1), Complex(1, 0), mNumFreqs, freq);
132  Math::setMatrixElement(systemMatrix, matrixNodeIndex(1), mVirtualNodes[0]->matrixNodeIndex(), Complex(1, 0), mNumFreqs, freq);
133  }
134 
135  mSLog->info("-- Stamp frequency {:d} ---", freq);
136  if (terminalNotGrounded(0)) {
137  mSLog->info("Add {:f} to system at ({:d},{:d})", -1., matrixNodeIndex(0), mVirtualNodes[0]->matrixNodeIndex());
138  mSLog->info("Add {:f} to system at ({:d},{:d})", -1., mVirtualNodes[0]->matrixNodeIndex(), matrixNodeIndex(0));
139  }
140  if (terminalNotGrounded(1)) {
141  mSLog->info("Add {:f} to system at ({:d},{:d})", 1., mVirtualNodes[0]->matrixNodeIndex(), matrixNodeIndex(1));
142  mSLog->info("Add {:f} to system at ({:d},{:d})", 1., matrixNodeIndex(1), mVirtualNodes[0]->matrixNodeIndex());
143  }
144  }
145 }
