123 Bool SP::Ph1::varResSwitch::hasParameterChanged() {
124 //Get present state
125 Bool presentState=this->mnaIsClosed();
126 
127 // Check if state of switch changed from open to closed
128 if (!(mPrevState == presentState)) {
129  // Switch is closed : change with 1/mDeltaRes
130  if (this->mnaIsClosed()==true) {
131  // mClosedResistance= 1./mDeltaRes*mPrevRes;
132  **mClosedResistance= mDeltaResClosed*mPrevRes;
133  mPrevRes= **mClosedResistance;
134  // check if target value is reached
135  if (**mClosedResistance < mInitClosedRes) {
136  **mClosedResistance= mInitClosedRes;
137  mPrevRes= **mClosedResistance;
138  mPrevState= this->mnaIsClosed();
139  }
140  }
141  // Switch is opened : change with mDeltaRes
142  else if (this->mnaIsClosed()==false) {
143  **mOpenResistance= mDeltaResOpen*mPrevRes;
144  mPrevRes= **mOpenResistance;
145  // check if target value is reached
146  if ( **mOpenResistance > mInitOpenRes) {
147  **mOpenResistance= mInitOpenRes;
148  mPrevRes= **mOpenResistance;
149  mPrevState= this->mnaIsClosed();
150  }
151  }
152  return 1; //recompute system matrix
153 }
154 else{
155  return 0; // do not recompute system matrix
156  }
157 }
