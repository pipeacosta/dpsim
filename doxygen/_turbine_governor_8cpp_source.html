36 Real Signal::TurbineGovernor::step(Real Om, Real OmRef, Real PmRef, Real dt) {
37  // ### Governing ###
38  // Input of speed relay
39  Psr_in = PmRef + (OmRef - Om)*mK;
40  // Input of servor motor
41  Psm_in = Math::StateSpaceEuler(Psm_in, -1, 1, dt / mTsr, Psr_in);
42  // rate of change of valve
43  mpVcv = (Psm_in - mVcv) / mTsm;
44  if (mpVcv >= 0.1)
45  mpVcv = 0.1;
46  else if (mpVcv <= -1)
47  mpVcv = -1;
48  //Valve position
49  mVcv = mVcv + dt*mpVcv;
50  if (mVcv >= 1)
51  mVcv = 1;
52  else if (mVcv <= 0)
53  mVcv = 0;
54 
55  //### Turbine ###
56  // Simplified equation
57 
58  T1 = Math::StateSpaceEuler(T1, -1, (1 - mFa), dt / mTa, mVcv);
59  T2 = mVcv*mFa;
60  mTm = T1 + T2;
61 
62  return mTm;
63 }
