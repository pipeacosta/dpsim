95 Logger::Log Logger::create(const std::string &name, Level filelevel, Level clilevel) {
96  String logDir = Logger::logDir();
97  String filename = logDir + "/" + name + ".log";
98  std::vector<spdlog::sink_ptr> sinks;
99  Logger::Log ret;
100 
101  // Create log folder if it does not exist
102  fs::path p = filename;
103  if (p.has_parent_path() && !fs::exists(p.parent_path()))
104  fs::create_directories(p.parent_path());
105 
106  if (clilevel != Logger::Level::off) {
107  auto console_sink = std::make_shared<spdlog::sinks::stderr_color_sink_mt>();
108  console_sink->set_level(clilevel);
109  console_sink->set_pattern(fmt::format("{}[%T.%f %n %^%l%$] %v", CPS::Logger::prefix()));
110  sinks.push_back(console_sink);
111  }
112 
113  if (filelevel != Logger::Level::off) {
114  auto file_sink = std::make_shared<spdlog::sinks::basic_file_sink_mt>(filename, true);
115  file_sink->set_level(filelevel);
116  file_sink->set_pattern(prefix() + "[%L] %v");
117  sinks.push_back(file_sink);
118  }
119 
120  if (filelevel == Logger::Level::off && clilevel == Logger::Level::off) {
121  ret = spdlog::create<spdlog::sinks::null_sink_st>(name);
122  } else {
123  ret = std::make_shared<spdlog::logger>(name, begin(sinks), end(sinks));
124  }
125 
126  return ret;
127 }
