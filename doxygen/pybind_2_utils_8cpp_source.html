15 std::string getAttributeList(CPS::IdentifiedObject &obj) {
16  std::stringstream output;
17  output << std::setiosflags(std::ios::left);
18  output << "Attribute list for object with name " << obj.name() << ":" << std::endl;
19  output << std::setw(30) << "name" << std::setw(20) << "type" << std::setw(15) << "size" << std::setw(30) << "value" << std::endl;
20  output << std::setw(95) << std::setfill('-') << "" << std::setfill(' ') << std::endl;
21  for (auto attr : obj.attributes()) {
22  std::string name = attr.first;
23  std::string type;
24  std::string value;
25  std::string size;
26  if (auto tryReal = std::dynamic_pointer_cast<CPS::Attribute<CPS::Real>>(attr.second)) {
27  type = "Real";
28  value = std::to_string(tryReal->get());
29  size = "";
30  } else if (auto tryComplex = std::dynamic_pointer_cast<CPS::Attribute<CPS::Complex>>(attr.second)) {
31  type = "Complex";
32  value = std::to_string(abs(tryComplex->get())) + "<" + std::to_string(arg(tryComplex->get()));
33  size = "";
34  } else if (auto tryMatrixReal = std::dynamic_pointer_cast<CPS::Attribute<CPS::Matrix>>(attr.second)) {
35  type = "MatrixReal";
36  value = "[...]";
37  size = std::to_string(tryMatrixReal->get().rows()) + "x" + std::to_string(tryMatrixReal->get().cols());
38  } else if (auto tryMatrixComp = std::dynamic_pointer_cast<CPS::Attribute<CPS::MatrixComp>>(attr.second)) {
39  type = "MatrixComplex";
40  value = "[...]";
41  size = std::to_string(tryMatrixComp->get().rows()) + "x" + std::to_string(tryMatrixComp->get().cols());
42  } else if (auto tryString = std::dynamic_pointer_cast<CPS::Attribute<CPS::String>>(attr.second)) {
43  type = "String";
44  value = "\"" + tryString->get() + "\"";
45  size = "";
46  } else {
47  type = "Unknown";
48  value = "Unknown";
49  size = "Unknown";
50  }
51  output << std::setw(30) << name << std::setw(20) << type << std::setw(15) << size << std::setw(30) << value << std::endl;
52  }
53  return output.str();
54 }

60 void printAttribute(CPS::IdentifiedObject &obj, std::string attrName) {
61  auto attr = obj.attribute(attrName);
62  if (auto tryReal = std::dynamic_pointer_cast<CPS::Attribute<CPS::Real>>(attr)) {
63  py::print("Attribute " + attrName + " on object " + obj.name() + " has type Real and value " + std::to_string(tryReal->get()));
64  } else if (auto tryComplex = std::dynamic_pointer_cast<CPS::Attribute<CPS::Complex>>(attr)) {
65  std::string value = std::to_string(abs(tryComplex->get())) + "<" + std::to_string(arg(tryComplex->get()));
66  py::print("Attribute " + attrName + " on object " + obj.name() + " has type Complex and value " + value);
67  } else if (auto tryMatrixReal = std::dynamic_pointer_cast<CPS::Attribute<CPS::Matrix>>(attr)) {
68  std::string size = std::to_string(tryMatrixReal->get().rows()) + "x" + std::to_string(tryMatrixReal->get().cols());
69  py::print("Attribute " + attrName + " on object " + obj.name() + " has type MatrixReal (size " + size + ") and value:");
70  py::print(py::cast(tryMatrixReal->get()));
71  } else if (auto tryMatrixComp = std::dynamic_pointer_cast<CPS::Attribute<CPS::MatrixComp>>(attr)) {
72  std::string size = std::to_string(tryMatrixComp->get().rows()) + "x" + std::to_string(tryMatrixComp->get().cols());
73  py::print("Attribute " + attrName + " on object " + obj.name() + " has type MatrixComplex (size " + size + ") and value:");
74  py::print(py::cast(tryMatrixComp->get()));
75  } else if (auto tryString = std::dynamic_pointer_cast<CPS::Attribute<CPS::String>>(attr)) {
76  py::print("Attribute " + attrName + " on object " + obj.name() + " has type String and value \"" + tryString->get() + "\"");
77  } else {
78  py::print("Could not determine type of attribute " + attrName);
79  }
80 }
